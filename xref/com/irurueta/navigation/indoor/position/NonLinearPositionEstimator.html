33  /**
34   * Estimates position non-linearly using located radio sources and their readings at unknown
35   * locations.
36   * These kind of estimators can be used to determine the position of a given device by
37   * getting readings at an unknown location of different radio source whose locations are
38   * known.
39   * @param <P> a {@link Point} type.
40   */
41  @SuppressWarnings({"WeakerAccess", "Duplicates"})
42  public abstract class NonLinearPositionEstimator<P extends Point> extends PositionEstimator<P> {
43  
44      /**
45       * Minimum required number of sources to solve trilateration.
46       */
47      public static final int MIN_SOURCES = NonLinearLeastSquaresTrilaterationSolver.MIN_POINTS;
48  
49      /**
50       * Distance standard deviation assumed for provided distances as a fallback when
51       * none can be determined.
52       */
53      public static final double FALLBACK_DISTANCE_STANDARD_DEVIATION =
54              NonLinearLeastSquaresTrilaterationSolver.DEFAULT_DISTANCE_STANDARD_DEVIATION;
55  
56      /**
57       * A non-linear trilateration solver to solve position.
58       */
59      protected NonLinearLeastSquaresTrilaterationSolver<P> mTrilaterationSolver;
60  
61      /**
62       * Listener for the trilateration solver.
63       */
64      protected TrilaterationSolverListener<P> mTrilaterationSolverListener;
65  
66      /**
67       * Initial position to start position estimation.
68       * If not defined, centroid of provided located sources will be used.
69       */
70      private P mInitialPosition;
71  
72      /**
73       * Indicates whether located radio source position covariance must be taken
74       * into account (if available) to determine distance standard deviation.
75       */
76      private boolean mUseRadioSourcePositionCovariance;
77  
78      /**
79       * Distance standard deviation fallback value to use when none can be
80       * determined from provided radio sources and fingerprint readings.
81       */
82      private double mFallbackDistanceStandardDeviation =
83              FALLBACK_DISTANCE_STANDARD_DEVIATION;
84  
85      /**
86       * Constructor.
87       */
88      public NonLinearPositionEstimator() {
89          super();
90          init();
91      }
92  
93      /**
94       * Constructor.
95       * @param listener listener in charge of handling events.
96       */
97      public NonLinearPositionEstimator(PositionEstimatorListener<P> listener) {
98          super(listener);
99          init();
100     }
101 
102     /**
103      * Constructor.
104      * @param initialPosition initial position to start position estimation.
105      */
106     public NonLinearPositionEstimator(P initialPosition) {
107         this();
108         mInitialPosition = initialPosition;
109     }
110 
111     /**
112      * Constructor.
113      * @param initialPosition initial position to start position estimation.
114      * @param listener listener in charge of handling events.
115      */
116     public NonLinearPositionEstimator(P initialPosition,
117             PositionEstimatorListener<P> listener) {
118         this(listener);
119         mInitialPosition = initialPosition;
120     }

122     /**
123      * Gets initial position to start position estimation.
124      * If not defined, centroid of located sources position will be used to start
125      * the estimation.
126      * @return initial position to start position estimation.
127      */
128     public P getInitialPosition() {
129         return mInitialPosition;
130     }
131 
132     /**
133      * Sets initial position to start position estimation.
134      * If not defined, centroid of located sources position will be used to start
135      * the estimation.
136      * @param initialPosition initial position to start position estimation.
137      * @throws LockedException if estimator is locked.
138      */
139     public void setInitialPosition(P initialPosition) throws LockedException {
140         if (isLocked()) {
141             throw new LockedException();
142         }
143         mInitialPosition = initialPosition;
144     }
145 
146     /**
147      * Indicates whether located radio source position covariance must be taken into
148      * account (if available) to determine distance standard deviation.
149      * @return true to take radio source position covariance into account, false
150      * otherwise.
151      */
152     public boolean isRadioSourcePositionCovarianceUsed() {
153         return mUseRadioSourcePositionCovariance;
154     }
155 
156     /**
157      * Specifies whether located radio source position covariance must be taken into
158      * account (if available) to determine distance standard deviation.
159      * @param useRadioSourcePositionCovariance true to take radio source position
160      *                                         covariance into account, false otherwise.
161      * @throws LockedException if estimator is locked.
162      */
163     public void setRadioSourcePositionCovarianceUsed(
164             boolean useRadioSourcePositionCovariance) throws LockedException {
165         if (isLocked()) {
166             throw new LockedException();
167         }
168         mUseRadioSourcePositionCovariance = useRadioSourcePositionCovariance;
169     }
170 
171     /**
172      * Gets distance standard deviation fallback value to use when none can be
173      * determined from provided radio sources and fingerprint readings.
174      * @return distance standard deviation to use as fallback.
175      */
176     public double getFallbackDistanceStandardDeviation() {
177         return mFallbackDistanceStandardDeviation;
178     }
179 
180     /**
181      * Sets distance standard deviation fallback value to use when none can be
182      * determined from provided radio sources and fingerprint readings.
183      * @param fallbackDistanceStandardDeviation distance standard deviation to use
184      *                                          as fallback.
185      * @throws LockedException if estimator is locked.
186      */
187     public void setFallbackDistanceStandardDeviation(
188             double fallbackDistanceStandardDeviation) throws LockedException {
189         if (isLocked()) {
190             throw new LockedException();
191         }
192         mFallbackDistanceStandardDeviation = fallbackDistanceStandardDeviation;
193     }
194 
195     /**
196      * Gets minimum required number of located radio sources to perform trilateration.
197      * @return minimum required number of located radio sources to perform trilateration.
198      */
199     @Override
200     public int getMinRequiredSources() {
201         return MIN_SOURCES;
202     }
203 
204     /**
205      * Indicates whether estimator is ready to find a solution.
206      * @return true if estimator is ready, false otherwise.
207      */
208     @Override
209     public boolean isReady() {
210         return mTrilaterationSolver.isReady();
211     }
212 
213     /**
214      * Returns boolean indicating whether this estimator is locked because an estimation is already in progress.
215      * @return true if estimator is locked, false otherwise.
216      */
217     @Override
218     public boolean isLocked() {
219         return mTrilaterationSolver.isLocked();
220     }
221 
222     /**
223      * Gets standard deviations of distances from known located radio sources to
224      * the location of provided readings in a fingerprint.
225      * Distance standard deviations are used internally to solve trilateration.
226      * @return standard deviations used internally.
227      */
228     public double[] getDistanceStandardDeviations() {
229         return mTrilaterationSolver.getDistanceStandardDeviations();
230     }
231 
232     /**
233      * Estimates position based on provided located radio sources and readings of such radio sources at
234      * an unknown location.
235      * @throws LockedException if estimator is locked.
236      * @throws NotReadyException if estimator is not ready.
237      * @throws PositionEstimationException if estimation fails for some other reason.
238      */
239     @Override
240     public void estimate() throws LockedException, NotReadyException,
241             PositionEstimationException {
242         try {
243             mTrilaterationSolver.setInitialPosition(mInitialPosition);
244 
245             mTrilaterationSolver.solve();
246             mEstimatedPositionCoordinates =
247                     mTrilaterationSolver.getEstimatedPositionCoordinates();
248         } catch (TrilaterationException e) {
249             throw new PositionEstimationException(e);
250         }
251     }
252 
253     /**
254      * Gets known positions of radio sources used internally to solve trilateration.
255      * @return known positions used internally.
256      */
257     @Override
258     public P[] getPositions() {
259         return mTrilaterationSolver.getPositions();
260     }
261 
262     /**
263      * Gets euclidean distances from known located radio sources to
264      * the location of provided readings in a fingerprint.
265      * Distance values are used internally to solve trilateration.
266      * @return euclidean distances used internally.
267      */
268     @Override
269     public double[] getDistances() {
270         return mTrilaterationSolver.getDistances();
271     }
272 
273     /**
274      * Internally sets located radio sources used for trilateration.
275      * @param sources located radio sources used for trilateration.
276      * @throws IllegalArgumentException if provided value is null or the number of provided sources is less
277      * than the required minimum.
278      */
279     protected void internalSetSources(List<? extends RadioSourceLocated<P>> sources) {
280         super.internalSetSources(sources);
281         buildPositionsDistancesAndDistanceStandardDeviations();
282     }
283 
284     /**
285      * Internally sets fingerprint containing readings at an unknown location for provided located radio sources.
286      * @param fingerprint fingerprint containing readings at an unknown location for provided located radio sources.
287      * @throws IllegalArgumentException if provided value is null.
288      */
289     protected void internalSetFingerprint(
290             Fingerprint<? extends RadioSource, ? extends Reading<? extends RadioSource>> fingerprint) {
291         super.internalSetFingerprint(fingerprint);
292         buildPositionsDistancesAndDistanceStandardDeviations();
293     }
294 
295     /**
296      * Sets positions, distances and standard deviations of distances on internal trilateration solver.
297      * @param positions positions to be set.
298      * @param distances distances to be set.
299      * @param distanceStandardDeviations standard deviations of distances to be set.
300      */
301     protected abstract void setPositionsDistancesAndDistanceStandardDeviations(List<P> positions,
302             List<Double> distances, List<Double> distanceStandardDeviations);
303 
304     /**
305      * Initializes trilateration solver listener.
306      */
307     private void init() {
308         mTrilaterationSolverListener = new TrilaterationSolverListener<P>() {
309             @Override
310             public void onSolveStart(TrilaterationSolver<P> solver) {
311                 if (mListener != null) {
312                     mListener.onEstimateStart(NonLinearPositionEstimator.this);
313                 }
314             }
315 
316             @Override
317             public void onSolveEnd(TrilaterationSolver<P> solver) {
318                 if (mListener != null) {
319                     mListener.onEstimateEnd(NonLinearPositionEstimator.this);
320                 }
321             }
322         };
323     }
324 
325     /**
326      * Builds positions, distances and standard deviation of distances for the internal
327      * trilateration solver.
328      */
329     private void buildPositionsDistancesAndDistanceStandardDeviations() {
330         if (mTrilaterationSolver == null) {
331             return;
332         }
333 
334         int min = getMinRequiredSources();
335         if (mSources == null || mFingerprint == null ||
336                 mSources.size() < min ||
337                 mFingerprint.getReadings() == null ||
338                 mFingerprint.getReadings().size() < min) {
339             return;
340         }
341 
342         List<P> positions = new ArrayList<>();
343         List<Double> distances = new ArrayList<>();
344         List<Double> distanceStandardDeviations = new ArrayList<>();
345         PositionEstimatorHelper.buildPositionsDistancesAndDistanceStandardDeviations(
346                 mSources, mFingerprint, mUseRadioSourcePositionCovariance,
347                 mFallbackDistanceStandardDeviation, positions, distances,
348                 distanceStandardDeviations);
349 
350         setPositionsDistancesAndDistanceStandardDeviations(positions, distances,
351                 distanceStandardDeviations);
352     }
