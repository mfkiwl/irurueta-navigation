23  /**
24   * Evenly sorts readings of a fingerprint among different radio sources taking into
25   * account their respective quality scores as well.
26   *
27   * @param <P> a {@link Point} type.
28   * @param <R> a {@link Reading} type.
29   */
30  public class ReadingSorter<P extends Point<?>, R extends Reading<? extends RadioSource>> {
31  
32      /**
33       * Sources to be taken into account within readings.
34       */
35      private List<? extends RadioSourceLocated<P>> mSources;
36  
37      /**
38       * Fingerprint containing readings of different sources.
39       */
40      private Fingerprint<? extends RadioSource, ? extends R> mFingerprint;
41  
42      /**
43       * Quality scores associated to each radio source.
44       */
45      private double[] mSourceQualityScores;
46  
47      /**
48       * Quality scores associated to each reading within the fingerprint.
49       */
50      private double[] mFingerprintReadingsQualityScores;
51  
52      /**
53       * Contains sorted sources and readings.
54       */
55      private List<RadioSourceSourceWithQualityScore<P, R>> mSortedSourcesAndReadings;
56  
57      /**
58       * Constructor.
59       *
60       * @param sources                           sources to take into account within
61       *                                          readings.
62       * @param fingerprint                       fingerprint containing readings of
63       *                                          different sources.
64       * @param sourceQualityScores               quality scores associated to each radio
65       *                                          source.
66       * @param fingerprintReadingsQualityScores  quality scores associated to each reading
67       *                                          within the fingerprint.
68       * @throws IllegalArgumentException if number of source quality scores is not equal
69       *                                  to the number of sources, or if number of
70       *                                  fingerprint reading quality scores is not equal
71       *                                  to the number of readings within fingerprint.
72       */
73      ReadingSorter(List<? extends RadioSourceLocated<P>> sources,
74                           Fingerprint<? extends RadioSource, ? extends R> fingerprint,
75                           double[] sourceQualityScores,
76                           double[] fingerprintReadingsQualityScores) {
77          if (sources.size() != sourceQualityScores.length) {
78              throw new IllegalArgumentException();
79          }
80  
81          if (fingerprint.getReadings().size() != fingerprintReadingsQualityScores.length) {
82              throw new IllegalArgumentException();
83          }
84  
85          mSources = sources;
86          mFingerprint = fingerprint;
87          mSourceQualityScores = sourceQualityScores;
88          mFingerprintReadingsQualityScores = fingerprintReadingsQualityScores;
89      }
90  
91      /**
92       * Gets sources to be taken into account within readings.
93       *
94       * @return sources to be taken into account within readings.
95       */
96      public List<? extends RadioSourceLocated<P>> getSources() {
97          return mSources;
98      }
99  
100     /**
101      * Gets fingerprint containing readings of different sources.
102      *
103      * @return fingerprint containing readings of different sources.
104      */
105     public Fingerprint<? extends RadioSource, ? extends R> getFingerprint() {
106         return mFingerprint;
107     }
108 
109     /**
110      * Gets quality scores associated to each radio source.
111      *
112      * @return quality scores associated to each radio source.
113      */
114     double[] getSourceQualityScores() {
115         return mSourceQualityScores;
116     }
117 
118     /**
119      * Gets quality scores associated to each reading within the fingerprint.
120      *
121      * @return quality scores associated to each reading within the fingerprint.
122      */
123     double[] getFingerprintReadingsQualityScores() {
124         return mFingerprintReadingsQualityScores;
125     }
126 
127     /**
128      * Sorts readings with quality scores.
129      */
130     void sort() {
131 
132         Map<RadioSourceLocated<P>, RadioSourceSourceWithQualityScore<P, R>> sourcesMap = new HashMap<>();
133 
134         // build sources
135         List<RadioSourceSourceWithQualityScore<P, R>> sourcesWithQualityScores = new ArrayList<>();
136         int sourcePosition = 0;
137         for(RadioSourceLocated<P> source : mSources) {
138             RadioSourceSourceWithQualityScore<P, R> sourceWithQualityScore =
139                     new RadioSourceSourceWithQualityScore<>();
140             sourceWithQualityScore.source = source;
141             sourceWithQualityScore.qualityScore = mSourceQualityScores[sourcePosition];
142             sourceWithQualityScore.position = sourcePosition;
143             sourceWithQualityScore.readingsWithQualityScores = new ArrayList<>();
144 
145             sourcesWithQualityScores.add(sourceWithQualityScore);
146 
147             sourcesMap.put(source, sourceWithQualityScore);
148 
149             sourcePosition++;
150         }
151 
152         // build readings
153         int readingPosition = 0;
154         for (@SuppressWarnings("unchecked") R reading : mFingerprint.getReadings()) {
155             // noinspection all
156             RadioSourceSourceWithQualityScore<P, R> sourceWithQualityScore =
157                     sourcesMap.get(reading.getSource());
158             if (sourceWithQualityScore == null) {
159                 continue;
160             }
161 
162             List<ReadingWithQualityScore<R>> readingsWithQualityScores =
163                     sourceWithQualityScore.readingsWithQualityScores;
164 
165             ReadingWithQualityScore<R> readingWithQualityScore = new ReadingWithQualityScore<>();
166             readingWithQualityScore.reading = reading;
167             readingWithQualityScore.qualityScore = mFingerprintReadingsQualityScores[readingPosition];
168             readingWithQualityScore.position = readingPosition;
169 
170             readingsWithQualityScores.add(readingWithQualityScore);
171 
172             readingPosition++;
173         }
174 
175         // sort all readings within sources
176         for (RadioSourceSourceWithQualityScore<P, R> sourceWithQualityScore : sourcesWithQualityScores) {
177             // sort all readings for this source from highest to lowest quality
178 
179             // noinspection unchecked
180             ReadingWithQualityScore<R>[] readingsWithQualityScoresArray =
181                     new ReadingWithQualityScore[sourceWithQualityScore.readingsWithQualityScores.size()];
182             sourceWithQualityScore.readingsWithQualityScores.toArray(readingsWithQualityScoresArray);
183             Arrays.sort(readingsWithQualityScoresArray, new ReadingComparator<R>());
184 
185             sourceWithQualityScore.readingsWithQualityScores = Arrays.asList(
186                     readingsWithQualityScoresArray);
187         }
188 
189         // sort all sources from highest to lowest quality
190 
191         // noinspection unchecked
192         RadioSourceSourceWithQualityScore<P, R>[] sourcesWithQualityScoresArray =
193                 new RadioSourceSourceWithQualityScore[sourcesWithQualityScores.size()];
194         sourcesWithQualityScores.toArray(sourcesWithQualityScoresArray);
195         Arrays.sort(sourcesWithQualityScoresArray, new RadioSourceComparator<P, R>());
196 
197         mSortedSourcesAndReadings = Arrays.asList(sourcesWithQualityScoresArray);
198     }
199 
200     /**
201      * Gets sorted sources and readings.
202      *
203      * @return sorted sources and readings.
204      */
205     List<RadioSourceSourceWithQualityScore<P, R>> getSortedSourcesAndReadings() {
206         return mSortedSourcesAndReadings;
207     }

232      * Contains a radio source with its associated quality score.
233      *

235      * @param <R> a {@link Reading} type.
236      */
237     static class RadioSourceSourceWithQualityScore<P extends Point<?>,
238             R extends Reading<? extends RadioSource>> {
239         /**
240          * Radio source.
241          */
242         RadioSourceLocated<P> source;

246          */

257     /**
258      * Comparator to order readings based on their quality scores.
259      *
260      * @param <R> a {@link Reading} type.
261      */
262     private static class ReadingComparator<R extends Reading<? extends RadioSource>> implements
263             Comparator<ReadingWithQualityScore<R>> {
264 
265         /**
266          * Orders readings so that readings with higher scores go first.
267          *
268          * @param o1 1st item to be compared.
269          * @param o2 2nd item to be compared.
270          * @return -1 if first item goes first, 0 if both are equal and 1 if first item
271          * goes second.
272          */
273         @Override
274         public int compare(ReadingWithQualityScore<R> o1, ReadingWithQualityScore<R> o2) {
275 
276             // Take reading type into account so that ranging readings go first,
277             // then ranging+RSSI and finally RSSI readings.
278             if (o1.reading.getType() != o2.reading.getType()) {
279                 switch (o1.reading.getType()) {
280                     case RANGING_READING:
281                         // o1 goes first (o2 last)
282                         return  - 1;
283 
284                     case RANGING_AND_RSSI_READING:
285                         switch (o2.reading.getType()) {
286                             case RANGING_READING:
287                                 // o2 goes first (o1 last)
288                                 return 1;
289                             case RSSI_READING:
290                                 // o1 goes first (o2 last)
291                                 return -1;
292                         }
293                         break;
294                     case RSSI_READING:
295                         // o2 goes first (o1 last)
296                         return 1;
297                 }
298             }
299 
300 
301             // same reading type
302 
303             // order by reading quality score
304             return Double.compare(o2.qualityScore, o1.qualityScore);
305         }
306     }
307 
308     /**
309      * Comparator to order radio sources based on their quality scores.
310      *
311      * @param <P> a {@link Point} type.
312      * @param <R> a {@link Reading} type.
313      */
314     private static class RadioSourceComparator<P extends Point<?>,
315             R extends Reading<? extends RadioSource>> implements
316             Comparator<RadioSourceSourceWithQualityScore<P, R>> {
317 
318         /**
319          * Orders radio sources so that radio sources with higher scores go first.
320          *
321          * @param o1 1st item to be compared.
322          * @param o2 2nd item to be compared.
323          * @return -1 if first item goes first, 0 if both are equal and 1 if first item
324          * goes second.
325          */
326         @Override
327         public int compare(RadioSourceSourceWithQualityScore<P, R> o1,
328                            RadioSourceSourceWithQualityScore<P, R> o2) {
329             return Double.compare(o2.qualityScore, o1.qualityScore);
330         }
331     }
