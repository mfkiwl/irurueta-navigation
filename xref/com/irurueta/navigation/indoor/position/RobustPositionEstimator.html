35  /**
36   * Base class for robust position estimators using located radio sources and their
37   * readings at unknown locations.
38   * These kind of estimators can be used to robustly determine the position of a given
39   * device by getting readings at an unknown location of different radio sources whose
40   * locations are known.
41   * Implementations of this class should be able to detect and discard outliers in order
42   * to find the best solution.
43   * @param <P> a {@link Point} type.
44   */
45  @SuppressWarnings({"WeakerAccess", "Duplicates"})
46  public abstract class RobustPositionEstimator<P extends Point> {

592     /**
593      * Initializes robust trilateration solver listener.
594      */
595     private void init() {
596         mTrilaterationSolverListener = new RobustTrilaterationSolverListener<P>() {
597             @Override
598             public void onSolveStart(RobustTrilaterationSolver<P> solver) {
599                 if (mListener != null) {
600                     mListener.onEstimateStart(RobustPositionEstimator.this);
601                 }
602             }
603 
604             @Override
605             public void onSolveEnd(RobustTrilaterationSolver<P> solver) {
606                 if (mListener != null) {
607                     mListener.onEstimateEnd(RobustPositionEstimator.this);
608                 }
609             }
610 
611             @Override
612             public void onSolveNextIteration(RobustTrilaterationSolver<P> solver,
613                     int iteration) {
614                 if (mListener != null) {
615                     mListener.onEstimateNextIteration(
616                             RobustPositionEstimator.this, iteration);
617                 }
618             }
619 
620             @Override
621             public void onSolveProgressChange(RobustTrilaterationSolver<P> solver,
622                     float progress) {
623                 if (mListener != null) {
624                     mListener.onEstimateProgressChange(
625                             RobustPositionEstimator.this, progress);
626                 }
627             }
628         };
629     }

631     /**
632      * Builds positions, distances, standard deviation of distances and quality scores
633      * for the internal trilateration solver.
634      */
635     protected void buildPositionsDistancesDistanceStandardDeviationsAndQualityScores() {
636         if (mTrilaterationSolver == null) {
637             return;
638         }
639 
640         int min = getMinRequiredSources();
641         if (mSources == null || mFingerprint == null ||
642                 mSources.size() < min ||
643                 mFingerprint.getReadings() == null ||
644                 mFingerprint.getReadings().size() < min) {
645             return;
646         }
647 
648         List<P> positions = new ArrayList<>();
649         List<Double> distances = new ArrayList<>();
650         List<Double> distanceStandardDeviations = new ArrayList<>();
651 
652         double[] qualityScores = getQualityScores();
653         List<Double> distanceQualityScores = null;
654         if (qualityScores != null) {
655             distanceQualityScores = new ArrayList<>();
656         }
657         PositionEstimatorHelper.buildPositionsDistancesDistanceStandardDeviationsAndQualityScores(
658                 mSources, mFingerprint, qualityScores,
659                 mUseRadioSourcePositionCovariance,
660                 mFallbackDistanceStandardDeviation, positions, distances,
661                 distanceStandardDeviations, distanceQualityScores);
662 
663         setPositionsDistancesDistanceStandardDeviationsAndQualityScores(positions, distances,
664                 distanceStandardDeviations, distanceQualityScores);
665     }
