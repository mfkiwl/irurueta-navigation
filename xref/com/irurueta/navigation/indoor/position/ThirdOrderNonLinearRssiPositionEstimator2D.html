23  /**
24   * 2D position estimator based on located fingerprints containing only RSSI readings and
25   * having as well prior knowledge of the location of radio sources associated to those
26   * readings.
27   * This implementation uses a third-order Taylor approximation over provided located
28   * fingerprints to determine an approximate position for a non-located fingerprint using
29   * a non-linear solving algorithm.
30   * An initial position can be provided as a starting point to solve the position,
31   * otherwise the average point of selected nearest fingerprints is used as a starting
32   * point.
33   */
34  @SuppressWarnings("WeakerAccess")
35  public class ThirdOrderNonLinearRssiPositionEstimator2D extends
36          NonLinearRssiPositionEstimator2D {
37

39       * Constructor.
40       */
41      public ThirdOrderNonLinearRssiPositionEstimator2D() { }

46       */
47      public ThirdOrderNonLinearRssiPositionEstimator2D(
48              SourcedRssiPositionEstimatorListener<Point2D> listener) {
49          super(listener);
50      }

155     /**
156      * Evaluates a non-linear multi dimension function at provided point using
157      * provided parameters and returns its evaluation and derivatives of the
158      * function respect the function parameters.
159      * @param i number of sample being evaluated.
160      * @param point point where function will be evaluated.
161      * @param params initial parameters estimation to be tried. These will
162      * change as the Levenberg-Marquard algorithm iterates to the best solution.
163      * These are used as input parameters along with point to evaluate function.
164      * @param derivatives partial derivatives of the function respect to each
165      * provided parameter.
166      * @return function evaluation at provided point.
167      */
168     @Override
169     @SuppressWarnings("Duplicates")
170     protected double evaluate(int i, double[] point, double[] params, double[] derivatives) {
171         //Demonstration in 2D:
172         //--------------------
173         //Taylor series expansion can be expressed as:
174         //f(x) = f(a) + 1/1!*f'(a)*(x - a) + 1/2!*f''(a)*(x - a)^2 + 1/3!*f'''(a)*(x - a)^3 ...
175 
176         //where f'(x) is the derivative of f respect x, which can also be expressed as:
177         //f'(x) = diff(f(x))/diff(x)
178 
179         //and f'(a) is the derivative of f respect x evaluated at a, which can be expressed
180         //as f'(a) = diff(f(a))/diff(x)
181 
182         //consequently f''(a) is the second derivative respect x evaluated at a, which can
183         //be expressed as:
184         //f''(x) = diff(f(x))/diff(x^2)
185 
186         //and:
187         //f''(a) = diff(f(a))/diff(x^2)
188 
189         //and finally f'''(a) is the third derivative respect x evaluated at a, which can
190         //be expressed as:
191         //f'''(x) = diff(f(x))/diff(x^3)
192 
193         //and:
194         //f'''(a) = diff(f(a))/diff(x^3)
195 
196         //Received power expressed in dBm is:
197         //k = (c/(4*pi*f))
198         //Pr = Pte*k^n / d^n
199 
200         //where c is the speed of light, pi is 3.14159..., f is the frequency of the radio source,
201         //Pte is the equivalent transmitted power by the radio source, n is the path-loss exponent
202         // (typically 2.0), and d is the distance from a point to the location of the radio source.
203 
204         //Hence:
205         //Pr(dBm) = 10*log(Pte*k^n/d^n) = 10*n*log(k) + 10*log(Pte) - 10*n*log(d) =
206         //          10*n*log(k) + 10*log(Pte) - 5*n*log(d^2)
207 
208         //The former 2 terms are constant, and only the last term depends on distance
209 
210         //Hence, assuming the constant K = 10*n*log(k) + Pte(dBm), where Pte(dBm) = 10*log(Pte),
211         //assuming that transmitted power by the radio source Pte is known (so that K is also known),
212         //and assuming that the location of the radio source is known and it is located at pa = (xa, ya)
213         //so that d^2 = (x - xa)^2 + (y - ya)^2 then the received power at an unknown point pi = (xi, yi) is:
214 
215         //Pr(pi) = Pr(xi,yi) = K - 5*n*log(d^2) = K - 5*n*log((xi - xa)^2 + (yi - ya)^2)
216 
217         //Suppose that received power at point p1=(x1,y1) is known on a located fingerprint
218         //containing readings Pr(p1).
219 
220         //Then, for an unknown point pi=(xi,yi) close to fingerprint 1 located at p1 where we
221         //have measured received power Pr(pi), we can get the following third-order Taylor
222         //approximation:
223 
224         //Pr(pi = (xi,yi)) = Pr(p1) +
225         //  diff(Pr(p1))/diff(x)*(xi - x1) +
226         //  diff(Pr(p1))/diff(y)*(yi - y1) +
227         //  1/2*(diff(Pr(p1))/diff(x^2)*(xi - x1)^2 +
228         //      diff(Pr(p1))/diff(y^2)*(yi - y1)^2 +
229         //      2*diff(Pr(p1))/diff(x*y)*(xi - x1)*(yi - y1)) +
230         //  1/6*(diff(Pr(p1))/diff(x^3)*(xi - x1)^3 +
231         //      diff(Pr(p1))/diff(y^3)*(yi - y1)^3 +
232         //      3*diff(Pr(p1))/diff(x^2*y)*(xi - x1)^2*(yi - y1) +
233         //      3*diff(Pr(p1))/diff(x*y^2)*(xi - x1)*(yi - y1)^2
234 
235         //where the first order derivatives of Pr(p = (x,y)) are:
236         //diff(Pr(x,y))/diff(x) = -5*n/(ln(10)*((x - xa)^2 + (y - ya)^2)*2*(x - xa)
237         //diff(Pr(x,y))/diff(x) = -10*n*(x - xa)/(ln(10)*((x - xa)^2 + (y - ya)^2))
238 
239         //diff(Pr(x,y))/diff(y) = -5*n/(ln(10)*((x - xa)^2 + (y - ya)^2)*2*(y - ya)
240         //diff(Pr(x,y))/diff(y) = -10*n*(y - ya)/(ln(10)*((x - xa)^2 + (y - ya)^2))
241 
242         //If we evaluate first order derivatives at p1 = (x1,y1), we get:
243         //diff(Pr(p1))/diff(x) = -10*n*(x1 - xa)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2))
244         //diff(Pr(p1))/diff(y) = -10*n*(y1 - ya)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2))
245 
246         //where square distance from fingerprint 1 to radio source a can be expressed as:
247         //d1a^2 = (x1 - xa)^2 + (y1 - ya)^2
248 
249         //where both the fingerprint and radio source positions are known, and hence d1a is known.
250 
251         //Then first order derivatives can be expressed as:
252         //diff(Pr(p1))/diff(x) = -10*n*(x1 - xa)/(ln(10)*d1a^2)
253         //diff(Pr(p1))/diff(y) = -10*n*(y1 - ya)/(ln(10)*d1a^2)
254 
255         //To obtain second order derivatives we take into account that:
256         //(f(x)/g(x))' = (f'(x)*g(x) - f(x)*g'(x))/g(x)^2
257 
258         //hence, second order derivatives of Pr(p = (x,y)) are:
259         //diff(Pr(x,y))/diff(x^2) = -10*n/ln(10)*(1*((x - xa)^2 + (y - ya)^2) - (x - xa)*2*(x - xa)) / ((x - xa)^2 + (y - ya)^2)^2
260         //diff(Pr(x,y))/diff(x^2) = -10*n*((y - ya)^2 - (x - xa)^2)/(ln(10)*((x - xa)^2 + (y - ya)^2)^2)
261 
262         //diff(Pr(x,y))/diff(y^2) = -10*n/ln(10)*(1*((x - xa)^2 + (y - ya)^2) - (y - ya)*2*(y - ya)) / ((x - xa)^2 + (y - ya)^2)^2
263         //diff(Pr(x,y))/diff(y^2) = -10*n*((x - xa)^2 - (y - ya)^2)/(ln(10)*((x - xa)^2 + (y - ya)^2)^2)
264 
265         //diff(Pr(x,y))/diff(x*y) = -10*n/ln(10)*(0*((x - xa)^2 + (y - ya)^2) - (x - xa)*2*(y - ya))/((x - xa)^2 + (y - ya)^2)^2
266         //diff(Pr(x,y))/diff(x*y) = 20*n*((x - xa)*(y - ya))/(ln(10)*((x - xa)^2 + (y - ya)^2)^2)
267 
268         //If we evaluate second order derivatives at p1 = (x1,y1), we get:
269         //diff(Pr(p1))/diff(x^2) = -10*n*((y1 - ya)^2 - (x1 - xa)^2)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2)^2)
270         //diff(Pr(p1))/diff(y^2) = -10*n*((x1 - xa)^2 - (y1 - ya)^2)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2)^2)
271         //diff(Pr(p1))/diff(x*y) = 20*n*(x1 - xa)*(y1 - ya)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2)^2)
272 
273         //and expressing the second order derivatives in terms of distance between
274         //fingerprint 1 and radio source a d1a, we get:
275         //diff(Pr(p1))/diff(x^2) = -10*n*((y1 - ya)^2 - (x1 - xa)^2)/(ln(10)*d1a^4)
276         //diff(Pr(p1))/diff(y^2) = -10*n*((x1 - xa)^2 - (y1 - ya)^2)/(ln(10)*d1a^4)
277         //diff(Pr(p1))/diff(x*y) = 20*n*(x1 - xa)*(y1 - ya)/(ln(10)*d1a^4)
278 
279         //Finally, third order derivatives of Pr(p = (x,y)) are:
280         //diff(Pr(x,y))/diff(x^3) = -10*n/ln(10)*(-2*(x - xa)*((x - xa)^2 + (y - ya)^2)^2 - ((y - ya)^2 - (x - xa)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(x - xa))/((x - xa)^2 + (y - ya)^2)^4
281         //diff(Pr(x,y))/diff(y^3) = -10*n/ln(10)*(-2*(y - ya)*((x - xa)^2 + (y - ya)^2)^2 - ((x - xa)^2 - (y - ya)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(y - ya))/((x - xa)^2 + (y - ya)^2)^4
282         //diff(Pr(x,y)/diff(x^2*y) = -10*n/ln(10)*(2*(y - ya)*((x - xa)^2 + (y - ya)^2)^2 - ((y - ya)^2 - (x - xa)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(y - ya))/((x - xa)^2 + (y - ya)^2)^4
283         //diff(Pr(x,y)/diff(x*y^2) = -10*n/ln(10)*(2*(x - xa)*((x - xa)^2 + (y - ya)^2)^2 - ((x - xa)^2 - (y - ya)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(x - xa))/((x - xa)^2 + (y - ya)^2)^4
284 
285         //evaluating at p1 = (x1, y1), we get:
286         //diff(Pr(p1))/diff(x^3) = -10*n/ln(10)*(-2*(x1 - xa)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((y1 - ya)^2 - (x1 - xa)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(x1 - xa))/((x1 - xa)^2 + (y1 - ya)^2)^4
287         //diff(Pr(p1))/diff(y^3) = -10*n/ln(10)*(-2*(y1 - ya)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((x1 - xa)^2 - (y1 - ya)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(y1 - ya))/((x1 - xa)^2 + (y1 - ya)^2)^4
288         //diff(Pr(p1)/diff(x^2*y) = -10*n/ln(10)*(2*(y1 - ya)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((y1 - ya)^2 - (x1 - xa)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(y1 - ya))/((x1 - xa)^2 + (y1 - ya)^2)^4
289         //diff(Pr(p1)/diff(x*y^2) = -10*n/ln(10)*(2*(x1 - xa)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((x1 - xa)^2 - (y1 - ya)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(x1 - xa))/((x1 - xa)^2 + (y1 - ya)^2)^4
290 
291         //and substituting the distance between fingerprint and radio source d1a, we get:
292         //diff(Pr(p1))/diff(x^3) = -10*n/ln(10)*(-2*(x1 - xa)*dia^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(x1 - xa))/d1a^8
293         //diff(Pr(p1))/diff(y^3) = -10*n/ln(10)*(-2*(y1 - ya)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(y1 - ya))/d1a^8
294         //diff(Pr(p1)/diff(x^2*y) = -10*n/ln(10)*(2*(y1 - ya)*d1a^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(y1 - ya))/d1a^8
295         //diff(Pr(p1)/diff(x*y^2) = -10*n/ln(10)*(2*(x1 - xa)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(x1 - xa))/d1a^8
296 
297 
298         //Hence, the third order Taylor expansion can be expressed as:
299         //Pr(pi = (xi,yi)) = Pr(p1) +
300         //  diff(Pr(p1))/diff(x)*(xi - x1) +
301         //  diff(Pr(p1))/diff(y)*(yi - y1) +
302         //  1/2*(diff(Pr(p1))/diff(x^2)*(xi - x1)^2 +
303         //      diff(Pr(p1))/diff(y^2)*(yi - y1)^2 +
304         //      2*diff(Pr(p1))/diff(x*y)*(xi - x1)*(yi - y1)) +
305         //  1/6*(diff(Pr(p1))/diff(x^3)*(xi - x1)^3 +
306         //      diff(Pr(p1))/diff(y^3)*(yi - y1)^3 +
307         //      3*diff(Pr(p1))/diff(x^2*y)*(xi - x1)^2*(yi - y1) +
308         //      3*diff(Pr(p1))/diff(x*y^2)*(xi - x1)*(yi - y1)^2)
309 
310         //Pr(pi) = Pr(p1)
311         //  -10*n*(x1 - xa)/(ln(10)*d1a^2)*(xi - x1) +
312         //  -10*n*(y1 - ya)/(ln(10)*d1a^2)*(yi - y1) +
313         //  -5*n*((y1 - ya)^2 - (x1 - xa)^2)/(ln(10)*d1a^4)*(xi - x1)^2 +
314         //  -5*n*((x1 - xa)^2 - (y1 - ya)^2)/(ln(10)*d1a^4)*(yi - y1)^2 +
315         //  20*n*(x1 - xa)*(y1 - ya)/(ln(10)*d1a^4)*(xi - x1)*(yi - y1) +
316         //  -10/6*n/ln(10)*(-2*(x1 - xa)*dia^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(x1 - xa))/d1a^8*(xi - x1)^3 +
317         //  -10/6*n/ln(10)*(-2*(y1 - ya)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(y1 - ya))/d1a^8*(yi - y1)^3 +
318         //  -5*n/ln(10)*(2*(y1 - ya)*d1a^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(y1 - ya))/d1a^8*(xi - x1)^2*(yi - y1) +
319         //  -5*n/ln(10)*(2*(x1 - xa)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(x1 - xa))/d1a^8*(xi - x1)*(yi - y1)^2
320 
321 
322         //The equation above can be solved using a non-linear fitter such as Levenberg-Marquardt
323 
324         //This method implements received power at point pi = (xi, yi) and its derivatives
325 
326         double xi = params[0];
327         double yi = params[1];
328 
329         //received power
330         double pr = point[0];
331 
332         //fingerprint coordinates
333         double x1 = point[1];
334         double y1 = point[2];
335 
336         //radio source coordinates
337         double xa = point[3];
338         double ya = point[4];
339 
340         //path loss exponent
341         double n = point[5];
342 
343         double ln10 = Math.log(10.0);
344 
345         double diffXi1 = xi - x1;
346         double diffYi1 = yi - y1;
347 
348         double diffX1a = x1 - xa;
349         double diffY1a = y1 - ya;
350 
351         double diffXi12 = diffXi1 * diffXi1;
352         double diffYi12 = diffYi1 * diffYi1;
353 
354         double diffXi13 = diffXi12 * diffXi1;
355         double diffYi13 = diffYi12 * diffYi1;
356 
357         double diffX1a2 = diffX1a * diffX1a;
358         double diffY1a2 = diffY1a * diffY1a;
359 
360         double d1a2 = diffX1a2 + diffY1a2;
361         double d1a4 = d1a2 * d1a2;
362         double d1a8 = d1a4 * d1a4;
363 
364         double value1 = - 10.0 * n * diffX1a / (ln10 * d1a2);
365         double value2 = - 10.0 * n * diffY1a / (ln10 * d1a2);
366         double value3 = - 5.0 * n * (-diffX1a2 + diffY1a2) / (ln10 * d1a4);
367         double value4 = - 5.0 * n * (diffX1a2 - diffY1a2) / (ln10 * d1a4);
368         double value5 = 20.0 * n * diffX1a * diffY1a / (ln10 * d1a4);
369         double value6 = - 10.0 / 6.0 * n / ln10 * (-2.0 * diffX1a * d1a4 - (-diffX1a2 + diffY1a2) * 4.0 * d1a2 * diffX1a) / d1a8;
370         double value7 = - 10.0 / 6.0 * n / ln10 * (-2.0 * diffY1a * d1a4 - (diffX1a2 - diffY1a2) * 4.0 * d1a2 * diffY1a) / d1a8;
371         double value8 = - 5.0 * n / ln10 * (2.0 * diffY1a * d1a4 - (-diffX1a2 + diffY1a2) * 4.0 * d1a2 * diffY1a) / d1a8;
372         double value9 = - 5.0 * n / ln10 * (2.0 * diffX1a * d1a4 - (diffX1a2 - diffY1a2) * 4.0 * d1a2 * diffX1a) / d1a8;
373 
374         //hence:
375         //Pr(pi) = Pr(p1) +
376         //  value1*(xi - x1) +
377         //  value2*(yi - y1) +
378         //  value3*(xi - x1)^2 +
379         //  value4*(yi - y1)^2 +
380         //  value5*(xi - x1)*(yi - y1) +
381         //  value6*(xi - x1)^3 +
382         //  value7*(yi - y1)^3 +
383         //  value8*(xi - x1)^2*(yi - y1) +
384         //  value9*(xi - x1)*(yi - y1)^2 +
385 
386         double result = pr
387                 + value1 * diffXi1
388                 + value2 * diffYi1
389                 + value3 * diffXi12
390                 + value4 * diffYi12
391                 + value5 * diffXi1 * diffYi1
392                 + value6 * diffXi13
393                 + value7 * diffYi13
394                 + value8 * diffXi12 * diffYi1
395                 + value9 * diffXi1 * diffYi12;
396 
397         //derivative respect xi
398 
399         //diff(Pr(pi))/diff(xi) = value1 +
400         //  2*value3*(xi - x1) +
401         //  value5*(yi - y1) +
402         //  3*value6*(xi - x1)^2 +
403         //  2*value8*(xi - x1)*(yi - y1) +
404         //  value9*(yi - y1)^2
405 
406         derivatives[0] = value1 + 2.0 * value3 * diffXi1 + value5 * diffYi1 +
407                 3.0 * value6 * diffXi12 + 2.0 * value8 * diffXi1 * diffYi1 +
408                 value9 * diffYi12;
409 
410         //derivative respect yi
411 
412         //diff(Pr(pi))/diff(yi) = value2 +
413         //  2*value4*(yi - y1) +
414         //  value5*(xi - x1) +
415         //  3*value7*(yi - y1)^2 +
416         //  value8*(xi - x1)^2 +
417         //  2*value9*(xi - x1)*(yi - y1)
418 
419         derivatives[1] = value2 + 2.0 * value4 * diffYi1 + value5 * diffXi1 +
420                 3.0 * value7 * diffYi12 + value8 * diffXi12 +
421                 2.0 * value9 * diffXi1 * diffYi1;
422 
423         return result;
424     }
