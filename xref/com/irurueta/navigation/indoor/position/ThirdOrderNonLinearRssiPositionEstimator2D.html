25  /**
26   * 2D position estimator based on located fingerprints containing only RSSI readings and
27   * having as well prior knowledge of the location of radio sources associated to those
28   * readings.
29   * This implementation uses a third-order Taylor approximation over provided located
30   * fingerprints to determine an approximate position for a non-located fingerprint using
31   * a non-linear solving algorithm.
32   * An initial position can be provided as a starting point to solve the position,
33   * otherwise the average point of selected nearest fingerprints is used as a starting
34   * point.
35   */
36  @SuppressWarnings("WeakerAccess")
37  public class ThirdOrderNonLinearRssiPositionEstimator2D extends
38          NonLinearRssiPositionEstimator2D {
39

41       * Constructor.
42       */
43      public ThirdOrderNonLinearRssiPositionEstimator2D() { }

48       */
49      public ThirdOrderNonLinearRssiPositionEstimator2D(
50              SourcedRssiPositionEstimatorListener<Point2D> listener) {
51          super(listener);
52      }

157     /**
158      * Evaluates a non-linear multi dimension function at provided point using
159      * provided parameters and returns its evaluation and derivatives of the
160      * function respect the function parameters.
161      * @param i number of sample being evaluated.
162      * @param point point where function will be evaluated.
163      * @param params initial parameters estimation to be tried. These will
164      * change as the Levenberg-Marquard algorithm iterates to the best solution.
165      * These are used as input parameters along with point to evaluate function.
166      * @param derivatives partial derivatives of the function respect to each
167      * provided parameter.
168      * @return function evaluation at provided point.
169      */
170     @Override
171     @SuppressWarnings("Duplicates")
172     protected double evaluate(int i, double[] point, double[] params, double[] derivatives) {
173         //Demonstration in 2D:
174         //--------------------
175         //Taylor series expansion can be expressed as:
176         //f(x) = f(a) + 1/1!*f'(a)*(x - a) + 1/2!*f''(a)*(x - a)^2 + 1/3!*f'''(a)*(x - a)^3 ...
177 
178         //where f'(x) is the derivative of f respect x, which can also be expressed as:
179         //f'(x) = diff(f(x))/diff(x)
180 
181         //and f'(a) is the derivative of f respect x evaluated at a, which can be expressed
182         //as f'(a) = diff(f(a))/diff(x)
183 
184         //consequently f''(a) is the second derivative respect x evaluated at a, which can
185         //be expressed as:
186         //f''(x) = diff(f(x))/diff(x^2)
187 
188         //and:
189         //f''(a) = diff(f(a))/diff(x^2)
190 
191         //and finally f'''(a) is the third derivative respect x evaluated at a, which can
192         //be expressed as:
193         //f'''(x) = diff(f(x))/diff(x^3)
194 
195         //and:
196         //f'''(a) = diff(f(a))/diff(x^3)
197 
198         //Received power expressed in dBm is:
199         //k = (c/(4*pi*f))
200         //Pr = Pte*k^n / d^n
201 
202         //where c is the speed of light, pi is 3.14159..., f is the frequency of the radio source,
203         //Pte is the equivalent transmitted power by the radio source, n is the path-loss exponent
204         // (typically 2.0), and d is the distance from a point to the location of the radio source.
205 
206         //Hence:
207         //Pr(dBm) = 10*log(Pte*k^n/d^n) = 10*n*log(k) + 10*log(Pte) - 10*n*log(d) =
208         //          10*n*log(k) + 10*log(Pte) - 5*n*log(d^2)
209 
210         //The former 2 terms are constant, and only the last term depends on distance
211 
212         //Hence, assuming the constant K = 10*n*log(k) + Pte(dBm), where Pte(dBm) = 10*log(Pte),
213         //assuming that transmitted power by the radio source Pte is known (so that K is also known),
214         //and assuming that the location of the radio source is known and it is located at pa = (xa, ya)
215         //so that d^2 = (x - xa)^2 + (y - ya)^2 then the received power at an unknown point pi = (xi, yi) is:
216 
217         //Pr(pi) = Pr(xi,yi) = K - 5*n*log(d^2) = K - 5*n*log((xi - xa)^2 + (yi - ya)^2)
218 
219         //Suppose that received power at point p1=(x1,y1) is known on a located fingerprint
220         //containing readings Pr(p1).
221 
222         //Then, for an unknown point pi=(xi,yi) close to fingerprint 1 located at p1 where we
223         //have measured received power Pr(pi), we can get the following third-order Taylor
224         //approximation:
225 
226         //Pr(pi = (xi,yi)) = Pr(p1) +
227         //  diff(Pr(p1))/diff(x)*(xi - x1) +
228         //  diff(Pr(p1))/diff(y)*(yi - y1) +
229         //  1/2*(diff(Pr(p1))/diff(x^2)*(xi - x1)^2 +
230         //      diff(Pr(p1))/diff(y^2)*(yi - y1)^2 +
231         //      2*diff(Pr(p1))/diff(x*y)*(xi - x1)*(yi - y1)) +
232         //  1/6*(diff(Pr(p1))/diff(x^3)*(xi - x1)^3 +
233         //      diff(Pr(p1))/diff(y^3)*(yi - y1)^3 +
234         //      3*diff(Pr(p1))/diff(x^2*y)*(xi - x1)^2*(yi - y1) +
235         //      3*diff(Pr(p1))/diff(x*y^2)*(xi - x1)*(yi - y1)^2
236 
237         //where the first order derivatives of Pr(p = (x,y)) are:
238         //diff(Pr(x,y))/diff(x) = -5*n/(ln(10)*((x - xa)^2 + (y - ya)^2)*2*(x - xa)
239         //diff(Pr(x,y))/diff(x) = -10*n*(x - xa)/(ln(10)*((x - xa)^2 + (y - ya)^2))
240 
241         //diff(Pr(x,y))/diff(y) = -5*n/(ln(10)*((x - xa)^2 + (y - ya)^2)*2*(y - ya)
242         //diff(Pr(x,y))/diff(y) = -10*n*(y - ya)/(ln(10)*((x - xa)^2 + (y - ya)^2))
243 
244         //If we evaluate first order derivatives at p1 = (x1,y1), we get:
245         //diff(Pr(p1))/diff(x) = -10*n*(x1 - xa)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2))
246         //diff(Pr(p1))/diff(y) = -10*n*(y1 - ya)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2))
247 
248         //where square distance from fingerprint 1 to radio source a can be expressed as:
249         //d1a^2 = (x1 - xa)^2 + (y1 - ya)^2
250 
251         //where both the fingerprint and radio source positions are known, and hence d1a is known.
252 
253         //Then first order derivatives can be expressed as:
254         //diff(Pr(p1))/diff(x) = -10*n*(x1 - xa)/(ln(10)*d1a^2)
255         //diff(Pr(p1))/diff(y) = -10*n*(y1 - ya)/(ln(10)*d1a^2)
256 
257         //To obtain second order derivatives we take into account that:
258         //(f(x)/g(x))' = (f'(x)*g(x) - f(x)*g'(x))/g(x)^2
259 
260         //hence, second order derivatives of Pr(p = (x,y)) are:
261         //diff(Pr(x,y))/diff(x^2) = -10*n/ln(10)*(1*((x - xa)^2 + (y - ya)^2) - (x - xa)*2*(x - xa)) / ((x - xa)^2 + (y - ya)^2)^2
262         //diff(Pr(x,y))/diff(x^2) = -10*n*((y - ya)^2 - (x - xa)^2)/(ln(10)*((x - xa)^2 + (y - ya)^2)^2)
263 
264         //diff(Pr(x,y))/diff(y^2) = -10*n/ln(10)*(1*((x - xa)^2 + (y - ya)^2) - (y - ya)*2*(y - ya)) / ((x - xa)^2 + (y - ya)^2)^2
265         //diff(Pr(x,y))/diff(y^2) = -10*n*((x - xa)^2 - (y - ya)^2)/(ln(10)*((x - xa)^2 + (y - ya)^2)^2)
266 
267         //diff(Pr(x,y))/diff(x*y) = -10*n/ln(10)*(0*((x - xa)^2 + (y - ya)^2) - (x - xa)*2*(y - ya))/((x - xa)^2 + (y - ya)^2)^2
268         //diff(Pr(x,y))/diff(x*y) = 20*n*((x - xa)*(y - ya))/(ln(10)*((x - xa)^2 + (y - ya)^2)^2)
269 
270         //If we evaluate second order derivatives at p1 = (x1,y1), we get:
271         //diff(Pr(p1))/diff(x^2) = -10*n*((y1 - ya)^2 - (x1 - xa)^2)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2)^2)
272         //diff(Pr(p1))/diff(y^2) = -10*n*((x1 - xa)^2 - (y1 - ya)^2)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2)^2)
273         //diff(Pr(p1))/diff(x*y) = 20*n*(x1 - xa)*(y1 - ya)/(ln(10)*((x1 - xa)^2 + (y1 - ya)^2)^2)
274 
275         //and expressing the second order derivatives in terms of distance between
276         //fingerprint 1 and radio source a d1a, we get:
277         //diff(Pr(p1))/diff(x^2) = -10*n*((y1 - ya)^2 - (x1 - xa)^2)/(ln(10)*d1a^4)
278         //diff(Pr(p1))/diff(y^2) = -10*n*((x1 - xa)^2 - (y1 - ya)^2)/(ln(10)*d1a^4)
279         //diff(Pr(p1))/diff(x*y) = 20*n*(x1 - xa)*(y1 - ya)/(ln(10)*d1a^4)
280 
281         //Finally, third order derivatives of Pr(p = (x,y)) are:
282         //diff(Pr(x,y))/diff(x^3) = -10*n/ln(10)*(-2*(x - xa)*((x - xa)^2 + (y - ya)^2)^2 - ((y - ya)^2 - (x - xa)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(x - xa))/((x - xa)^2 + (y - ya)^2)^4
283         //diff(Pr(x,y))/diff(y^3) = -10*n/ln(10)*(-2*(y - ya)*((x - xa)^2 + (y - ya)^2)^2 - ((x - xa)^2 - (y - ya)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(y - ya))/((x - xa)^2 + (y - ya)^2)^4
284         //diff(Pr(x,y)/diff(x^2*y) = -10*n/ln(10)*(2*(y - ya)*((x - xa)^2 + (y - ya)^2)^2 - ((y - ya)^2 - (x - xa)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(y - ya))/((x - xa)^2 + (y - ya)^2)^4
285         //diff(Pr(x,y)/diff(x*y^2) = -10*n/ln(10)*(2*(x - xa)*((x - xa)^2 + (y - ya)^2)^2 - ((x - xa)^2 - (y - ya)^2)*2*((x - xa)^2 + (y - ya)^2)*2*(x - xa))/((x - xa)^2 + (y - ya)^2)^4
286 
287         //evaluating at p1 = (x1, y1), we get:
288         //diff(Pr(p1))/diff(x^3) = -10*n/ln(10)*(-2*(x1 - xa)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((y1 - ya)^2 - (x1 - xa)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(x1 - xa))/((x1 - xa)^2 + (y1 - ya)^2)^4
289         //diff(Pr(p1))/diff(y^3) = -10*n/ln(10)*(-2*(y1 - ya)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((x1 - xa)^2 - (y1 - ya)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(y1 - ya))/((x1 - xa)^2 + (y1 - ya)^2)^4
290         //diff(Pr(p1)/diff(x^2*y) = -10*n/ln(10)*(2*(y1 - ya)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((y1 - ya)^2 - (x1 - xa)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(y1 - ya))/((x1 - xa)^2 + (y1 - ya)^2)^4
291         //diff(Pr(p1)/diff(x*y^2) = -10*n/ln(10)*(2*(x1 - xa)*((x1 - xa)^2 + (y1 - ya)^2)^2 - ((x1 - xa)^2 - (y1 - ya)^2)*2*((x1 - xa)^2 + (y1 - ya)^2)*2*(x1 - xa))/((x1 - xa)^2 + (y1 - ya)^2)^4
292 
293         //and substituting the distance between fingerprint and radio source d1a, we get:
294         //diff(Pr(p1))/diff(x^3) = -10*n/ln(10)*(-2*(x1 - xa)*dia^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(x1 - xa))/d1a^8
295         //diff(Pr(p1))/diff(y^3) = -10*n/ln(10)*(-2*(y1 - ya)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(y1 - ya))/d1a^8
296         //diff(Pr(p1)/diff(x^2*y) = -10*n/ln(10)*(2*(y1 - ya)*d1a^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(y1 - ya))/d1a^8
297         //diff(Pr(p1)/diff(x*y^2) = -10*n/ln(10)*(2*(x1 - xa)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(x1 - xa))/d1a^8
298 
299 
300         //Hence, the third order Taylor expansion can be expressed as:
301         //Pr(pi = (xi,yi)) = Pr(p1) +
302         //  diff(Pr(p1))/diff(x)*(xi - x1) +
303         //  diff(Pr(p1))/diff(y)*(yi - y1) +
304         //  1/2*(diff(Pr(p1))/diff(x^2)*(xi - x1)^2 +
305         //      diff(Pr(p1))/diff(y^2)*(yi - y1)^2 +
306         //      2*diff(Pr(p1))/diff(x*y)*(xi - x1)*(yi - y1)) +
307         //  1/6*(diff(Pr(p1))/diff(x^3)*(xi - x1)^3 +
308         //      diff(Pr(p1))/diff(y^3)*(yi - y1)^3 +
309         //      3*diff(Pr(p1))/diff(x^2*y)*(xi - x1)^2*(yi - y1) +
310         //      3*diff(Pr(p1))/diff(x*y^2)*(xi - x1)*(yi - y1)^2)
311 
312         //Pr(pi) = Pr(p1)
313         //  -10*n*(x1 - xa)/(ln(10)*d1a^2)*(xi - x1) +
314         //  -10*n*(y1 - ya)/(ln(10)*d1a^2)*(yi - y1) +
315         //  -5*n*((y1 - ya)^2 - (x1 - xa)^2)/(ln(10)*d1a^4)*(xi - x1)^2 +
316         //  -5*n*((x1 - xa)^2 - (y1 - ya)^2)/(ln(10)*d1a^4)*(yi - y1)^2 +
317         //  20*n*(x1 - xa)*(y1 - ya)/(ln(10)*d1a^4)*(xi - x1)*(yi - y1) +
318         //  -10/6*n/ln(10)*(-2*(x1 - xa)*dia^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(x1 - xa))/d1a^8*(xi - x1)^3 +
319         //  -10/6*n/ln(10)*(-2*(y1 - ya)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(y1 - ya))/d1a^8*(yi - y1)^3 +
320         //  -5*n/ln(10)*(2*(y1 - ya)*d1a^4 - ((y1 - ya)^2 - (x1 - xa)^2)*4*d1a^2*(y1 - ya))/d1a^8*(xi - x1)^2*(yi - y1) +
321         //  -5*n/ln(10)*(2*(x1 - xa)*d1a^4 - ((x1 - xa)^2 - (y1 - ya)^2)*4*d1a^2*(x1 - xa))/d1a^8*(xi - x1)*(yi - y1)^2
322 
323 
324         //The equation above can be solved using a non-linear fitter such as Levenberg-Marquardt
325 
326         //This method implements received power at point pi = (xi, yi) and its derivatives
327 
328         double xi = params[0];
329         double yi = params[1];
330 
331         //received power
332         double pr = point[0];
333 
334         //fingerprint coordinates
335         double x1 = point[1];
336         double y1 = point[2];
337 
338         //radio source coordinates
339         double xa = point[3];
340         double ya = point[4];
341 
342         //path loss exponent
343         double n = point[5];
344 
345         double ln10 = Math.log(10.0);
346 
347         double diffXi1 = xi - x1;
348         double diffYi1 = yi - y1;
349 
350         double diffX1a = x1 - xa;
351         double diffY1a = y1 - ya;
352 
353         double diffXi12 = diffXi1 * diffXi1;
354         double diffYi12 = diffYi1 * diffYi1;
355 
356         double diffXi13 = diffXi12 * diffXi1;
357         double diffYi13 = diffYi12 * diffYi1;
358 
359         double diffX1a2 = diffX1a * diffX1a;
360         double diffY1a2 = diffY1a * diffY1a;
361 
362         double d1a2 = diffX1a2 + diffY1a2;
363         double d1a4 = d1a2 * d1a2;
364         double d1a8 = d1a4 * d1a4;
365 
366         double value1 = - 10.0 * n * diffX1a / (ln10 * d1a2);
367         double value2 = - 10.0 * n * diffY1a / (ln10 * d1a2);
368         double value3 = - 5.0 * n * (-diffX1a2 + diffY1a2) / (ln10 * d1a4);
369         double value4 = - 5.0 * n * (diffX1a2 - diffY1a2) / (ln10 * d1a4);
370         double value5 = 20.0 * n * diffX1a * diffY1a / (ln10 * d1a4);
371         double value6 = - 10.0 / 6.0 * n / ln10 * (-2.0 * diffX1a * d1a4 - (-diffX1a2 + diffY1a2) * 4.0 * d1a2 * diffX1a) / d1a8;
372         double value7 = - 10.0 / 6.0 * n / ln10 * (-2.0 * diffY1a * d1a4 - (diffX1a2 - diffY1a2) * 4.0 * d1a2 * diffY1a) / d1a8;
373         double value8 = - 5.0 * n / ln10 * (2.0 * diffY1a * d1a4 - (-diffX1a2 + diffY1a2) * 4.0 * d1a2 * diffY1a) / d1a8;
374         double value9 = - 5.0 * n / ln10 * (2.0 * diffX1a * d1a4 - (diffX1a2 - diffY1a2) * 4.0 * d1a2 * diffX1a) / d1a8;
375 
376         //hence:
377         //Pr(pi) = Pr(p1) +
378         //  value1*(xi - x1) +
379         //  value2*(yi - y1) +
380         //  value3*(xi - x1)^2 +
381         //  value4*(yi - y1)^2 +
382         //  value5*(xi - x1)*(yi - y1) +
383         //  value6*(xi - x1)^3 +
384         //  value7*(yi - y1)^3 +
385         //  value8*(xi - x1)^2*(yi - y1) +
386         //  value9*(xi - x1)*(yi - y1)^2 +
387 
388         double result = pr
389                 + value1 * diffXi1
390                 + value2 * diffYi1
391                 + value3 * diffXi12
392                 + value4 * diffYi12
393                 + value5 * diffXi1 * diffYi1
394                 + value6 * diffXi13
395                 + value7 * diffYi13
396                 + value8 * diffXi12 * diffYi1
397                 + value9 * diffXi1 * diffYi12;
398 
399         //derivative respect xi
400 
401         //diff(Pr(pi))/diff(xi) = value1 +
402         //  2*value3*(xi - x1) +
403         //  value5*(yi - y1) +
404         //  3*value6*(xi - x1)^2 +
405         //  2*value8*(xi - x1)*(yi - y1) +
406         //  value9*(yi - y1)^2
407 
408         derivatives[0] = value1 + 2.0 * value3 * diffXi1 + value5 * diffYi1 +
409                 3.0 * value6 * diffXi12 + 2.0 * value8 * diffXi1 * diffYi1 +
410                 value9 * diffYi12;
411 
412         //derivative respect yi
413 
414         //diff(Pr(pi))/diff(yi) = value2 +
415         //  2*value4*(yi - y1) +
416         //  value5*(xi - x1) +
417         //  3*value7*(yi - y1)^2 +
418         //  value8*(xi - x1)^2 +
419         //  2*value9*(xi - x1)*(yi - y1)
420 
421         derivatives[1] = value2 + 2.0 * value4 * diffYi1 + value5 * diffXi1 +
422                 3.0 * value7 * diffYi12 + value8 * diffXi12 +
423                 2.0 * value9 * diffXi1 * diffYi1;
424 
425         return result;
426     }

428     /**
429      * Propagates provided variances into RSSI variance of non-located fingerprint
430      * reading.
431      * @param fingerprintRssi closest located fingerprint reading RSSI expressed in dBm's.
432      * @param pathlossExponent path-loss exponent.
433      * @param fingerprintPosition position of closest fingerprint.
434      * @param radioSourcePosition radio source position associated to fingerprint reading.
435      * @param estimatedPosition position to be estimated. Usually this is equal to the
436      *                          initial position used by a non linear algorithm.
437      * @param fingerprintRssiVariance variance of fingerprint RSSI or null if unknown.
438      * @param pathlossExponentVariance variance of path-loss exponent or null if unknown.
439      * @param fingerprintPositionCovariance covariance of fingerprint position or null if
440      *                                      unknown.
441      * @param radioSourcePositionCovariance covariance of radio source position or null if
442      *                                      unknown.
443      * @return variance of RSSI measured at non located fingerprint reading.
444      */
445     @Override
446     @SuppressWarnings("Duplicates")
447     protected Double propagateVariances(double fingerprintRssi,
448             double pathlossExponent, Point2D fingerprintPosition,
449             Point2D radioSourcePosition, Point2D estimatedPosition,
450             Double fingerprintRssiVariance, Double pathlossExponentVariance,
451             Matrix fingerprintPositionCovariance,
452             Matrix radioSourcePositionCovariance) {
453         try {
454             MultivariateNormalDist dist =
455                     Utils.propagateVariancesToRssiVarianceThirdOrderNonLinear2D(
456                             fingerprintRssi, pathlossExponent, fingerprintPosition,
457                             radioSourcePosition, estimatedPosition, fingerprintRssiVariance,
458                             pathlossExponentVariance, fingerprintPositionCovariance,
459                             radioSourcePositionCovariance, null);
460             if (dist == null) {
461                 return null;
462             }
463 
464             Matrix covariance = dist.getCovariance();
465             if (covariance == null) {
466                 return null;
467             }
468 
469             return covariance.getElementAt(0, 0);
470 
471         } catch (IndoorException e) {
472             return null;
473         }
474     }
