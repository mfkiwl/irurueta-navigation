27  /**
28   * Robustly estimated 2D position of a radio source (e.g. WiFi
29   * access point or bluetooth beacon), by discarding outliers using MSAC
30   * algorithm.
31   *
32   * @param <S> a {@link RadioSource} type.
33   */
34  @SuppressWarnings({"WeakerAccess", "Duplicates"})
35  public class MSACRobustRangingRadioSourceEstimator2D<S extends RadioSource> extends
36          RobustRangingRadioSourceEstimator2D<S> {
37  
38      /**
39       * Constant defining default threshold to determine whether samples are
40       * inliers or not.
41       */
42      public static final double DEFAULT_THRESHOLD = 0.1;
43  
44      /**
45       * Minimum value that can be set as threshold.
46       * Threshold must be strictly greater than 0.0.
47       */
48      public static final double MIN_THRESHOLD = 0.0;
49  
50      /**
51       * Threshold to determine whether samples are inliers or not when
52       * testing possible estimation solutions.
53       */
54      private double mThreshold = DEFAULT_THRESHOLD;
55  
56      /**
57       * Constructor.
58       */
59      public MSACRobustRangingRadioSourceEstimator2D() {
60          super();
61      }
62  
63      /**
64       * Constructor.
65       * <p>
66       * Sets radio signal ranging readings belonging to the same radio source.
67       *
68       * @param readings radio signal ranging readings belonging to the same
69       *                 radio source.
70       * @throws IllegalArgumentException if readings are not valid.
71       */
72      public MSACRobustRangingRadioSourceEstimator2D(
73              final List<? extends RangingReadingLocated<S, Point2D>> readings) {
74          super(readings);
75      }
76  
77      /**
78       * Constructor.
79       *
80       * @param listener listener in charge of attending events raised by this instance.
81       */
82      public MSACRobustRangingRadioSourceEstimator2D(
83              final RobustRangingRadioSourceEstimatorListener<S, Point2D> listener) {
84          super(listener);
85      }
86  
87      /**
88       * Constructor.
89       * Sets radio signal readings belonging to the same radio source.
90       *
91       * @param readings radio signal readings belonging to the same radio source.
92       * @param listener listener in charge of attending events raised by this instance.
93       * @throws IllegalArgumentException if readings are not valid.
94       */
95      public MSACRobustRangingRadioSourceEstimator2D(
96              final List<? extends RangingReadingLocated<S, Point2D>> readings,
97              final RobustRangingRadioSourceEstimatorListener<S, Point2D> listener) {
98          super(readings, listener);
99      }

154         super(readings, initialPosition, listener);

157     /**
158      * Returns threshold to determine whether samples are inliers or not.
159      *
160      * @return threshold to determine whether samples are inliers or not.
161      */
162     public double getThreshold() {
163         return mThreshold;
164     }
165 
166     /**
167      * Sets threshold to determine whether samples are inliers or not.
168      *
169      * @param threshold threshold to be set.
170      * @throws IllegalArgumentException if provided value is equal or less than
171      *                                  zero.
172      * @throws LockedException          if robust estimator is locked because an
173      *                                  estimation is already in progress.
174      */
175     public void setThreshold(final double threshold) throws LockedException {
176         if (isLocked()) {
177             throw new LockedException();
178         }
179         if (threshold <= MIN_THRESHOLD) {
180             throw new IllegalArgumentException();
181         }
182         mThreshold = threshold;
183     }
184 
185     /**
186      * Robustly estimates position for a radio source.
187      *
188      * @throws LockedException          if instance is busy during estimation.
189      * @throws NotReadyException        if estimator is not ready.
190      * @throws RobustEstimatorException if estimation fails for any reason
191      *                                  (i.e. numerical instability, no solution available, etc).
192      */
193     @Override
194     public void estimate() throws LockedException, NotReadyException,
195             RobustEstimatorException {
196         if (isLocked()) {
197             throw new LockedException();
198         }
199         if (!isReady()) {
200             throw new NotReadyException();
201         }
202 
203         final MSACRobustEstimator<Solution<Point2D>> innerEstimator =
204                 new MSACRobustEstimator<>(
205                         new MSACRobustEstimatorListener<Solution<Point2D>>() {
206                             @Override
207                             public double getThreshold() {
208                                 return mThreshold;
209                             }
210 
211                             @Override
212                             public int getTotalSamples() {
213                                 return mReadings.size();
214                             }
215 
216                             @Override
217                             public int getSubsetSize() {
218                                 return Math.max(mPreliminarySubsetSize, getMinReadings());
219                             }
220 
221                             @Override
222                             public void estimatePreliminarSolutions(
223                                     final int[] sampleIndices,
224                                     final List<Solution<Point2D>> solutions) {
225                                 solvePreliminarSolutions(sampleIndices, solutions);
226                             }
227 
228                             @Override
229                             public double computeResidual(
230                                     final Solution<Point2D> currentEstimation,
231                                     final int i) {
232                                 return residual(currentEstimation, i);
233                             }
234 
235                             @Override
236                             public boolean isReady() {
237                                 return MSACRobustRangingRadioSourceEstimator2D.this.isReady();
238                             }
239 
240                             @Override
241                             public void onEstimateStart(
242                                     final RobustEstimator<Solution<Point2D>> estimator) {
243                             }
244 
245                             @Override
246                             public void onEstimateEnd(
247                                     final RobustEstimator<Solution<Point2D>> estimator) {
248                             }
249 
250                             @Override
251                             public void onEstimateNextIteration(
252                                     final RobustEstimator<Solution<Point2D>> estimator,
253                                     final int iteration) {
254                                 if (mListener != null) {
255                                     mListener.onEstimateNextIteration(
256                                             MSACRobustRangingRadioSourceEstimator2D.this,
257                                             iteration);
258                                 }
259                             }
260 
261                             @Override
262                             public void onEstimateProgressChange(
263                                     final RobustEstimator<Solution<Point2D>> robustEstimator,
264                                     final float progress) {
265                                 if (mListener != null) {
266                                     mListener.onEstimateProgressChange(
267                                             MSACRobustRangingRadioSourceEstimator2D.this,
268                                             progress);
269                                 }
270                             }
271                         });
272 
273         try {
274             mLocked = true;
275 
276             if (mListener != null) {
277                 mListener.onEstimateStart(this);
278             }
279 
280             mInliersData = null;
281             innerEstimator.setConfidence(mConfidence);
282             innerEstimator.setMaxIterations(mMaxIterations);
283             innerEstimator.setProgressDelta(mProgressDelta);
284             final Solution<Point2D> result = innerEstimator.estimate();
285             mInliersData = innerEstimator.getInliersData();
286             attemptRefine(result);
287 
288             if (mListener != null) {
289                 mListener.onEstimateEnd(this);
290             }
291 
292         } catch (final com.irurueta.numerical.LockedException e) {
293             throw new LockedException(e);
294         } catch (final com.irurueta.numerical.NotReadyException e) {
295             throw new NotReadyException(e);
296         } finally {
297             mLocked = false;
298         }
299     }
300 
301     /**
302      * Returns method being used for robust estimation.
303      *
304      * @return method being used for robust estimation.
305      */
306     @Override
307     public RobustEstimatorMethod getMethod() {
308         return RobustEstimatorMethod.MSAC;
309     }
