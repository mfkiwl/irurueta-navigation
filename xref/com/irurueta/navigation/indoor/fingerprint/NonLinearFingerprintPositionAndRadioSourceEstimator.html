33  /**
34   * Base class for position and radio source estimators based only on located
35   * fingerprints containing RSSI readings.
36   * All implementations of this class estimate the position of a new fingerprint
37   * and the position of all radio sources associated to fingerprints whose location
38   * is known.
39   * All implementations solve the problem in a non linear way using Levenberg-Marquardt
40   * algorithm.
41   */
42  @SuppressWarnings("WeakerAccess")
43  public abstract class NonLinearFingerprintPositionAndRadioSourceEstimator<P extends Point<?>> extends
44          FingerprintPositionAndRadioSourceEstimator<P> {

604     /**
605      * Estimates position and radio sources based on provided located radio sources and readings of
606      * such radio sources at an unknown location.
607      *
608      * @throws LockedException                if estimator is locked.
609      * @throws NotReadyException              if estimator is not ready.
610      * @throws FingerprintEstimationException if estimation fails for some other reason.
611      */
612     @Override
613     @SuppressWarnings("Duplicates")
614     public void estimate() throws LockedException, NotReadyException,
615             FingerprintEstimationException {
616 
617         if (!isReady()) {
618             throw new NotReadyException();
619         }
620         if (isLocked()) {
621             throw new LockedException();
622         }
623 
624         try {
625             mLocked = true;
626 
627             if (mListener != null) {
628                 mListener.onEstimateStart(this);
629             }
630 
631             RadioSourceNoMeanKNearestFinder<P, RadioSource> noMeanfinder = null;
632             RadioSourceKNearestFinder<P, RadioSource> finder = null;
633             if (mUseNoMeanNearestFingerprintFinder) {
634                 //noinspection unchecked
635                 noMeanfinder = new RadioSourceNoMeanKNearestFinder<>(
636                         (Collection<? extends RssiFingerprintLocated<RadioSource,
637                                 RssiReading<RadioSource>, P>>) mLocatedFingerprints);
638             } else {
639                 //noinspection unchecked
640                 finder = new RadioSourceKNearestFinder<>(
641                         (Collection<? extends RssiFingerprintLocated<RadioSource,
642                                 RssiReading<RadioSource>, P>>) mLocatedFingerprints);
643             }
644 
645             mEstimatedPositionCoordinates = null;
646             mCovariance = null;
647             mEstimatedPositionCovariance = null;
648             mNearestFingerprints = null;
649             mEstimatedLocatedSources = null;
650 
651             final int min = Math.max(1, mMinNearestFingerprints);
652             final int max = mMaxNearestFingerprints < 0 ?
653                     mLocatedFingerprints.size() :
654                     Math.min(mMaxNearestFingerprints, mLocatedFingerprints.size());
655             for (int k = min; k <= max; k++) {
656                 if (noMeanfinder != null) {
657                     //noinspection unchecked
658                     mNearestFingerprints = noMeanfinder.findKNearestTo(
659                             (RssiFingerprint<RadioSource, RssiReading<RadioSource>>) mFingerprint, k);
660                 } else {
661                     //noinspection unchecked
662                     mNearestFingerprints = finder.findKNearestTo(
663                             (RssiFingerprint<RadioSource, RssiReading<RadioSource>>) mFingerprint, k);
664                 }
665 
666                 //Demonstration in 2D:
667                 //--------------------
668 
669                 //The expression of received power expressed in dBm's is:
670                 //k = (c/(4*pi*f))
671                 //Pr = Pte*k^n / d^n
672 
673                 //where c is the speed of light, pi is 3.14159..., f is the frequency of the radio source,
674                 //Pte is the equivalent transmitted power by the radio source, n is the path-loss exponent
675                 // (typically 2.0), and d is the distance from a point to the location of the radio source.
676 
677                 //Hence:
678                 //Pr(dBm) = 10*log(Pte*k^n/d^n) = 10*n*log(k) + 10*log(Pte) - 10*n*log(d) =
679                 //          10*n*log(k) + 10*log(Pte) - 5*n*log(d^2)
680 
681                 //where d^2 = dia^2 = (xi - xa)^2 + (yi - ya)^2 is the squared distance between
682                 //fingerprint and unknown point pi = (xi, yi) and
683                 //radio source a = a,b... M
684                 //10*n*log(k) is constant for a given radio source "a", and
685                 //Pte is the equivalent transmitted power of radio source "a".
686 
687                 //We assume that the 2 former terms are constant and known for a given radio source
688                 //K = 10*n*log(k) + 10*log(Pte), and the only unknown term is
689                 //the latter one depending on the distance of the
690                 //measuring point and the radio source.
691 
692                 //Hence for a given radio source "a" at unknown location "i":
693                 //Pr(pi) = K - 5*n*log((xi - xa)^2 + (yi - ya)^2)
694 
695                 //we assume that a known located fingerprint is located at p1 = (x1, y1),
696                 //Both readings Pr(pi) and Pr(p1) belong to the same radio source "a", hence
697                 //K term is the same.
698 
699                 //Pr(p1) = K - 5*n*log((x1 - xa)^2 + (y1 - ya)^2)
700 
701                 //where d1a^2 = (x1 - xa)^2 + (y1 - ya)^2 is the squared distance between
702                 //fingerprint 1 and radio source 2
703 
704                 //To remove possible bias effects on readings, we consider the difference of received
705                 //power for fingerprint "1" and radio source "a" as:
706 
707                 //Prdiff1a = Pr(pi) - Pr(p1) = (K - 5*n*log(dia^2)) - (K - 5*n*log(d1a^2)) =
708                 //  = 5*n*log(d1a^2) - 5*n*log(dia^2)
709 
710                 //where both d1a^2 and dia^2 are unknown, because location pi=(xi,yi) and pa=(xa,ya) are unknown.
711                 //Now we have no dependencies on the amount of transmitted power of each radio source
712                 //contained on constant term K, and we only depend on squared distances d1a^2 and dia^2.
713 
714                 //Consequently, the difference of received power for fingerprint "2" and radio source "a" is:
715                 //Prdiff2a = 5*n*log(d2a^2) - 5*n*log(dia^2)
716 
717                 //the difference of received power for fingerprint "1" and radio source "b" is:
718                 //Prdiff1b = 5*n*log(d1b^2) - 5*n*log(dib^2)
719 
720                 //and so on.
721 
722                 //we want to find unknown location pi, and location of radio source pa, pb,... pM so that Prdiff
723                 //errors are minimized in LMSE (Least Mean Square Error) terms.
724 
725                 //Assuming that we have M radio sources and N fingerprints, we have
726                 //y = [Prdiff1a Prdiff2a Prdiff1b Prdiff2b ... PrdiffNa PrdiffNb ... PrdiffNM]
727 
728                 //and the unknowns to be found are:
729                 //x = [xi yi xa ya xb yb ... xM yM], which are the location of the unknown fingerprint
730                 //pi = (xi, yi) and the locations of the radio sources a, b ... M that we want to find pa = (xa, ya),
731                 //pb = (xb, yb) ... pM = (xM, yM)
732 
733                 try {
734                     final List<RadioSource> sourcesToBeEstimated = setupFitter();
735                     if (mMinNearestFingerprints < 0) {
736                         //if no limit is set in minimum value, then a minimum of
737                         //dims * (1 + numSources) is used
738                         final int numSources = sourcesToBeEstimated.size();
739                         final int dims = getNumberOfDimensions();
740                         final int minNearest = dims * (1 + numSources);
741                         if (k < minNearest) {
742                             continue;
743                         }
744                     }
745 
746                     mFitter.fit();
747 
748                     //estimated position
749                     final double[] a = mFitter.getA();
750                     mCovariance = mFitter.getCovar();
751                     mChiSq = mFitter.getChisq();
752 
753                     final int dims = getNumberOfDimensions();
754 
755                     //obtain estimated position coordinates and covariance
756                     mEstimatedPositionCoordinates = new double[dims];
757                     System.arraycopy(a, 0, mEstimatedPositionCoordinates,
758                             0, dims);
759 
760                     final int dimsMinusOne = dims - 1;
761                     mEstimatedPositionCovariance = mCovariance.getSubmatrix(0, 0,
762                             dimsMinusOne, dimsMinusOne);
763 
764                     //obtain radio sources estimated positions and covariance
765                     final int totalSources = sourcesToBeEstimated.size();
766                     mEstimatedLocatedSources = new ArrayList<>();
767                     for (int j = 0; j < totalSources; j++) {
768                         final P sourcePosition = createPoint();
769 
770                         final int start = dims * (1 + j);
771                         final int end = start + dimsMinusOne;
772                         for (int i = 0; i < dims; i++) {
773                             sourcePosition.setInhomogeneousCoordinate(i, a[start + i]);
774                         }
775 
776                         final Matrix sourceCovariance = mCovariance.getSubmatrix(
777                                 start, start, end, end);
778 
779                         final RadioSource source = sourcesToBeEstimated.get(j);
780                         mEstimatedLocatedSources.add(
781                                 createRadioSource(source, sourcePosition,
782                                         sourceCovariance));
783                     }
784 
785                     //a solution was found so we exit loop
786                     break;
787                 } catch (final NumericalException e) {
788                     //solution could not be found with current data
789                     //Iterate to use additional nearby fingerprints
790                     mEstimatedPositionCoordinates = null;
791                     mCovariance = null;
792                     mEstimatedPositionCovariance = null;
793                     mEstimatedLocatedSources = null;
794                 }
795             }
796 
797             if (mEstimatedPositionCoordinates == null ||
798                     mEstimatedLocatedSources == null) {
799                 //no position could be estimated
800                 throw new FingerprintEstimationException();
801             }
802 
803             if (mListener != null) {
804                 mListener.onEstimateEnd(this);
805             }
806         } finally {
807             mLocked = false;
808         }
809     }

896     /**
897      * Builds data required to solve the problem.
898      * This method takes into account current nearest fingerprints and discards those
899      * readings belonging to radio sources not having enough data to be estimated.
900      *
901      * @param allPowerDiffs               list of received power differences of RSSI readings between a
902      *                                    located fingerprint and an unknown fingerprint for a given radio
903      *                                    source.
904      * @param allFingerprintPositions     positions of all located fingerprints being taken into
905      *                                    account.
906      * @param allInitialSourcesPositions  initial positions of all radio sources to be taken
907      *                                    into account. If initial located sources where provided,
908      *                                    their positions will be used, otherwise the centroid of
909      *                                    all located fingerprints associated to a radio source will
910      *                                    be used as initial position.
911      * @param allSourcesToBeEstimated     all radio sources that will be estimated.
912      * @param allSourcesIndices           indices indicating the position radio source being used
913      *                                    within the list of sources for current reading.
914      * @param allPathLossExponents        list of path loss exponents.
915      * @param allStandardDeviations       list of standard deviations for readings being used.
916      * @param nearestFingerprintsCentroid centroid of nearest fingerprints being taken into account.
917      */
918     @SuppressWarnings("Duplicates")
919     private void buildData(
920             final List<Double> allPowerDiffs,
921             final List<P> allFingerprintPositions,
922             final List<P> allInitialSourcesPositions,
923             final List<RadioSource> allSourcesToBeEstimated,
924             final List<Integer> allSourcesIndices,
925             final List<Double> allPathLossExponents,
926             final List<Double> allStandardDeviations,
927             final P nearestFingerprintsCentroid) {
928 
929         final int dims = getNumberOfDimensions();
930         int num = 0;
931         final double[] centroidCoords = new double[dims];
932         for (final RssiFingerprintLocated<? extends RadioSource,
933                 ? extends RssiReading<? extends RadioSource>, P> fingerprint : mNearestFingerprints) {
934             final P position = fingerprint.getPosition();
935             if (position == null) {
936                 continue;
937             }
938 
939             for (int i = 0; i < dims; i++) {
940                 centroidCoords[i] += position.getInhomogeneousCoordinate(i);
941             }
942             num++;
943         }
944 
945         for (int i = 0; i < dims; i++) {
946             centroidCoords[i] /= num;
947             nearestFingerprintsCentroid.setInhomogeneousCoordinate(i, centroidCoords[i]);
948         }
949 
950         //maps to keep cached in memory computed values to speed up computations
951         final HashMap<RadioSource, Integer> numReadingsMap = new HashMap<>();
952         final HashMap<RadioSource, P> centroidsMap = new HashMap<>();
953 
954         for (final RssiFingerprintLocated<RadioSource, RssiReading<RadioSource>, P> locatedFingerprint :
955                 mNearestFingerprints) {
956 
957             final List<RssiReading<RadioSource>> locatedReadings =
958                     locatedFingerprint.getReadings();
959             if (locatedReadings == null) {
960                 continue;
961             }
962 
963             final P fingerprintPosition = locatedFingerprint.getPosition();
964             final Matrix fingerprintPositionCovariance = locatedFingerprint.
965                     getPositionCovariance();
966 
967             for (final RssiReading<RadioSource> locatedReading : locatedReadings) {
968                 final RadioSource source = locatedReading.getSource();
969 
970                 //obtain the total number of readings available for this source and
971                 //the centroid of all located fingerprints containing readings for
972                 //such source
973                 final int numReadings;
974                 if (!numReadingsMap.containsKey(source)) {
975                     numReadings = totalReadingsForSource(source, mNearestFingerprints, null);
976                     numReadingsMap.put(source, numReadings);
977                 } else {
978                     numReadings = numReadingsMap.get(source);
979                 }
980 
981                 if (numReadings < dims) {
982                     continue;
983                 }
984 
985                 final P centroid;
986                 if (!centroidsMap.containsKey(source)) {
987                     centroid = createPoint();
988 
989                     //noinspection unchecked
990                     totalReadingsForSource(source,
991                             (List<RssiFingerprintLocated<RadioSource, RssiReading<RadioSource>, P>>) mLocatedFingerprints,
992                             centroid);
993 
994                     centroidsMap.put(source, centroid);
995                 } else {
996                     centroid = centroidsMap.get(source);
997                 }
998 
999                 //find within the list of located sources (if available) the source
1000                 //of current located fingerprint
1001                 @SuppressWarnings("SuspiciousMethodCalls") final int pos = mInitialLocatedSources != null ?
1002                         mInitialLocatedSources.indexOf(source) : -1;
1003 
1004                 double pathLossExponent = mPathLossExponent;
1005                 Double pathLossExponentVariance = null;
1006                 if (mUseSourcesPathLossExponentWhenAvailable &&
1007                         source instanceof RadioSourceWithPower) {
1008                     RadioSourceWithPower sourceWithPower = (RadioSourceWithPower) source;
1009                     pathLossExponent = sourceWithPower.getPathLossExponent();
1010                     final Double std = sourceWithPower.getPathLossExponentStandardDeviation();
1011                     pathLossExponentVariance = std != null ? std * std : null;
1012                 }
1013 
1014 
1015                 final P sourcePosition;
1016                 Matrix sourcePositionCovariance = null;
1017                 if (pos < 0) {
1018                     //located source is not available, so we use centroid
1019                     sourcePosition = centroid;
1020                 } else {
1021                     final RadioSourceLocated<P> locatedSource = mInitialLocatedSources.get(pos);
1022                     sourcePosition = locatedSource.getPosition();
1023 
1024                     if (mUseSourcesPathLossExponentWhenAvailable &&
1025                             locatedSource instanceof RadioSourceWithPower &&
1026                             pathLossExponentVariance == null) {
1027                         final RadioSourceWithPower locatedSourceWithPower =
1028                                 (RadioSourceWithPower) locatedSource;
1029                         pathLossExponent = locatedSourceWithPower.getPathLossExponent();
1030                         final Double std = locatedSourceWithPower.
1031                                 getPathLossExponentStandardDeviation();
1032                         pathLossExponentVariance = std != null ? std * std : null;
1033                     }
1034 
1035                     sourcePositionCovariance = locatedSource.getPositionCovariance();
1036                 }
1037 
1038                 final int sourceIndex;
1039                 if (!allSourcesToBeEstimated.contains(source)) {
1040                     sourceIndex = allSourcesToBeEstimated.size();
1041 
1042                     allSourcesToBeEstimated.add(source);
1043                     allInitialSourcesPositions.add(sourcePosition);
1044                 } else {
1045                     sourceIndex = allSourcesToBeEstimated.indexOf(source);
1046                 }
1047 
1048 
1049                 final double locatedRssi = locatedReading.getRssi();
1050 
1051                 final Double locatedRssiStd = locatedReading.getRssiStandardDeviation();
1052                 final Double locatedRssiVariance = locatedRssiStd != null ?
1053                         locatedRssiStd * locatedRssiStd : null;
1054 
1055                 final List<? extends RssiReading<? extends RadioSource>> readings =
1056                         mFingerprint.getReadings();
1057                 for (final RssiReading<? extends RadioSource> reading : readings) {
1058                     if (reading.getSource() == null ||
1059                             !reading.getSource().equals(source)) {
1060                         continue;
1061                     }
1062 
1063                     //only take into account reading for matching sources on located
1064                     //and non-located readings
1065                     final double rssi = reading.getRssi();
1066 
1067                     final double powerDiff = rssi - locatedRssi;
1068 
1069                     Double standardDeviation = null;
1070                     if (mPropagatePathlossExponentStandardDeviation ||
1071                             mPropagateFingerprintPositionCovariance ||
1072                             mPropagateRadioSourcePositionCovariance) {
1073 
1074                         //compute initial position
1075                         final P initialPosition = mInitialPosition != null ?
1076                                 mInitialPosition : nearestFingerprintsCentroid;
1077 
1078                         final Double variance = propagateVariances(pathLossExponent,
1079                                 fingerprintPosition, sourcePosition, initialPosition,
1080                                 mPropagatePathlossExponentStandardDeviation ? pathLossExponentVariance : null,
1081                                 mPropagateFingerprintPositionCovariance ? fingerprintPositionCovariance : null,
1082                                 mPropagateRadioSourcePositionCovariance ? sourcePositionCovariance : null);
1083                         if (variance != null) {
1084                             standardDeviation = Math.sqrt(variance);
1085                         }
1086                     }
1087 
1088                     if (standardDeviation == null) {
1089                         standardDeviation = reading.getRssiStandardDeviation();
1090                     }
1091 
1092                     if (mPropagateFingerprintRssiStandardDeviation) {
1093                         if (standardDeviation != null &&
1094                                 reading.getRssiStandardDeviation() != null) {
1095                             //consider propagated variance and reading variance independent, so we
1096                             //sum them both
1097                             standardDeviation = standardDeviation * standardDeviation +
1098                                     reading.getRssiStandardDeviation() * reading.getRssiStandardDeviation();
1099                             standardDeviation = Math.sqrt(standardDeviation);
1100                         }
1101 
1102                         if (locatedRssiVariance != null && standardDeviation != null) {
1103                             //consider propagated variance and located reading variance
1104                             //independent, so we sum them both
1105                             standardDeviation = standardDeviation * standardDeviation +
1106                                     locatedRssiVariance;
1107                             standardDeviation = Math.sqrt(standardDeviation);
1108                         }
1109                     }
1110 
1111                     if (standardDeviation == null || standardDeviation < TINY) {
1112                         standardDeviation = mFallbackRssiStandardDeviation;
1113                     }
1114 
1115                     allPowerDiffs.add(powerDiff);
1116                     allFingerprintPositions.add(fingerprintPosition);
1117                     allSourcesIndices.add(sourceIndex);
1118                     allPathLossExponents.add(pathLossExponent);
1119                     allStandardDeviations.add(standardDeviation);
1120                 }
1121             }
1122         }
1123     }

1126      * Setups fitter to solve positions.
1127      *
1128      * @return list of radio sources whose location will be estimated.
1129      * @throws FittingException if Levenberg-Marquardt fitting fails.
1130      */
1131     @SuppressWarnings("Duplicates")
1132     private List<RadioSource> setupFitter() throws FittingException {
1133         //build lists of data
1134         final List<Double> allPowerDiffs = new ArrayList<>();
1135         final List<P> allFingerprintPositions = new ArrayList<>();
1136         final List<P> allInitialSourcesPositions = new ArrayList<>();
1137         final List<RadioSource> allSourcesToBeEstimated = new ArrayList<>();
1138         final List<Integer> allSourcesIndices = new ArrayList<>();
1139         final List<Double> allPathLossExponents = new ArrayList<>();
1140         final List<Double> allStandardDeviations = new ArrayList<>();
1141         final P nearestFingerprintsCentroid = createPoint();
1142         buildData(allPowerDiffs, allFingerprintPositions, allInitialSourcesPositions,
1143                 allSourcesToBeEstimated, allSourcesIndices, allPathLossExponents,
1144                 allStandardDeviations, nearestFingerprintsCentroid);
1145 
1146         final int totalReadings = allPowerDiffs.size();
1147         final int totalSources = allSourcesToBeEstimated.size();
1148         final int dims = getNumberOfDimensions();
1149         final int n = 1 + dims;
1150 
1151         mFitter.setFunctionEvaluator(new LevenbergMarquardtMultiDimensionFunctionEvaluator() {
1152             @Override
1153             public int getNumberOfDimensions() {
1154                 return n;
1155             }
1156 
1157             @Override
1158             public double[] createInitialParametersArray() {
1159                 final double[] initial = new double[dims * (totalSources + 1)];
1160 
1161                 if (mInitialPosition == null) {
1162                     //use centroid of nearest fingerprints as initial value
1163                     for (int i = 0; i < dims; i++) {
1164                         initial[i] = nearestFingerprintsCentroid.getInhomogeneousCoordinate(i);
1165                     }
1166                 } else {
1167                     //use provided initial position
1168                     for (int i = 0; i < dims; i++) {
1169                         initial[i] = mInitialPosition.getInhomogeneousCoordinate(i);
1170                     }
1171                 }
1172 
1173                 int pos = dims;
1174                 for (int j = 0; j < totalSources; j++) {
1175                     final P initialSourcePosition = allInitialSourcesPositions.get(j);
1176                     for (int i = 0; i < dims; i++) {
1177                         initial[pos] = initialSourcePosition.
1178                                 getInhomogeneousCoordinate(i);
1179                         pos++;
1180                     }
1181                 }
1182 
1183                 return initial;
1184             }

1186             @Override
1187             public double evaluate(
1188                     final int i, final double[] point, final double[] params,
1189                     final double[] derivatives) {
1190 
1191                 //For 2D:
1192                 //-------
1193 
1194                 //Prdiff1a = Pr(pi) - Pr(p1) = 5*n*log(d1a^2) - 5*n*log(dia^2) =
1195                 //  = 5*n*log((x1 - xa)^2 + (y1 - ya)^2) - 5*n*log((xi - xa)^2 + (yi - ya)^2)
1196 
1197                 //derivatives respect parameters being estimated (xi,yi,xa,ya...,xM,yM)
1198                 //for unknown point pi = (xi, yi)
1199                 //diff(Prdiff1a)/diff(xi) = -5*n/(log(10)*((xi - xa)^2 + (yi - ya)^2))*2*(xi - xa)
1200                 //  = -10*n*(xi - xa)/(log(10)*((xi - xa)^2 + (yi - ya)^2))
1201                 //  = -10*n*(xi - xa)/(log(10)*dia^2)
1202 
1203                 //diff(Prdiff1a)/diff(yi) = -5*n/(log(10)*((xi - xa)^2 + (yi - ya)^2))*2*(yi - ya)
1204                 //  = -10*n*(yi - ya)/(log(10)*((xi - xa)^2 + (yi - ya)^2))
1205                 //  = -10*n*(yi - ya)/(log(10)*dia^2)
1206 
1207                 //for same radio source pa=(xa,ya)
1208                 //diff(Prdiff1a)/diff(xa) = 5*n/(log(10)*((x1 - xa)^2 + (y1 - ya)^2))*-2*(x1 - xa) -5*n/(log(10)*((xi - xa)^2 + (yi - ya)^2))*-2*(xi - xa) =
1209                 //  = -10*n*(x1 - xa)/(log(10)*((x1 - xa)^2 + (y1 - ya)^2)) + 10*n*(xi - xa)/(log(10)*((xi - xa)^2 + (yi - ya)^2))
1210                 //  = 10*n*(-(x1 - xa)/(log(10)*d1a^2) + (xi - xa)/(log(10)*dia^2))
1211 
1212                 //diff(Prdiff1a)/diff(ya) = 5*n/(log(10)*((x1 - xa)^2 + (y1 - ya)^2))*-2*(y1 - ya) -5*n/(log(10)*((xi - xa)^2 + (yi - ya)^2))*-2*(yi - ya) =
1213                 //  = -10*n*(y1 - ya)/(log(10)*((x1 - xa)^2 + (y1 - ya)^2)) + 10*n*(yi - ya)/(log(10)*((xi - xa)^2 + (yi - ya)^2))
1214                 //  = 10*n*(-(y1 - ya)/(log(10)*d1a^2) + (xi - xa)/(log(10)*dia^2))
1215 
1216                 //for other radio source pb=(xb,yb)
1217                 //diff(Prdiff1a)/diff(xb) = diff(Prdiff1a)/diff(yb) = 0
1218 
1219                 //For 3D:
1220                 //-------
1221 
1222                 //Prdiff1a = Pr(pi) - Pr(p1) = 5*n*log(d1a^2) - 5*n*log(dia^2) =
1223                 //  = 5*n*log((x1 - xa)^2 + (y1 - ya)^2 + (z1 - za)^2) - 5*n*log((xi - xa)^2 + (yi - ya)^2 + (zi - za)^2)
1224 
1225                 //derivatives respect parameters being estimated (xi,yi,zi,xa,ya,za...,xM,yM,zM)
1226                 //for unknown point pi = (xi, yi,zi)
1227                 //diff(Prdiff1a)/diff(xi) = -5*n/(log(10)*((xi - xa)^2 + (yi - ya)^2 + (zi - za)^2))*2*(xi - xa)

1377     /**
1378      * Gets the total number of readings associated to provided radio source.
1379      * This method uses only current nearest fingerprints.
1380      *
1381      * @param source       radio source to be checked
1382      * @param centroid     centroid to be computed.
1383      * @param fingerprints fingerprints where search is made.
1384      * @return total number of readings associated to provided radio source.
1385      */
1386     private int totalReadingsForSource(
1387             final RadioSource source,
1388             final List<RssiFingerprintLocated<RadioSource, RssiReading<RadioSource>, P>> fingerprints,
1389             final P centroid) {
1390         if (source == null) {
1391             return 0;
1392         }
1393 
1394         final int dims = getNumberOfDimensions();
1395 
1396         int result = 0;
1397         final double[] centroidCoords = centroid != null ?
1398                 new double[dims] : null;
1399 
1400         for (final RssiFingerprintLocated<RadioSource, RssiReading<RadioSource>, P> fingerprint : fingerprints) {
1401             final List<RssiReading<RadioSource>> readings = fingerprint.getReadings();
1402             if (readings == null) {
1403                 continue;
1404             }
1405 
1406             final P fingerprintPosition = fingerprint.getPosition();
1407 
1408             for (final RssiReading<RadioSource> reading : readings) {
1409                 final RadioSource readingSource = reading.getSource();
1410                 if (readingSource != null && readingSource.equals(source)) {
1411                     result++;
1412 
1413                     if (centroid != null) {
1414                         for (int i = 0; i < dims; i++) {
1415                             final double coord = fingerprintPosition.getInhomogeneousCoordinate(i);
1416                             centroidCoords[i] += coord;
1417                         }
1418                     }
1419                 }
1420             }
1421         }
1422 
1423         if (centroid != null) {
1424             for (int i = 0; i < dims; i++) {
1425                 centroidCoords[i] /= result;
1426                 centroid.setInhomogeneousCoordinate(i, centroidCoords[i]);
1427             }
1428         }
1429 
1430         return result;
