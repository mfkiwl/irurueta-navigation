16  package com.irurueta.navigation.indoor;
17  
18  import com.irurueta.geometry.Point;
19  
20  import java.util.ArrayList;
21  import java.util.Collection;
22  import java.util.List;
23  
24  /**
25   * Finds k-nearest radio source fingerprints based on their signal euclidean distances
26   * (not their actual location) where mean values are removed to account for possible
27   * biases when measuring from different devices.
28   * Typically this class should be preferred over {@link RadioSourceKNearestFinder}.
29   * @param <P> a {@link Point} type.
30   * @param <S> a {@link RadioSource type}.
31   */
32  @SuppressWarnings("WeakerAccess")
33  public class RadioSourceNoMeanKNearestFinder<P extends Point<?>, S extends RadioSource> {
34  
35      /**
36       * Collection of fingerprints to match against.
37       */
38      private Collection<? extends RssiFingerprintLocated<S, RssiReading<S>, P>> mFingerprints;
39  
40      /**
41       * Constructor.
42       * @param fingerprints collection of fingerprints to match against.
43       * @throws IllegalArgumentException if collection of fingerprints is null.
44       */
45      public RadioSourceNoMeanKNearestFinder(
46              Collection<? extends RssiFingerprintLocated<S, RssiReading<S>, P>> fingerprints) {
47          if (fingerprints == null) {
48              throw new IllegalArgumentException();
49          }
50          mFingerprints = fingerprints;
51      }
52  
53      /**
54       * Finds nearest fingerprint to provided one, in terms of signal euclidean distances
55       * (with removed signal means), within the collection of provided fingerprints.
56       * @param fingerprint fingerprint to find the nearest to.
57       * @return nearest fingerprint or null if none could be found.
58       */
59      public RssiFingerprintLocated<S, RssiReading<S>, P> findNearestTo(RssiFingerprint<S, RssiReading<S>> fingerprint) {
60          return findNearestTo(fingerprint, mFingerprints);
61      }
62  
63      /**
64       * Finds k-nearest fingerprints to provided one, in terms of signal euclidean distances
65       * (with removed signal means), within the collection of provided fingerprints.
66       * @param fingerprint fingerprint to find the k-nearest ones to.
67       * @param k number of nearest fingerprints to find.
68       * @return nearest fingerprints ordered from closest to farthest or an empty list if none could be found.
69       * @throws IllegalArgumentException if either fingerprint is null or k is les than 1.
70       */
71      public List<RssiFingerprintLocated<S, RssiReading<S>, P>> findKNearestTo(
72              RssiFingerprint<S, RssiReading<S>> fingerprint, int k) {
73          return findKNearestTo(fingerprint, mFingerprints, k);
74      }
75  
76      /**
77       * Finds k-nearest fingerprints to provided one, in terms of signal euclidean distances
78       * (with removed signal means), within the collection of provided fingerprints.
79       * @param fingerprint fingerprint to find the k-nearest ones to.
80       * @param k number of nearest fingerprints to find.
81       * @param nearestFingerprints list where found nearest fingerprints will be stored ordered from closest to farthest
82       *                            or an empty list if none could be found.
83       * @param nearestSqrDistances list where squared signal euclidean distances corresponding to found fingerprints will
84       *                            be stored or an empty list if no fingerprint is found.
85       * @throws IllegalArgumentException if any parameter is null or k is less than 1.
86       */
87      public void findKNearestTo(RssiFingerprint<S, RssiReading<S>> fingerprint, int k,
88              List<RssiFingerprintLocated<S, RssiReading<S>, P>> nearestFingerprints,
89              List<Double> nearestSqrDistances) {
90          findKNearestTo(fingerprint, mFingerprints, k, nearestFingerprints, nearestSqrDistances);
91      }
92  
93      /**
94       * Gets collection of fingerprints to match against.
95       * @return collection of fingerprints to match against.
96       */
97      public Collection<? extends RssiFingerprintLocated<S, RssiReading<S>, P>> getFingerprints() {
98          return mFingerprints;
99      }
100 
101     /**
102      * Finds nearest fingerprint to provided one, in terms of signal euclidean distances
103      * (with removed signal means), within the collection of provided fingerprints.
104      * @param fingerprint fingerprint to find the nearest to.
105      * @param fingerprints collection of fingerprints to make the search for the nearest one.
106      * @return nearest fingerprint or null if none could be found.
107      * @throws IllegalArgumentException if either fingerprint or collection of fingerprints is null.
108      * @param <P> a {@link Point} type.
109      * @param <S> a {@link RadioSource} type.
110      */
111     @SuppressWarnings("Duplicates")
112     public static <P extends Point<?>, S extends RadioSource> RssiFingerprintLocated<S, RssiReading<S>, P>
113             findNearestTo(RssiFingerprint<S, RssiReading<S>> fingerprint,
114             Collection<? extends RssiFingerprintLocated<S, RssiReading<S>, P>> fingerprints) {
115         if (fingerprint == null || fingerprints == null) {
116             throw new IllegalArgumentException();
117         }
118 
119         double bestSqrDist = Double.MAX_VALUE;
120         RssiFingerprintLocated<S, RssiReading<S>, P> result = null;
121         for(RssiFingerprintLocated<S, RssiReading<S>, P> f : fingerprints) {
122             double sqrDist = f.noMeanSqrDistanceTo(fingerprint);
123             if (sqrDist < bestSqrDist) {
124                 bestSqrDist = sqrDist;
125                 result = f;
126             }
127         }
128 
129         return result;
130     }
131 
132     /**
133      * Finds k-nearest fingerprints to provided one, in terms of signal euclidean distances, within the collection
134      * of provided fingerprints.
135      * @param fingerprint fingerprint to find the k-nearest ones to.
136      * @param fingerprints collection of fingerprints to make the search for the nearest ones.
137      * @param k number of nearest fingerprints to find.
138      * @return nearest fingerprints ordered from closest to farthest or an empty list if none could be found.
139      * @throws IllegalArgumentException if either fingerprint or collection of fingerprints is null, or k is less than
140      * 1.
141      * @param <P> a {@link Point} type.
142      * @param <S> a {@link RadioSource} type.
143      */
144     public static <P extends Point<?>, S extends RadioSource> List<RssiFingerprintLocated<S, RssiReading<S>, P>>
145             findKNearestTo(RssiFingerprint<S, RssiReading<S>> fingerprint,
146             Collection<? extends RssiFingerprintLocated<S, RssiReading<S>, P>> fingerprints,
147             int k) {
148 
149         List<RssiFingerprintLocated<S, RssiReading<S>, P>> result =
150                 new ArrayList<>();
151         List<Double> nearestSqrDistances = new ArrayList<>();
152         findKNearestTo(fingerprint, fingerprints, k, result, nearestSqrDistances);
153 
154         return result;
155     }
156 
157     /**
158      * Finds k-nearest fingerprints to provided one, in terms of signal euclidean distances, within the collection
159      * of provided fingerprints.
160      * @param fingerprint fingerprint to find the k-nearest ones to.
161      * @param fingerprints collection of fingerprints ot make the search for the nearest ones.
162      * @param k number of nearest fingerprints to find.
163      * @param nearestFingerprints list where found nearest fingerprints will be stored ordered from closest to farthest
164      *                            or an empty list if none could be found.
165      * @param nearestSqrDistances list where squared signal euclidean distances corresponding to found fingerprints will
166      *                            be stored or an empty list if no fingerprint is found.
167      * @throws IllegalArgumentException if any parameter is null or k is less than 1.
168      * @param <P> a {@link Point} type.
169      * @param <S> a {@link RadioSource} type.
170      */
171     @SuppressWarnings("Duplicates")
172     public static <P extends Point<?>, S extends RadioSource> void findKNearestTo(
173             RssiFingerprint<S, RssiReading<S>> fingerprint,
174             Collection<? extends RssiFingerprintLocated<S, RssiReading<S>, P>> fingerprints,
175             int k, List<RssiFingerprintLocated<S, RssiReading<S>, P>> nearestFingerprints,
176             List<Double> nearestSqrDistances) {
177 
178         if (fingerprint == null || fingerprints == null || k < 1 || nearestFingerprints == null ||
179                 nearestSqrDistances == null) {
180             throw new IllegalArgumentException();
181         }
182 
183         nearestSqrDistances.clear();
184         nearestFingerprints.clear();
185 
186         double maxSqrDist = Double.MAX_VALUE;
187         for (RssiFingerprintLocated<S, RssiReading<S>, P> f : fingerprints) {
188             double sqrDist = f.noMeanSqrDistanceTo(fingerprint);
189             if (sqrDist < maxSqrDist || nearestSqrDistances.size() < k) {
190 
191                 //find insertion point
192                 int pos = -1;
193                 int i = 0;
194                 for(Double sd : nearestSqrDistances) {
195                     if (sqrDist < sd) {
196                         //insertion point found
197                         pos = i;
198                         break;
199                     }
200                     i++;
201                 }
202 
203                 if (pos >= 0) {
204                     nearestSqrDistances.add(pos, sqrDist);
205                     nearestFingerprints.add(pos, f);
206                 } else {
207                     nearestSqrDistances.add(sqrDist);
208                     nearestFingerprints.add(f);
209                 }
210 
211                 //remove results exceeding required number of k neighbours to be found
212                 if (nearestFingerprints.size() > k) {
213                     nearestSqrDistances.remove(k);
214                     nearestFingerprints.remove(k);
215                 }
216 
217                 //update maxSqrDist to the largest squared distance value contained in result list distances
218                 maxSqrDist = nearestSqrDistances.get(nearestSqrDistances.size() - 1);
219             }
220         }
221     }
