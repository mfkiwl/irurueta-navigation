14  /**
15   * Contains a coordinate transformation matrix, or rotation matrix. The coordinate transformation matrix is a 3x3
16   * matrix where a vector may be transformed in one step from one set of resolving axes to another by pre-multiplying it
17   * by the appropriate coordinate transformation matrix. The coordinate transformation matrix is associated to a source
18   * and destination frame.
19   */

168     /**
169      * Determines whether provided matrix is a valid rotation matrix (3x3 and orthonormal)
170      * up to provided threshold.
171      *
172      * @param matrix    matrix to be checked.
173      * @param threshold threshold to determine whether matrix is valid.
174      * @return true if matrix is valid, false otherwise.
175      * @throws IllegalArgumentException if provided threshold value is negative.
176      */
177     public static boolean isValidMatrix(final Matrix matrix, final double threshold) {
178         if (threshold < Rotation3D.MIN_THRESHOLD) {
179             throw new IllegalArgumentException();
180         }
181 
182         try {
183             //TODO: use Rotation3D.isValidRotationMatrix when geometry library is updated
184             // instead of Utils.isOrthogonal and checking determinant (note that size must also
185             // be checked anyways)
186             return matrix.getRows() == Rotation3D.INHOM_COORDS
187                     && matrix.getColumns() == Rotation3D.INHOM_COORDS
188                     && Utils.isOrthogonal(matrix, threshold) &&
189                     Math.abs(Math.abs(Utils.det(matrix)) - 1.0) < threshold;
190         } catch (final AlgebraException e) {
191             return false;
192         }
193     }

201      */
202     public static boolean isValidMatrix(final Matrix matrix) {

355     /**
356      * Computes the inverse of this coordinate transformation matrix and stores the result into provided instance.
357      *
358      * @param result instance where inverse will be stored.
359      */
360     public void inverse(final CoordinateTransformationMatrix result) {
361         try {
362             final FrameType source = mSourceType;
363             final FrameType destination = mDestinationType;
364             final Matrix matrix = Matrix.identity(ROWS, COLS);
365             matrix.copyFrom(mMatrix);
366 
367             result.setSourceType(destination);
368             result.setDestinationType(source);
369 
370             // Because matrix needs to be a rotation (3x3 and orthonormal), its inverse is the transpose
371             matrix.transpose();
372             result.setMatrix(matrix);
373         } catch (WrongSizeException | InvalidRotationMatrixException ignore) {
374             // never happens
375         }
376     }

396     /**
397      * Computes matrix to convert ECEF to NED coordinates.
398      *
399      * @param latitude  latitude expressed in radians.
400      * @param longitude longitude expressed in radians.
401      * @param result    instance where computed matrix will be stored.
402      */
403     public static void ecefToNedMatrix(final double latitude, final double longitude, final Matrix result) {
404         if (result.getRows() != ROWS || result.getColumns() != COLS) {
405             try {
406                 result.resize(ROWS, COLS);
407             } catch (final WrongSizeException ignore) {
408             }
409         }
410         // Calculate ECEF to NED coordinate transformation matrix using (2.150)
411         final double cosLat = Math.cos(latitude);
412         final double sinLat = Math.sin(latitude);
413         final double cosLong = Math.cos(longitude);
414         final double sinLong = Math.sin(longitude);
415 
416         result.setElementAtIndex(0, -sinLat * cosLong);
417         result.setElementAtIndex(1, -sinLong);
418         result.setElementAtIndex(2, -cosLat * cosLong);
419 
420         result.setElementAtIndex(3, -sinLat * sinLong);
421         result.setElementAtIndex(4, cosLong);
422         result.setElementAtIndex(5, -cosLat * sinLong);
423 
424         result.setElementAtIndex(6, cosLat);
425         result.setElementAtIndex(7, 0.0);
426         result.setElementAtIndex(8, -sinLat);
427     }

482      * Computes matrix to convert NED to ECEF coordinates.
483      *
484      * @param latitude  latitude expressed in radians.
485      * @param longitude longitude expressed in radians.
486      * @param result    instance where computed matrix will be stored.
487      */
488     public static void nedToEcefMatrix(final double latitude, final double longitude, final Matrix result) {
489         //NED to ECEF matrix is the inverse of ECEF to NED matrix.
490         //Since ECEF to NED matrix is a rotation (3x3 and orthonormal), its inverse is the transpose.
491         ecefToNedMatrix(latitude, longitude, result);
492         result.transpose();
493     }
