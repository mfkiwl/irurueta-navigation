13  /**
14   * Converts from NED frame to ECEF frame.
15   * This implementation is based on the equations defined in "Principles of GNSS, Inertial, and Multisensor
16   * Integrated Navigation Systems, Second Edition" and on the companion software available at:
17   * https://github.com/ymjdz/InertialDemoECEF
18   */
19  @SuppressWarnings("WeakerAccess")
20  public class NEDtoECEFFrameConverter implements FrameConverter<NEDFrame, ECEFFrame> {
21  
22      /**
23       * The equatorial radius of WGS84 ellipsoid (6378137 m) defining Earth's shape.
24       */
25      public static final double EARTH_EQUATORIAL_RADIUS_WGS84 = Constants.EARTH_EQUATORIAL_RADIUS_WGS84;
26  
27      /**
28       * Earth eccentricity as defined on the WGS84 ellipsoid.
29       */
30      public static final double EARTH_ECCENTRICITY = Constants.EARTH_ECCENTRICITY;
31  
32      /**
33       * Converts source NED frame to a new ECEF frame instance.
34       *
35       * @param source source frame to convert from.
36       * @return a new destination frame instance.
37       */
38      @Override
39      public ECEFFrame convertAndReturnNew(final NEDFrame source) {
40          final ECEFFrame result = new ECEFFrame();
41          convert(source, result);
42          return result;
43      }
44  
45      /**
46       * Converts source NED frame to destination ECEF frame.
47       * @param source      source frame to convert from.
48       * @param destination destination frame instance to convert to.
49       */
50      @Override
51      public void convert(final NEDFrame source, final ECEFFrame destination) {
52          convertNEDtoECEF(source, destination);
53      }
54  
55      /**
56       * Gets source frame type.
57       *
58       * @return source frame type.
59       */
60      @Override
61      public FrameType getSourceType() {
62          return FrameType.LOCAL_NAVIGATION_FRAME;
63      }
64  
65      /**
66       * Gets destination frame type.
67       *
68       * @return destination frame type.
69       */
70      @Override
71      public FrameType getDestinationType() {
72          return FrameType.EARTH_CENTERED_EARTH_FIXED_FRAME;
73      }
74  
75      /**
76       * Converts source NED frame to a new ECEF frame instance.
77       *
78       * @param source source frame to convert from.
79       * @return a new destination frame instance.
80       */
81      public static ECEFFrame convertNEDtoECEFAndReturnNew(final NEDFrame source) {
82          final ECEFFrame result = new ECEFFrame();
83          convertNEDtoECEF(source, result);
84          return result;
85      }
86  
87      /**
88       * Converts source NED frame to destination ECEF frame.
89       * @param source      source frame to convert from.
90       * @param destination destination frame instance to convert to.
91       */
92      public static void convertNEDtoECEF(final NEDFrame source, final ECEFFrame destination) {
93          try {
94              final double latitude = source.getLatitude();
95              final double longitude = source.getLongitude();
96              final double height = source.getHeight();
97  
98              final double cosLat = Math.cos(latitude);
99              final double sinLat = Math.sin(latitude);
100             final double cosLong = Math.cos(longitude);
101             final double sinLong = Math.sin(longitude);
102 
103             // Calculate transverse radius of curvature using (2.105)
104             final double eSinLat = EARTH_ECCENTRICITY * sinLat;
105             final double eSinLat2 = eSinLat * eSinLat;
106             double re = EARTH_EQUATORIAL_RADIUS_WGS84 / Math.sqrt(1.0 - eSinLat2);
107 
108             // Convert position using (2.112)
109             final double e2 = EARTH_ECCENTRICITY * EARTH_ECCENTRICITY;
110             final double x = (re + height) * cosLat * cosLong;
111             final double y = (re + height) * cosLat * sinLong;
112             final double z = ((1.0 - e2) * re + height) * sinLat;
113 
114             // Calculate NED to ECEF coordinate transformation matrix
115             final Matrix cne = CoordinateTransformation.nedToEcefMatrix(latitude, longitude);
116 
117             // Transform velocity using (2.73)
118             final double vn = source.getVn();
119             final double ve = source.getVe();
120             final double vd = source.getVd();
121             final Matrix vEbn = new Matrix(NEDFrame.NUM_VELOCITY_COORDINATES, 1);
122             vEbn.setElementAtIndex(0, vn);
123             vEbn.setElementAtIndex(1, ve);
124             vEbn.setElementAtIndex(2, vd);
125 
126             final Matrix vEbe = cne.multiplyAndReturnNew(vEbn);
127             final double vx = vEbe.getElementAtIndex(0);
128             final double vy = vEbe.getElementAtIndex(1);
129             final double vz = vEbe.getElementAtIndex(2);
130 
131             // Transform attitude using (2.15)
132             final Matrix cbn = source.getCoordinateTransformation().getMatrix();
133             cne.multiply(cbn); // cne is now cbe
134 
135             final CoordinateTransformation c = new CoordinateTransformation(cne, FrameType.BODY_FRAME,
136                     FrameType.EARTH_CENTERED_EARTH_FIXED_FRAME);
137 
138             // set result
139             destination.setX(x);
140             destination.setY(y);
141             destination.setZ(z);
142 
143             destination.setVx(vx);
144             destination.setVy(vy);
145             destination.setVz(vz);
146 
147             destination.setCoordinateTransformation(c);
148 
149         } catch (WrongSizeException | InvalidRotationMatrixException |
150                 InvalidSourceAndDestinationFrameTypeException ignore) {
151             // never happens
152         }
153     }
