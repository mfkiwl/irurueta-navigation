16  package com.irurueta.navigation.gnss;
17  
18  import com.irurueta.algebra.AlgebraException;
19  import com.irurueta.algebra.Matrix;
20  import com.irurueta.algebra.Utils;
21  import com.irurueta.navigation.frames.CoordinateTransformation;
22  import com.irurueta.navigation.geodesic.Constants;
23  
24  import java.util.Collection;
25  
26  
27  /**
28   * Implements one cycle of the GNSS extended Kalman filter.
29   * This implementation is based on the equations defined in "Principles of GNSS, Inertial, and Multisensor
30   * Integrated Navigation Systems, Second Edition" and on the companion software available at:
31   * https://github.com/ymjdz/MATLAB-Codes
32   */
33  public class GNSSKalmanEpochEstimator {
34  
35      /**
36       * Speed of light in the vacuum expressed in meters per second (m/s).
37       */
38      public static final double SPEED_OF_LIGHT = 299792458;
39  
40      /**
41       * Earth rotation rate expressed in radians per second (rad/s).
42       */
43      public static final double EARTH_ROTATION_RATE = Constants.EARTH_ROTATION_RATE;
44  
45      /**
46       * Number of rows and columns of transition and system noise covariance matrices.
47       */
48      private static final int MATRIX_SIZE = 8;
49  
50      /**
51       * Estimates the update of Kalman filter state and covariance matrix for a single epoch.
52       *
53       * @param measurements        satellite measurements data.
54       * @param propagationInterval propagation interval expressed in seconds (s).
55       * @param previousState       previous Kalman filter state estimates.
56       * @param previousCovariance  previous Kalman filter error covariance matrix.
57       * @param config              system configuration (usually obtained through calibration).
58       * @param updatedState        instance where updated Kalman filter state will be stored after executing this method.
59       * @param updatedCovariance   instance where updated Kalman filter error covariance matrix will be stored.
60       * @throws IllegalArgumentException if provided previous covariance matrix is not 8x8.
61       */
62      public static void estimate(final Collection<GNSSMeasurement> measurements, final double propagationInterval,
63                                  final GNSSKalmanState previousState, final Matrix previousCovariance,
64                                  final GNSSKalmanConfig config, GNSSKalmanState updatedState,
65                                  final Matrix updatedCovariance) throws AlgebraException {
66  
67          if (previousCovariance.getRows() != GNSSKalmanState.NUM_PARAMETERS ||
68                  previousCovariance.getColumns() != GNSSKalmanState.NUM_PARAMETERS) {
69              throw new IllegalArgumentException();
70          }
71  
72          // SYSTEM PROPAGATION PHASE
73  
74          // 1. Determine transition matrix using (9.147) and (9.150)
75          final Matrix phiMatrix = Matrix.identity(MATRIX_SIZE, MATRIX_SIZE);
76          phiMatrix.setElementAt(0, 3, propagationInterval);
77          phiMatrix.setElementAt(1, 4, propagationInterval);
78          phiMatrix.setElementAt(2, 5, propagationInterval);
79          phiMatrix.setElementAt(6, 7, propagationInterval);
80  
81          // 2. Determine system noise covariance matrix using (9.152)
82          final double propagationInterval2 = propagationInterval * propagationInterval;
83          final double propagationInterval3 = propagationInterval2 * propagationInterval;
84          final double accelerationPSD = config.getAccelerationPSD();
85          final double clockFrequencyPSD = config.getClockFrequencyPSD();
86          final double clockPhasePSD = config.getClockPhasePSD();
87  
88          final double value1 = accelerationPSD * propagationInterval3 / 3.0;
89          final double value2 = accelerationPSD * propagationInterval2 / 2.0;
90          final double value3 = accelerationPSD * propagationInterval;
91          final double value4 = clockFrequencyPSD * propagationInterval3 / 3.0 + clockPhasePSD * propagationInterval;
92          final double value5 = clockFrequencyPSD * propagationInterval2 / 2.0;
93          final double value6 = clockFrequencyPSD * propagationInterval;
94  
95          final Matrix qMatrix = new Matrix(MATRIX_SIZE, MATRIX_SIZE);
96          qMatrix.setElementAt(0, 0, value1);
97          qMatrix.setElementAt(1, 1, value1);
98          qMatrix.setElementAt(2, 2, value1);
99  
100         qMatrix.setElementAt(0, 3, value2);
101         qMatrix.setElementAt(1, 4, value2);
102         qMatrix.setElementAt(2, 5, value2);
103 
104         qMatrix.setElementAt(3, 0, value2);
105         qMatrix.setElementAt(4, 1, value2);
106         qMatrix.setElementAt(5, 2, value2);
107 
108         qMatrix.setElementAt(3, 3, value3);
109         qMatrix.setElementAt(4, 4, value3);
110         qMatrix.setElementAt(5, 5, value3);
111 
112         qMatrix.setElementAt(6, 6, value4);
113         qMatrix.setElementAt(6, 7, value5);
114         qMatrix.setElementAt(7, 6, value5);
115         qMatrix.setElementAt(7, 7, value6);
116 
117 
118         // 3. Propagate state estimates using (3.14)
119         final Matrix xEstOld = previousState.asMatrix();
120         final Matrix xEstPropagated = phiMatrix.multiplyAndReturnNew(xEstOld);
121         final Matrix propagatedVelocity = xEstPropagated.getSubmatrix(
122                 3, 0, 5, 0);
123         final Matrix propagatedPosition = xEstPropagated.getSubmatrix(
124                 0, 0, 2, 0);
125 
126         // 4. Propagate state estimation error covariance matrix using (3.15)
127         final Matrix pMatrixPropagated = phiMatrix.multiplyAndReturnNew(previousCovariance);
128         phiMatrix.transpose();
129         pMatrixPropagated.multiply(phiMatrix);
130         pMatrixPropagated.add(qMatrix);
131 
132         // MEASUREMENT UPDATE PHASE
133 
134         // Skew symmetric matrix of Earth rate
135         final Matrix omegaIe = Utils.skewMatrix(new double[]{0.0, 0.0, EARTH_ROTATION_RATE});
136 
137         final int numberOfMeasurements = measurements.size();
138         final Matrix uAseT = new Matrix(numberOfMeasurements, 3);
139         final Matrix predMeas = new Matrix(numberOfMeasurements, 2);
140 
141         final Matrix cei = Matrix.identity(CoordinateTransformation.ROWS,
142                 CoordinateTransformation.COLS);
143         final Matrix satellitePosition = new Matrix(CoordinateTransformation.ROWS, 1);
144         final Matrix satelliteVelocity = new Matrix(CoordinateTransformation.ROWS, 1);
145         final Matrix deltaR = new Matrix(CoordinateTransformation.ROWS, 1);
146         final Matrix tmp1 = new Matrix(CoordinateTransformation.ROWS, 1);
147         final Matrix tmp2 = new Matrix(CoordinateTransformation.ROWS, 1);
148         final Matrix tmp3 = new Matrix(CoordinateTransformation.ROWS, 1);
149         final Matrix tmp4 = new Matrix(CoordinateTransformation.ROWS, 1);
150         final Matrix tmp5 = new Matrix(CoordinateTransformation.ROWS, 1);
151         final Matrix tmp6 = new Matrix(CoordinateTransformation.ROWS, 1);
152         final Matrix tmp7 = new Matrix(1, CoordinateTransformation.ROWS);
153 
154         // Loop measurements
155         int j = 0;
156         for (final GNSSMeasurement measurement : measurements) {
157             // Predict approx range
158             final double deltaX = measurement.getX() - xEstPropagated.getElementAtIndex(0);
159             final double deltaY = measurement.getY() - xEstPropagated.getElementAtIndex(1);
160             final double deltaZ = measurement.getZ() - xEstPropagated.getElementAtIndex(2);
161             final double approxRange = Math.sqrt(deltaX * deltaX + deltaY * deltaY + deltaZ * deltaZ);
162 
163             // Calculate frame rotation during signal transit time using (8.36)
164             final double ceiValue = EARTH_ROTATION_RATE * approxRange / SPEED_OF_LIGHT;
165             cei.setElementAt(0, 1, ceiValue);
166             cei.setElementAt(1, 0, -ceiValue);
167 
168             // Predict pseudo-range using (9.165)
169             satellitePosition.setElementAtIndex(0, measurement.getX());
170             satellitePosition.setElementAtIndex(1, measurement.getY());
171             satellitePosition.setElementAtIndex(2, measurement.getZ());
172 
173             cei.multiply(satellitePosition, deltaR);
174             for (int i = 0; i < deltaR.getRows(); i++) {
175                 deltaR.setElementAtIndex(i, deltaR.getElementAtIndex(i)
176                         - xEstPropagated.getElementAtIndex(i));
177             }
178             final double range = Utils.normF(deltaR);
179 
180             predMeas.setElementAt(j, 0, range
181                     + xEstPropagated.getElementAtIndex(6));
182 
183             // Predict line of sight
184             for (int i = 0; i < deltaR.getRows(); i++) {
185                 uAseT.setElementAt(j, i, deltaR.getElementAtIndex(i) / range);
186             }
187 
188             // Predict pseudo-range rate using (9.165)
189             satelliteVelocity.setElementAtIndex(0, measurement.getVx());
190             satelliteVelocity.setElementAtIndex(1, measurement.getVy());
191             satelliteVelocity.setElementAtIndex(2, measurement.getVz());
192 
193             omegaIe.multiply(satellitePosition, tmp1);
194             satelliteVelocity.add(tmp1, tmp2);
195             cei.multiply(tmp2, tmp3);
196 
197             omegaIe.multiply(propagatedPosition, tmp4);
198             propagatedVelocity.add(tmp4, tmp6);
199 
200             tmp3.subtract(tmp6, tmp5);
201 
202             uAseT.getSubmatrix(j, 0, j, 2, tmp7);
203 
204             double rangeRate = Utils.dotProduct(tmp7, tmp5);
205 
206             predMeas.setElementAt(j, 1,
207                     rangeRate + xEstPropagated.getElementAtIndex(7));
208 
209             j++;
210         }
211 
212         // 5. Set-up measurement matrix using (9.163)
213         final Matrix hMatrix = new Matrix(2 * numberOfMeasurements, GNSSKalmanState.NUM_PARAMETERS);
214         for (int j1 = 0, j2 = numberOfMeasurements; j1 < numberOfMeasurements; j1++, j2++) {
215             for (int i1 = 0, i2 = 3; i1 < CoordinateTransformation.ROWS; i1++, i2++) {
216                 final double value = -uAseT.getElementAt(j1, i1);
217 
218                 hMatrix.setElementAt(j1, i1, value);
219                 hMatrix.setElementAt(j2, i2, value);
220             }
221             hMatrix.setElementAt(j1, 6, 1.0);
222             hMatrix.setElementAt(j2, 7, 1.0);
223         }
224 
225         // 6. Set-up measurement noise covariance matrix assuming all measurements are independent
226         // and have equal variance for a given measurement type
227         final double pseudoRangeSD = config.getPseudoRangeSD();
228         final double pseudoRangeSD2 = pseudoRangeSD * pseudoRangeSD;
229         final double rangeRateSD = config.getRangeRateSD();
230         final double rangeRateSD2 = rangeRateSD * rangeRateSD;
231         final Matrix rMatrix = new Matrix(2 * numberOfMeasurements, 2 * numberOfMeasurements);
232         for (int i1 = 0, i2 = numberOfMeasurements; i1 < numberOfMeasurements; i1++, i2++) {
233             rMatrix.setElementAt(i1, i1, pseudoRangeSD2);
234             rMatrix.setElementAt(i2, i2, rangeRateSD2);
235         }
236 
237         // 7. Calculate Kalman gain using (3.21)
238         final Matrix hMatrixTransposed = hMatrix.transposeAndReturnNew();
239         final Matrix tmp8 = hMatrix.multiplyAndReturnNew(
240                 pMatrixPropagated.multiplyAndReturnNew(hMatrixTransposed));
241         tmp8.add(rMatrix);
242         final Matrix tmp9 = Utils.inverse(tmp8);
243         final Matrix kMatrix = pMatrixPropagated.multiplyAndReturnNew(hMatrixTransposed);
244         kMatrix.multiply(tmp9);
245 
246         // 8. Formulate measurement innovations using (3.88)
247         final Matrix deltaZ = new Matrix(2 * numberOfMeasurements, 1);
248         int i1 = 0;
249         int i2 = numberOfMeasurements;
250         for (final GNSSMeasurement measurement : measurements) {
251             deltaZ.setElementAtIndex(i1, measurement.getPseudoRange()
252                     - predMeas.getElementAt(i1, 0));
253             deltaZ.setElementAtIndex(i2, measurement.getPseudoRate()
254                     - predMeas.getElementAt(i1, 1));
255             i1++;
256             i2++;
257         }
258 
259         // 9. Update state estimates using (3.24)
260         xEstPropagated.add(kMatrix.multiplyAndReturnNew(deltaZ));
261 
262         // xEstPropagated now contains updated state
263         updatedState.fromMatrix(xEstPropagated);
264 
265         // 10. Update state estimation error covariance matrix using (3.25)
266         if (updatedCovariance.getRows() != GNSSKalmanState.NUM_PARAMETERS ||
267                 updatedCovariance.getColumns() != GNSSKalmanState.NUM_PARAMETERS) {
268             updatedCovariance.resize(GNSSKalmanState.NUM_PARAMETERS,
269                     GNSSKalmanState.NUM_PARAMETERS);
270         }
271         Matrix.identity(updatedCovariance);
272         kMatrix.multiply(hMatrix);
273         updatedCovariance.subtract(kMatrix);
274         updatedCovariance.multiply(pMatrixPropagated);
275     }
276 }
