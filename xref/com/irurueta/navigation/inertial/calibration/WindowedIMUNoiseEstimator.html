1   package com.irurueta.navigation.inertial.calibration;
2   
3   import com.irurueta.navigation.inertial.BodyKinematics;
4   
5   import java.util.LinkedList;
6   
7   public class WindowedIMUNoiseEstimator {
8   
9       /**
10       * Number of samples to keep within the window by default.
11       * For an IMU generating 100 samples/second, this is equivalent to roughly
12       * 1 second.
13       * For an IMU generating 50 samples/second this is equivalent to roughly
14       * 2 seconds.
15       */
16      public static final int DEFAULT_WINDOW_SIZE = 101;
17  
18      /**
19       * Minimum allowed window size.
20       */
21      public static final int MIN_WINDOW_SIZE = 2;
22  
23      /**
24       * Default time interval between body kinematics samples expressed in seconds
25       * (s).
26       */
27      public static final double DEFAULT_TIME_INTERVAL_SECONDS = 0.02;
28  
29      /**
30       * Length of number of samples to keep within the window being processed.
31       * Window size must always be odd (so that a center sample exists), and
32       * must always be larger than allowed minimum value.
33       */
34      private int mWindowSize = DEFAULT_WINDOW_SIZE;
35  
36      /**
37       * Time interval expressed in seconds (s) between consecutive body kinematics
38       * samples.
39       */
40      private double mTimeInterval = DEFAULT_TIME_INTERVAL_SECONDS;
41  
42      /**
43       * Keeps the list of IMU samples that remain within the window.
44       */
45      private LinkedList<BodyKinematics> mWindowedSamples = new LinkedList<>();
46  
47      /**
48       * Contains estimated average of x coordinate of accelerometer sensed
49       * specific force within current window expressed in meters per squared
50       * second (m/s^2).
51       */
52      private double mAvgFx;
53  
54      /**
55       * Contains estimated average of y coordinate of accelerometer sensed
56       * specific force within current window expressed in meters per squared
57       * second (m/s^2).
58       */
59      private double mAvgFy;
60  
61      /**
62       * Contains estimated average of z coordinate of accelerometer sensed
63       * specific force within current window expressed in meters per squared
64       * second (m/s^2).
65       */
66      private double mAvgFz;
67  
68      /**
69       * Contains estimated average of x coordinate of gyroscope sensed angular
70       * rate within current window expressed in radians per second (rad/s).
71       */
72      private double mAvgAngularRateX;
73  
74      /**
75       * Contains estimated average of y coordinate of gyroscope sensed angular
76       * rate within current window expressed in radians per second (rad/s).
77       */
78      private double mAvgAngularRateY;
79  
80      /**
81       * Contains estimated average of z coordinate of gyroscope sensed angular
82       * rate within current window expressed in radians per second (rad/s).
83       */
84      private double mAvgAngularRateZ;
85  
86      /**
87       * Contains estimated variance of x coordinate of accelerometer sensed
88       * specific force within current window expressed in (m^2/s^4).
89       */
90      private double mVarianceFx;
91  
92      /**
93       * Contains estimated variance of y coordinate of accelerometer sensed
94       * specific force within current window expressed in (m^2/s^4).
95       */
96      private double mVarianceFy;
97  
98      /**
99       * Contains estimated variance of z coordinate of accelerometer sensed
100      * specific force within current window expressed in (m^2/s^4).
101      */
102     private double mVarianceFz;
103 
104     /**
105      * Contains estimated variance of x coordinate of gyroscope sensed angular
106      * rate within current window expressed in (rad^2/s^2).
107      */
108     private double mVarianceAngularRateX;
109 
110     /**
111      * Contains estimated variance of y coordinate of gyroscope sensed angular
112      * rate within current window expressed in (rad^2/s^2).
113      */
114     private double mVarianceAngularRateY;
115 
116     /**
117      * Contains estimated variance of z coordinate of gyroscope sensed angular
118      * rate within current window expressed in (rad^2/s^2).
119      */
120     private double mVarianceAngularRateZ;
121 }
