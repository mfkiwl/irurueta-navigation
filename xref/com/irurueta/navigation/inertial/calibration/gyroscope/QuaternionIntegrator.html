28  /**
29   * Class in charge of performing integration steps of rotations.
30   * This implementation uses a Runge-Kutta integration algorithm to obtain
31   * accurate results on {@link EasyGyroscopeCalibrator}
32   */
33  public class QuaternionIntegrator {

35      /**
36       * Integrates a sequence of gyroscope measurements contained within timed body kinematics,
37       * starting at an initial attitude to obtain a final attitude.
38       *
39       * @param sequence        sequence of gyroscope measurements to be integrated.
40       * @param initialAttitude (optional) initial attitude to be used. If null, then the
41       *                        identity attitude will be used.
42       * @param result          resulting rotation after integration.
43       */
44      public static void integrateGyroSequence(
45              final BodyKinematicsSequence<StandardDeviationTimedBodyKinematics> sequence,
46              final Quaternion initialAttitude,
47              final Quaternion result) {
48  
49          if (initialAttitude != null) {
50              // if provided initial attitude
51              result.fromRotation(initialAttitude);
52          } else {
53              // if no initial attitude is provided, we use the identity
54              result.setA(1.0);
55              result.setB(0.0);
56              result.setC(0.0);
57              result.setD(0.0);
58          }
59  
60  
61          final List<StandardDeviationTimedBodyKinematics> sortedMeasurements =
62                  sequence.getSortedItems();
63  
64          boolean first = true;
65          double previousTimestamp = 0.0;
66          final double[] previousGyroData = new double[BodyKinematics.COMPONENTS];
67          final double[] gyroData = new double[BodyKinematics.COMPONENTS];
68          for (final StandardDeviationTimedBodyKinematics measurement : sortedMeasurements) {
69              final BodyKinematics kinematics = measurement.getKinematics();
70  
71              gyroData[0] = kinematics.getAngularRateX();
72              gyroData[1] = kinematics.getAngularRateY();
73              gyroData[2] = kinematics.getAngularRateZ();
74  
75              if (first) {
76                  previousTimestamp = measurement.getTimestampSeconds();
77  
78                  // copy gyroData to previousGyroData
79                  System.arraycopy(gyroData, 0, previousGyroData, 0, gyroData.length);
80  
81                  first = false;
82                  continue;
83              }
84  
85              final double timestamp = measurement.getTimestampSeconds();
86  
87  
88              final double dt = timestamp - previousTimestamp;
89  
90              quatIntegrationStepRK4(result, previousGyroData, gyroData, dt, result);
91  
92  
93              // prepare data for next iteration
94  
95              // copy gyroData to previousGyroData
96              System.arraycopy(gyroData, 0, previousGyroData, 0, gyroData.length);
97              // copy timestamp to previous timestamp
98              previousTimestamp = timestamp;
99          }
100     }

146     /**
147      * Normalize an input matrix corresponding to a quaterniont into a unit
148      * vector.
149      *
150      * @param quaternion matrix containing quaternion values to be normalized.
151      */
152     private static void normalizeQuaternion(final Matrix quaternion) {
153         final double norm = Utils.normF(quaternion);
154         quaternion.multiplyByScalar(1.0 / norm);
155     }

157     /**
158      * Gets skew symmetric matrix representation of angular speed.
159      *
160      * @param omega  matrix containing angular speed components. Must be 3x1.
161      * @param result computed skew symmetricmatrix representation. Must be 4x4.
162      */
163     private static void computeOmegaSkew(final Matrix omega, final Matrix result) {
164         final double omega0 = omega.getElementAtIndex(0);
165         final double omega1 = omega.getElementAtIndex(1);
166         final double omega2 = omega.getElementAtIndex(2);
167 
168         result.setElementAtIndex(0, 0.0);
169         result.setElementAtIndex(1, omega0);
170         result.setElementAtIndex(2, omega1);
171         result.setElementAtIndex(3, omega2);
172 
173         result.setElementAtIndex(4, -omega0);
174         result.setElementAtIndex(5, 0.0);
175         result.setElementAtIndex(6, -omega2);
176         result.setElementAtIndex(7, omega1);
177 
178         result.setElementAtIndex(8, -omega1);
179         result.setElementAtIndex(9, omega2);
180         result.setElementAtIndex(10, 0.0);
181         result.setElementAtIndex(11, -omega0);
182 
183         result.setElementAtIndex(12, -omega2);
184         result.setElementAtIndex(13, -omega1);
185         result.setElementAtIndex(14, omega0);
186         result.setElementAtIndex(15, 0.0);
187     }

189     /**
190      * Performs a RK4 Runge-Kutta integration step.
191      *
192      * @param quat   the input 4D vector representing the initial rotation as
193      *               quaternion values. Must be 4x1.
194      * @param omega0 initial rotational velocity at time t0 expressed in radians
195      *               per second (rad/s). Must be 3x1.
196      * @param omega1 final rotational velocity at time t1 expressed in radians
197      *               per second (rad/s). Must be 3x1.
198      * @param dt     time step expressed in seconds (t1 - t0).
199      * @param result resulting final rotation. Must be 4x1.
200      */
201     private static void quatIntegrationStepRK4(
202             final Matrix quat,
203             final Matrix omega0,
204             final Matrix omega1,
205             final double dt,
206             final Matrix result) {
207         try {
208             final Matrix omega01 = omega0.addAndReturnNew(omega1);
209             omega01.multiplyByScalar(0.5);
210 
211             final Matrix k1 = new Matrix(Quaternion.N_PARAMS, 1);
212             final Matrix k2 = new Matrix(Quaternion.N_PARAMS, 1);
213             final Matrix k3 = new Matrix(Quaternion.N_PARAMS, 1);
214             final Matrix k4 = new Matrix(Quaternion.N_PARAMS, 1);
215             final Matrix tmpQ = new Matrix(Quaternion.N_PARAMS, 1);
216 
217             final Matrix omegaSkew = new Matrix(
218                     Quaternion.N_PARAMS, Quaternion.N_PARAMS);
219 
220             // First Runge-Kutta coefficient
221             computeOmegaSkew(omega0, omegaSkew);
222             omegaSkew.multiply(quat, k1);
223             k1.multiplyByScalar(0.5);
224 
225             // Second Runge-Kutta coefficient
226             tmpQ.copyFrom(k1);
227             tmpQ.multiplyByScalar(0.5 * dt);
228             tmpQ.add(quat);
229             computeOmegaSkew(omega01, omegaSkew);
230             omegaSkew.multiply(tmpQ, k2);
231             k2.multiplyByScalar(0.5);
232 
233             // Third Runge-Kutta coefficient (same omega skew as second coeff.)
234             tmpQ.copyFrom(k2);
235             tmpQ.multiplyByScalar(0.5 * dt);
236             tmpQ.add(quat);
237             omegaSkew.multiply(tmpQ, k3);
238             k3.multiplyByScalar(0.5);
239 
240             // Forth Runge-Kutta coefficient
241             tmpQ.copyFrom(k3);
242             tmpQ.multiplyByScalar(dt);
243             tmpQ.add(quat);
244             computeOmegaSkew(omega1, omegaSkew);
245             omegaSkew.multiply(tmpQ, k4);
246             k4.multiplyByScalar(0.5);
247 
248             final double mult1 = 1.0 / 6.0;
249             final double mult2 = 1.0 / 3.0;
250 
251             // result = quat + dt * (mult1 * k1 + mult2 * k2 + mult2 * k3 + mult1 * k4)
252             k1.multiplyByScalar(mult1);
253             k2.multiplyByScalar(mult2);
254             k3.multiplyByScalar(mult2);
255             k4.multiplyByScalar(mult1);
256 
257             result.copyFrom(k1);
258             result.add(k2);
259             result.add(k3);
260             result.add(k4);
261             result.multiplyByScalar(dt);
262             result.add(quat);
263         } catch (final WrongSizeException ignore) {
264             // never happens
265         }
266 
267         normalizeQuaternion(result);
268     }

270     /**
271      * Performs a RK4 Runge-Kutta integration step.
272      *
273      * @param quaternion the initial rotation.
274      * @param omega0     initial rotation velocity at time t0 expressed in radians
275      *                   per second (rad/s). Must have length 3.
276      * @param omega1     final rotation velocity at time t1 expressed in radians
277      *                   per second (rad/s). Must have length 3.
278      * @param dt         time step expressed in seconds (t1 - t0).
279      * @param result     resulting final rotation.
280      */
281     private static void quatIntegrationStepRK4(
282             final Quaternion quaternion,
283             final double[] omega0,
284             final double[] omega1,
285             final double dt,
286             final Quaternion result) {
287         try {
288             final Matrix quat = new Matrix(
289                     Quaternion.N_PARAMS, 1);
290             quaternion.values(quat.getBuffer());
291 
292             final Matrix omega0b = Matrix.newFromArray(omega0);
293             final Matrix omega1b = Matrix.newFromArray(omega1);
294 
295             final Matrix quatResult = new Matrix(
296                     Quaternion.N_PARAMS, 1);
297 
298             quatIntegrationStepRK4(quat, omega0b, omega1b, dt, quatResult);
299             result.setValues(quatResult.getBuffer());
300 
301         } catch (WrongSizeException ignore) {
302             // never happens
303         }
304     }
