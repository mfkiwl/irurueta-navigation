1   package com.irurueta.navigation.inertial.calibration;
2   
3   /**
4    * Interface defining events generated by this detector.
5    */
6   public interface StaticIntervalDetectorListener {
7       /**
8        * Called when initial static period starts so that base noise level
9        * starts being estimated.
10       *
11       * @param detected detector that raised the event.
12       */
13      void onInitializationStarted(final StaticIntervalDetector detected);
14  
15      /**
16       * Called when initial static period successfully completes and base
17       * noise level is estimated so that static and dynamic periods can be
18       * detected.
19       *
20       * @param detected       detector that raised the event.
21       * @param baseNoiseLevel base accelerometer noise level expressed in meters per squared second (m/s^2).
22       */
23      void onInitializationCompleted(final StaticIntervalDetector detected,
24                                     final double baseNoiseLevel);
25  
26      /**
27       * Called when an error is detected during initialization because there has been
28       * excessive movement forces.
29       *
30       * @param detector                detector that raised the event.
31       * @param accumulatedNoiseLevel   accumulated noise level expressed in meters per squared second (m/s^2).
32       * @param instantaneousNoiseLevel instantaneous noise level expressed in meters per squared second (m/s^2).
33       * @param reason                  reason why error was detected.
34       */
35      void onError(final StaticIntervalDetector detector,
36                   final double accumulatedNoiseLevel,
37                   final double instantaneousNoiseLevel,
38                   final StaticIntervalDetector.ErrorReason reason);
39  
40      /**
41       * Called when a static interval has been detected after initialization.
42       *
43       * @param detector detector that raised the event.
44       */
45      void onStaticIntervalDetected(final StaticIntervalDetector detector);
46  
47      /**
48       * Called when a dynamic interval has been detected after initialization.
49       *
50       * @param detector detector that raised the event.
51       * @param avgSpecificForceX average x-coordinate of specific force during last
52       *                          static period expressed in meters per squared
53       *                          second (m/s^2).
54       * @param avgSpecificForceY average y-coordinate of specific force during last
55       *                          static period expressed in meters per squared
56       *                          second (m/s^2).
57       * @param avgSpecificForceZ average z-coordinate of specific force during last
58       *                          static period expressed in meters per squared
59       *                          second (m/s^2).
60       */
61      void onDynamicIntervalDetected(final StaticIntervalDetector detector,
62                                     final double avgSpecificForceX,
63                                     final double avgSpecificForceY,
64                                     final double avgSpecificForceZ);
65  
66      /**
67       * Called when detector is reset.
68       *
69       * @param detector detector that raised the event.
70       */
71      void onReset(final StaticIntervalDetector detector);
72  }
