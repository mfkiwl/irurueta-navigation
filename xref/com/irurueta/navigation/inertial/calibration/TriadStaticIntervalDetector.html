23  /**
24   * Abstract base class for detectors in charge of determining when a static period of
25   * measurements starts and finishes.
26   * Static periods are periods of time where the device is considered to
27   * remain static (no movement applied to it).
28   *
29   * @param <U> type of unit.
30   * @param <M> a type of measurement.
31   * @param <T> a triad type.
32   * @param <D> a detector type.
33   * @param <L> a listener type.
34   */
35  public abstract class TriadStaticIntervalDetector<U extends Enum<?>, M extends Measurement<U>,
36          T extends Triad<U, M>, D extends TriadStaticIntervalDetector<U, M, T, D, L>,
37          L extends TriadStaticIntervalDetectorListener<U, M, T, D>> {
38  
39      /**
40       * Number of samples to keep within the window by default.
41       * For a sensor generating 100 samples/second, this is equivalent to 1 second.
42       * For a sensor generating 50 samples/second, this is equivalent to 2 seconds.
43       */
44      public static final int DEFAULT_WINDOW_SIZE = WindowedTriadNoiseEstimator.DEFAULT_WINDOW_SIZE;
45  
46      /**
47       * Number of samples to process during the initial static period to determine the sensor
48       * (accelerometer, gyroscope or magnetometer) noise level.
49       * For a sensor generating 100 samples/second, this is equivalent to 50 seconds.
50       * For a sensor generating 50 samples/second, this is equivalent to 100 seconds.
51       */
52      public static final int DEFAULT_INITIAL_STATIC_SAMPLES = 5000;
53  
54      /**
55       * Minimum allowed number of samples to be processed during the initial static period.
56       */
57      public static final int MINIMUM_INITIAL_STATIC_SAMPLES = 2;
58  
59      /**
60       * Default factor to be applied to detected base noise level in order to determine
61       * threshold for static/dynamic period changes. This factor is unit-less.
62       */
63      public static final double DEFAULT_THRESHOLD_FACTOR = 2.0;
64  
65      /**
66       * Default factor to determine that a sudden movement has occurred during initialization
67       * if instantaneous noise level exceeds accumulated noise level by this factor amount.
68       * This factor is unit-less.
69       */
70      public static final double DEFAULT_INSTANTANEOUS_NOISE_LEVEL_FACTOR = 2.0;
71  
72      /**
73       * Default overall absolute threshold to determine whether there has been excessive motion
74       * during the whole initialization phase.
75       * This threshold is expressed in meters per squared second (m/s^2) for acceleration, radians
76       * per second (rad/s) for angular speed or Teslas (T) for magnetic flux density, and by
77       * default it is set to the maximum allowed value, thus effectively disabling this error
78       * condition check during initialization.
79       */
80      public static final double DEFAULT_BASE_NOISE_LEVEL_ABSOLUTE_THRESHOLD = Double.MAX_VALUE;
81  
82      /**
83       * Number of samples to keep in window to find instantaneous noise level averaged within
84       * the window of samples.
85       * Window size should contain about 1 or 2 seconds of data to be averaged to obtain
86       * a more reliable instantaneous noise level.
87       */
88      private int mWindowSize = DEFAULT_WINDOW_SIZE;

686     /**
687      * Processes a new measurement triad sample.
688      *
689      * @param triad a new measurement triad to be processed.
690      * @return true if provided triad has been processed, false if provided triad has been skipped because detector
691      * previously failed. If detector previously failed, it will need to be reset before processing additional
692      * samples.
693      * @throws LockedException if detector is busy processing a previous sample.
694      */
695     public boolean process(final T triad) throws LockedException {
696         return process(convertMeasurement(triad.getValueX(), triad.getUnit()),
697                 convertMeasurement(triad.getValueY(), triad.getUnit()),
698                 convertMeasurement(triad.getValueZ(), triad.getUnit()));
699     }

701     /**
702      * Processes a new measurement triad sample.
703      *
704      * @param valueX x-coordinate of sensed measurement.
705      * @param valueY y-coordinate of sensed measurement.
706      * @param valueZ z-coordinate of sensed measurement.
707      * @return true if provided triad has been processed, false if provided triad has been skipped because detector
708      * previously failed. If detector previously failed, it will need to be reset before processing additional
709      * samples.
710      * @throws LockedException if detector is busy processing a previous sample.
711      */
712     public boolean process(final M valueX, final M valueY, final M valueZ)
713             throws LockedException {
714         return process(convertMeasurement(valueX),
715                 convertMeasurement(valueY),
716                 convertMeasurement(valueZ));
717     }

719     /**
720      * Processes a new measurement triad sample.
721      * Provided measurement coordinates are expressed in meters per squared second (m/s^2) for acceleration,
722      * radians per second (rad/s) for angular speed or Teslas (T) for magnetic flux density.
723      *
724      * @param valueX x-coordinate of sensed measurement.
725      * @param valueY y-coordinate of sensed measurement.
726      * @param valueZ z-coordinate of sensed measurement.
727      * @return true if provided triad has been processed, false if provided triad has been skipped because detector
728      * previously failed. If detector previously failed, it will need to be reset before processing additional
729      * samples.
730      * @throws LockedException if detector is busy processing a previous sample.
731      */
732     public boolean process(final double valueX,
733                            final double valueY,
734                            final double valueZ) throws LockedException {
735         if (mRunning) {
736             throw new LockedException();
737         }
738 
739         if (mStatus == Status.FAILED) {
740             return false;
741         }
742 
743         mRunning = true;
744 
745         if (mStatus == Status.IDLE) {
746             // start initialization
747             mStatus = Status.INITIALIZING;
748 
749             if (mListener != null) {
750                 //noinspection unchecked
751                 mListener.onInitializationStarted((D) this);
752             }
753         }
754 
755         mProcessedSamples++;
756 
757         final boolean windowedProcessed = mWindowedNoiseEstimator
758                 .addTriadAndProcess(valueX, valueY, valueZ);
759 
760         final double windowedStdNorm = mWindowedNoiseEstimator
761                 .getStandardDeviationNorm();
762 
763         final boolean filledWindow = mWindowedNoiseEstimator.isWindowFilled();
764 
765         if (mStatus == Status.INITIALIZING) {
766             // process sample during initialization
767             final boolean accumulatedProcessed = mAccumulatedNoiseEstimator
768                     .addTriad(valueX, valueY, valueZ);
769             final double accumulatedStdNorm = mAccumulatedNoiseEstimator
770                     .getStandardDeviationNorm();
771 
772             if (mProcessedSamples < mInitialStaticSamples) {
773                 if (windowedProcessed && accumulatedProcessed) {
774                     if (filledWindow && (windowedStdNorm / accumulatedStdNorm > mInstantaneousNoiseLevelFactor)) {
775                         // sudden motion detected
776                         mStatus = Status.FAILED;
777 
778                         // notify error
779                         if (mListener != null) {
780                             //noinspection unchecked
781                             mListener.onError((D) this, accumulatedStdNorm,
782                                     windowedStdNorm,
783                                     ErrorReason.SUDDEN_EXCESSIVE_MOVEMENT_DETECTED);
784                         }
785                     }
786                 }
787 
788             } else {
789                 // initialization completed
790                 // set base noise level and threshold
791                 mBaseNoiseLevel = accumulatedStdNorm;
792                 mThreshold = mBaseNoiseLevel * mThresholdFactor;
793 
794                 // keep average measurement triad in case we want to obtain
795                 // its value since initial period must be static
796                 mAvgX = mAccumulatedNoiseEstimator.getAvgX();
797                 mAvgY = mAccumulatedNoiseEstimator.getAvgY();
798                 mAvgZ = mAccumulatedNoiseEstimator.getAvgZ();
799 
800                 // reset accumulated estimator so that we can estimate
801                 // average specific force in static periods
802                 mAccumulatedNoiseEstimator.reset();
803 
804                 if (mBaseNoiseLevel > mBaseNoiseLevelAbsoluteThreshold) {
805                     // base noise level exceeds allowed value
806                     mStatus = Status.FAILED;
807 
808                     // notify error
809                     if (mListener != null) {
810                         //noinspection unchecked
811                         mListener.onError((D) this, accumulatedStdNorm,
812                                 windowedStdNorm,
813                                 ErrorReason.OVERALL_EXCESSIVE_MOVEMENT_DETECTED);
814                     }
815 
816                 } else {
817                     // initialization has been successfully completed
818                     mStatus = Status.INITIALIZATION_COMPLETED;
819 
820                     if (mListener != null) {
821                         //noinspection unchecked
822                         mListener.onInitializationCompleted(
823                                 (D) this, mBaseNoiseLevel);
824                     }
825                 }
826 
827             }
828 
829             mRunning = false;
830             return true;
831         } else {
832             // detect static or dynamic period
833             final Status previousStatus = mStatus;
834 
835             if (previousStatus == Status.STATIC_INTERVAL ||
836                     previousStatus == Status.INITIALIZATION_COMPLETED) {
837                 // while we are in static interval (or initial static period), keep
838                 // adding samples to estimate accumulated average measurement triad
839                 mAccumulatedNoiseEstimator.addTriad(valueX, valueY, valueZ);
840             }
841 
842             if (windowedStdNorm < mThreshold) {
843                 mStatus = Status.STATIC_INTERVAL;
844             } else {
845                 mStatus = Status.DYNAMIC_INTERVAL;
846             }
847 
848             if (previousStatus != mStatus) {
849                 // static/dynamic period change detected
850                 switch (mStatus) {
851                     case STATIC_INTERVAL:
852                         if (mListener != null) {
853                             //noinspection unchecked
854                             mListener.onStaticIntervalDetected((D) this);
855                         }
856                         break;
857                     case DYNAMIC_INTERVAL:
858                         // when switching from static to dynamic interval,
859                         // pick accumulated average measurement triad
860                         mAvgX = mAccumulatedNoiseEstimator.getAvgX();
861                         mAvgY = mAccumulatedNoiseEstimator.getAvgY();
862                         mAvgZ = mAccumulatedNoiseEstimator.getAvgZ();
863 
864                         // reset accumulated estimator when switching to dynamic period
865                         mAccumulatedNoiseEstimator.reset();
866 
867                         if (mListener != null) {
868                             //noinspection unchecked
869                             mListener.onDynamicIntervalDetected((D) this, mAvgX, mAvgY, mAvgZ);
870                         }
871                         break;
872                 }
873             }
874         }
875 
876         mRunning = false;
877         return true;
878     }

880     /**
881      * Resets this detector so that it is initialized again when new samples are added.
882      *
883      * @throws LockedException if detector is busy.
884      */
885     public void reset() throws LockedException {
886         if (mRunning) {
887             throw new LockedException();
888         }
889 
890         mRunning = true;
891 
892         mStatus = Status.IDLE;
893         mProcessedSamples = 0;
894         mBaseNoiseLevel = 0.0;
895         mThreshold = 0.0;
896 
897         mWindowedNoiseEstimator.reset();
898         mAccumulatedNoiseEstimator.reset();
899 
900         if (mListener != null) {
901             //noinspection unchecked
902             mListener.onReset((D) this);
903         }
904 
905         mRunning = false;
906     }

960     /**
961      * Possible detector status values.
962      */
963     public enum Status {
964         /**
965          * Detector is in idle status when it hasn't processed any sample yet.
966          */
967         IDLE,
968 
969         /**
970          * Detector is processing samples in the initial static process to determine base noise level.
971          */
972         INITIALIZING,
973 
974         /**
975          * Detector has sucessfully completed processing samples on the initial
976          * static period.
977          */
978         INITIALIZATION_COMPLETED,
979 
980         /**
981          * A static interval has been detected, where accelerometer is considered to be subject to no substantial
982          * movement forces.
983          */
984         STATIC_INTERVAL,
985 
986         /**
987          * A dynamic interval has been detected, where accelerometer is considered to be subject to substantial
988          * movement forces.
989          */
990         DYNAMIC_INTERVAL,
991 
992         /**
993          * Detector has failed. This happens if accelerometer is subject to sudden movement forces while detector
994          * is initializing during the initial static period.
995          * When detector has failed, no new samples will be allowed to be processed until detector is reset.
996          */
997         FAILED
