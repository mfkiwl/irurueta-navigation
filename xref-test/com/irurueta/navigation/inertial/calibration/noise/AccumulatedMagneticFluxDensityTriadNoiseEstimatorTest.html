396     @Test
397     public void testAddTriadAndReset1() throws LockedException, IOException {
398         final UniformRandomizer randomizer = new UniformRandomizer(new Random());
399         final WMMEarthMagneticFluxDensityEstimator wmmEstimator =
400                 new WMMEarthMagneticFluxDensityEstimator();
401 
402         final Matrix hardIron = Matrix.newFromArray(generateHardIron(randomizer));
403         final Matrix mm = generateSoftIronGeneral();
404         assertNotNull(mm);
405 
406         final GaussianRandomizer noiseRandomizer = new GaussianRandomizer(
407                 new Random(), 0.0, MAGNETOMETER_NOISE_STD);
408 
409         final NEDPosition position = createPosition(randomizer);
410         final CoordinateTransformation cnb = generateBodyC(randomizer);
411         final Date timestamp = new Date(createTimestamp(randomizer));
412 
413         final AccumulatedMagneticFluxDensityTriadNoiseEstimator estimator =
414                 new AccumulatedMagneticFluxDensityTriadNoiseEstimator(this);
415 
416         reset();
417         assertEquals(mStart, 0);
418         assertEquals(mTriadAdded, 0);
419         assertEquals(mReset, 0);
420         assertEquals(estimator.getNumberOfProcessedSamples(), 0);
421         assertNull(estimator.getLastTriad());
422         assertFalse(estimator.getLastTriad(null));
423         assertFalse(estimator.isRunning());
424 
425         MagneticFluxDensityTriad triad = new MagneticFluxDensityTriad();
426         final double timeInterval = estimator.getTimeInterval();
427         final MagneticFluxDensityTriad lastTriad = new MagneticFluxDensityTriad();
428         double valueX;
429         double valueY;
430         double valueZ;
431         double avgX = 0.0;
432         double avgY = 0.0;
433         double avgZ = 0.0;
434         double varX = 0.0;
435         double varY = 0.0;
436         double varZ = 0.0;
437         for (int i = 0, j = 1; i < N_SAMPLES; i++, j++) {
438             if (estimator.getLastTriad(lastTriad)) {
439                 assertEquals(estimator.getLastTriad(), lastTriad);
440                 assertEquals(lastTriad, triad);
441             }
442 
443             triad = generateTriad(hardIron.getBuffer(), mm, wmmEstimator,
444                     noiseRandomizer, timestamp, position, cnb);
445 
446             valueX = triad.getValueX();
447             valueY = triad.getValueY();
448             valueZ = triad.getValueZ();
449 
450             estimator.addTriad(valueX, valueY, valueZ);
451 
452             assertTrue(estimator.getLastTriad(lastTriad));
453             assertEquals(lastTriad, triad);
454             assertEquals(estimator.getNumberOfProcessedSamples(), i + 1);
455             assertFalse(estimator.isRunning());
456 
457             avgX = avgX * (double) i / (double) j + valueX / j;
458             avgY = avgY * (double) i / (double) j + valueY / j;
459             avgZ = avgZ * (double) i / (double) j + valueZ / j;
460 
461             final double diffX = valueX - avgX;
462             final double diffY = valueY - avgY;
463             final double diffZ = valueZ - avgZ;
464 
465             final double diffX2 = diffX * diffX;
466             final double diffY2 = diffY * diffY;
467             final double diffZ2 = diffZ * diffZ;
468 
469             varX = varX * (double) i / (double) j + diffX2 / j;
470             varY = varY * (double) i / (double) j + diffY2 / j;
471             varZ = varZ * (double) i / (double) j + diffZ2 / j;
472         }
473 
474         assertEquals(estimator.getNumberOfProcessedSamples(), N_SAMPLES);
475         assertFalse(estimator.isRunning());
476         assertEquals(mStart, 1);
477         assertEquals(mTriadAdded, N_SAMPLES);
