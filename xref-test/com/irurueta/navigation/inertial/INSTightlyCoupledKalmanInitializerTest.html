1   package com.irurueta.navigation.inertial;
2   
3   import com.irurueta.algebra.Matrix;
4   import com.irurueta.algebra.WrongSizeException;
5   import com.irurueta.statistics.UniformRandomizer;
6   import org.junit.Test;
7   
8   import java.util.Random;
9   
10  import static org.junit.Assert.assertEquals;
11  
12  public class INSTightlyCoupledKalmanInitializerTest {
13  
14      private static final double MIN_VALUE = 1e-4;
15      private static final double MAX_VALUE = 1e-3;
16  
17      private static final int NUM_PARAMS = INSTightlyCoupledKalmanInitializer.NUM_PARAMS;
18  
19      @Test
20      public void testInitialize() throws WrongSizeException {
21          final INSTightlyCoupledKalmanInitializerConfig config = generateConfig();
22  
23          final Matrix expected = Matrix.diagonal(new double[] {
24                  config.getInitialAttitudeUncertainty() * config.getInitialAttitudeUncertainty(),
25                  config.getInitialAttitudeUncertainty() * config.getInitialAttitudeUncertainty(),
26                  config.getInitialAttitudeUncertainty() * config.getInitialAttitudeUncertainty(),
27                  config.getInitialVelocityUncertainty() * config.getInitialVelocityUncertainty(),
28                  config.getInitialVelocityUncertainty() * config.getInitialVelocityUncertainty(),
29                  config.getInitialVelocityUncertainty() * config.getInitialVelocityUncertainty(),
30                  config.getInitialPositionUncertainty() * config.getInitialPositionUncertainty(),
31                  config.getInitialPositionUncertainty() * config.getInitialPositionUncertainty(),
32                  config.getInitialPositionUncertainty() * config.getInitialPositionUncertainty(),
33                  config.getInitialAccelerationBiasUncertainty() * config.getInitialAccelerationBiasUncertainty(),
34                  config.getInitialAccelerationBiasUncertainty() * config.getInitialAccelerationBiasUncertainty(),
35                  config.getInitialAccelerationBiasUncertainty() * config.getInitialAccelerationBiasUncertainty(),
36                  config.getInitialGyroscopeBiasUncertainty() * config.getInitialGyroscopeBiasUncertainty(),
37                  config.getInitialGyroscopeBiasUncertainty() * config.getInitialGyroscopeBiasUncertainty(),
38                  config.getInitialGyroscopeBiasUncertainty() * config.getInitialGyroscopeBiasUncertainty(),
39                  config.getInitialClockOffsetUncertainty() * config.getInitialClockOffsetUncertainty(),
40                  config.getInitialClockDriftUncertainty() * config.getInitialClockDriftUncertainty()
41          });
42  
43          final Matrix result1 = new Matrix(NUM_PARAMS, NUM_PARAMS);
44          INSTightlyCoupledKalmanInitializer.initialize(config, result1);
45          final Matrix result2 = new Matrix(1, 1);
46          INSTightlyCoupledKalmanInitializer.initialize(config, result2);
47          final Matrix result3 = INSTightlyCoupledKalmanInitializer.initialize(config);
48  
49          assertEquals(expected, result1);
50          assertEquals(expected, result2);
51          assertEquals(expected, result3);
52      }
53  
54      private static INSTightlyCoupledKalmanInitializerConfig generateConfig() {
55          final UniformRandomizer randomizer = new UniformRandomizer(new Random());
56          final double initialAttitudeUncertainty = randomizer.nextDouble(
57                  MIN_VALUE, MAX_VALUE);
58          final double initialVelocityUncertainty = randomizer.nextDouble(
59                  MIN_VALUE, MAX_VALUE);
60          final double initialPositionUncertainty = randomizer.nextDouble(
61                  MIN_VALUE, MAX_VALUE);
62          final double initialAccelerationBiasUncertainty = randomizer.nextDouble(
63                  MIN_VALUE, MAX_VALUE);
64          final double initialGyroscopeBiasUncertainty = randomizer.nextDouble(
65                  MIN_VALUE, MAX_VALUE);
66          final double initialClockOffsetUncertainty = randomizer.nextDouble(
67                  MIN_VALUE, MAX_VALUE);
68          final double initialClockDriftUncertainty = randomizer.nextDouble(
69                  MIN_VALUE, MAX_VALUE);
70          return new INSTightlyCoupledKalmanInitializerConfig(initialAttitudeUncertainty,
71                  initialVelocityUncertainty, initialPositionUncertainty,
72                  initialAccelerationBiasUncertainty, initialGyroscopeBiasUncertainty,
73                  initialClockOffsetUncertainty, initialClockDriftUncertainty);
74  
75      }
76  }
