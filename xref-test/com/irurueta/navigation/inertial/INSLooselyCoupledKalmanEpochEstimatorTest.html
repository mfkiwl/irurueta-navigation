698     private static INSLooselyCoupledKalmanState estimate(
699             final double x, final double y, final double z,
700             final double vx, final double vy, final double vz,
701             final INSLooselyCoupledKalmanState previousState,
702             final double fx, final double fy, final double fz,
703             final double previousLatitude,
704             final INSLooselyCoupledKalmanConfig config) throws AlgebraException {
705 
706         final Matrix omegaIe = Utils.skewMatrix(new double[]{0.0, 0.0, Constants.EARTH_ROTATION_RATE});
707 
708         // SYSTEM PROPAGATION PHASE
709 
710         // 1. Determine transition matrix using (14.50) (first-order approx)
711         final Matrix phiMatrix = Matrix.identity(15, 15);
712         phiMatrix.setSubmatrix(0, 0,
713                 2, 2,
714                 phiMatrix.getSubmatrix(0, 0,
715                         2, 2).subtractAndReturnNew(
716                                 omegaIe.multiplyByScalarAndReturnNew(
717                                         TIME_INTERVAL_SECONDS)));
718 
719         final Matrix estCbeOld = previousState.getBodyToEcefCoordinateTransformationMatrix();
720         phiMatrix.setSubmatrix(0, 12,
721                 2, 14,
722                 estCbeOld.multiplyByScalarAndReturnNew(
723                         TIME_INTERVAL_SECONDS));
724 
725         final Matrix measFibb = Matrix.newFromArray(new double[]{fx, fy, fz});
726         phiMatrix.setSubmatrix(3, 0,
727                 5, 2,
728                 Utils.skewMatrix(estCbeOld.multiplyAndReturnNew(measFibb))
729                         .multiplyByScalarAndReturnNew(
730                                 -TIME_INTERVAL_SECONDS));
731 
732         phiMatrix.setSubmatrix(3, 3,
733                 5, 5,
734                 phiMatrix.getSubmatrix(3, 3,
735                         5, 5).subtractAndReturnNew(
736                                 omegaIe.multiplyByScalarAndReturnNew(
737                                         2.0 * TIME_INTERVAL_SECONDS)));
738 
739         final double sinPrevLat = Math.sin(previousLatitude);
740         final double cosPrevLat = Math.cos(previousLatitude);
741         final double sinPrevLat2 = sinPrevLat * sinPrevLat;
742         final double cosPrevLat2 = cosPrevLat * cosPrevLat;
743 
744         final double geocentricRadius = Constants.EARTH_EQUATORIAL_RADIUS_WGS84
745                 / Math.sqrt(1.0 - Math.pow(Constants.EARTH_ECCENTRICITY * sinPrevLat, 2.0))
746                 * Math.sqrt(cosPrevLat2
747                         + Math.pow(1.0 - Constants.EARTH_ECCENTRICITY * Constants.EARTH_ECCENTRICITY, 2.0)
748                         * sinPrevLat2);
749 
750         final double prevX = previousState.getX();
751         final double prevY = previousState.getY();
752         final double prevZ = previousState.getZ();
753         final ECEFGravity gravity = ECEFGravityEstimator
754                 .estimateGravityAndReturnNew(prevX, prevY, prevZ);
755         final Matrix g = gravity.asMatrix();
756 
757         final Matrix estRebeOld = Matrix.newFromArray(new double[]{prevX, prevY, prevZ});
758 
759         final Matrix gScaled = g.multiplyByScalarAndReturnNew(
760                 -2.0 * TIME_INTERVAL_SECONDS / geocentricRadius);
761         final Matrix estRebeOldTrans = estRebeOld.transposeAndReturnNew();
762         final double previousPositionNorm = Math.sqrt(prevX * prevX +
763                 prevY * prevY + prevZ * prevZ);
764         final Matrix estRebeOldTransScaled = estRebeOldTrans.multiplyByScalarAndReturnNew(
765                 1.0 / previousPositionNorm);
766         phiMatrix.setSubmatrix(3, 6,
767                 5, 8,
768                 gScaled.multiplyAndReturnNew(estRebeOldTransScaled));
769 
770         phiMatrix.setSubmatrix(3, 9,
771                 5, 11,
772                 estCbeOld.multiplyByScalarAndReturnNew(
773                         TIME_INTERVAL_SECONDS));
774 
775         phiMatrix.setSubmatrix(6, 3,
776                 8, 5,
777                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
778                         TIME_INTERVAL_SECONDS));
779 
780         // 2. Determine approximate system noise covariance matrix using (14.82)
781         final Matrix qPrimeMatrix = new Matrix(15, 15);
782         qPrimeMatrix.setSubmatrix(0, 0,
783                 2, 2,
784                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
785                         config.getGyroNoisePSD()
786                                 * TIME_INTERVAL_SECONDS));
787         qPrimeMatrix.setSubmatrix(3, 3,
788                 5, 5,
789                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
790                         config.getAccelerometerNoisePSD()
791                                 * TIME_INTERVAL_SECONDS));
792         qPrimeMatrix.setSubmatrix(9, 9,
793                 11, 11,
794                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
795                         config.getAccelerometerBiasPSD()
796                                 * TIME_INTERVAL_SECONDS));
797         qPrimeMatrix.setSubmatrix(12, 12,
798                 14, 14,
799                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
800                         config.getGyroBiasPSD()
801                                 * TIME_INTERVAL_SECONDS));
802 
803         // 3. Propagate state estimates using (3.14) noting that all states are zero
804         final Matrix xEstPropagated = new Matrix(15, 1);
805 
806         // 4. Propagate state estimatoin error covariance matrix using (3.46)
807         final Matrix pMatrixOld = previousState.getCovariance();
808         final Matrix phiTrans = phiMatrix.transposeAndReturnNew();
809         final Matrix halfQ = qPrimeMatrix.multiplyByScalarAndReturnNew(0.5);
810         final Matrix tmp1 = pMatrixOld.addAndReturnNew(halfQ);
811         final Matrix pMatrixPropagated = phiMatrix.multiplyAndReturnNew(tmp1);
812         pMatrixPropagated.multiply(phiTrans);
813         pMatrixPropagated.add(halfQ);
814 
815         // MEASUREMENT UPDATE PHASE
816 
817         // 5. Set-up measurement matrix using (14.115)
818         final Matrix hMatrix = new Matrix(6, 15);
819         hMatrix.setSubmatrix(0, 6,
820                 2, 8,
821                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(-1.0));
822         hMatrix.setSubmatrix(3, 3,
823                 5, 5,
824                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(-1.0));
825 
826         // 6. Set-up measurement noise covariance matrix assuming all components of
827         // GNSS position and velocity are independent and have equal variance
828         final Matrix rMatrix = new Matrix(6, 6);
829         rMatrix.setSubmatrix(0, 0,
830                 2, 2,
831                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
832                         Math.pow(config.getPositionNoiseSD(), 2.0)));
833         rMatrix.setSubmatrix(0, 3,
834                 2, 5, new Matrix(3, 3));
835         rMatrix.setSubmatrix(3, 0,
836                 5, 2, new Matrix(3, 3));
837         rMatrix.setSubmatrix(3, 3,
838                 5, 5,
839                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
840                         Math.pow(config.getVelocityNoiseSD(), 2.0)));
841 
842         // 7. Calculate Kalman gain using (3.21)
843         final Matrix tmp2 = hMatrix.multiplyAndReturnNew(pMatrixPropagated)
844                 .multiplyAndReturnNew(hMatrix.transposeAndReturnNew());
845         tmp2.add(rMatrix);
846         final Matrix kMatrix = pMatrixPropagated
847                 .multiplyAndReturnNew(hMatrix.transposeAndReturnNew())
848                 .multiplyAndReturnNew(Utils.inverse(tmp2));
849 
850         // 8. Formulate measurement innovtions using (14.102), noting that zero
851         // lever arm is assumed here
852         final Matrix deltaZ = new Matrix(6, 1);
853 
854         final double prevVx = previousState.getVx();
855         final double prevVy = previousState.getVy();
856         final double prevVz = previousState.getVz();
857         final Matrix estVebeOld = Matrix.newFromArray(
858                 new double[]{prevVx, prevVy, prevVz});
859 
860         final Matrix gnssRebe = Matrix.newFromArray(new double[]{x, y, z});
861         final Matrix gnssVebe = Matrix.newFromArray(new double[]{vx, vy, vz});
862 
863         deltaZ.setSubmatrix(0, 0,
864                 2, 0,
865                 gnssRebe.subtractAndReturnNew(estRebeOld));
866         deltaZ.setSubmatrix(3, 0,
867                 5, 0,
868                 gnssVebe.subtractAndReturnNew(estVebeOld));
869 
870         // 9. Update state estimates using (3.24)
871         final Matrix xEstNew = xEstPropagated.addAndReturnNew(
872                 kMatrix.multiplyAndReturnNew(deltaZ));
873 
874         // 10. Update state estimation error covariance matrix using (3.25)
875         final Matrix pMatrixNew = (Matrix.identity(15, 15)
876                 .subtractAndReturnNew(kMatrix.multiplyAndReturnNew(hMatrix)))
877                 .multiplyAndReturnNew(pMatrixPropagated);
878 
879         // CLOSED-LOOP CORRECTION
880 
881         // Correct attitude, velocity, and position using (14.7-9)
882         final Matrix estCbeNew = (Matrix.identity(3, 3)
883                 .subtractAndReturnNew(
884                         Utils.skewMatrix(
885                                 xEstNew.getSubmatrix(0, 0,
886                                         2, 0))))
887                 .multiplyAndReturnNew(estCbeOld);
888         final Matrix estVebeNew = estVebeOld.subtractAndReturnNew(
889                 xEstNew.getSubmatrix(3, 0,
890                         5, 0));
891         final Matrix estRebeNew = estRebeOld.subtractAndReturnNew(
892                 xEstNew.getSubmatrix(6, 0,
893                         8, 0));
894 
895         // Update IMU bias estimates
896         final Matrix estIMUbiasOld = Matrix.newFromArray(new double[]{
897                 previousState.getAccelerationBiasX(),
898                 previousState.getAccelerationBiasY(),
899                 previousState.getAccelerationBiasZ(),
900                 previousState.getGyroBiasX(),
901                 previousState.getGyroBiasY(),
902                 previousState.getGyroBiasZ()
903         });
904         final Matrix estIMUbiasNew = estIMUbiasOld.addAndReturnNew(
905                 xEstNew.getSubmatrix(9, 0,
906                         14, 0));
907 
908         return new INSLooselyCoupledKalmanState(estCbeNew,
909                 estVebeNew.getElementAtIndex(0),
910                 estVebeNew.getElementAtIndex(1),
911                 estVebeNew.getElementAtIndex(2),
912                 estRebeNew.getElementAtIndex(0),
