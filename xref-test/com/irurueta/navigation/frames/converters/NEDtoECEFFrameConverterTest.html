44      @Test
45      public void testConvertAndReturnNew() throws InvalidRotationMatrixException,
46              InvalidSourceAndDestinationFrameTypeException, RotationException {
47  
48          int numValid = 0;
49          for (int t = 0; t < TIMES; t++) {
50              final UniformRandomizer randomizer = new UniformRandomizer(new Random());
51  
52              final double latitude = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
53              final double longitude = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
54              final double height = randomizer.nextDouble(MIN_HEIGHT, MAX_HEIGHT);
55  
56              final double vn = randomizer.nextDouble(MIN_VELOCITY_VALUE, MAX_VELOCITY_VALUE);
57              final double ve = randomizer.nextDouble(MIN_VELOCITY_VALUE, MAX_VELOCITY_VALUE);
58              final double vd = randomizer.nextDouble(MIN_VELOCITY_VALUE, MAX_VELOCITY_VALUE);
59  
60              final double roll = Math.toRadians(
61                      randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
62              final double pitch = Math.toRadians(
63                      randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
64              final double yaw = Math.toRadians(
65                      randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
66              final Quaternion q = new Quaternion(roll, pitch, yaw);
67  
68              final Matrix m = q.asInhomogeneousMatrix();
69              final CoordinateTransformationMatrix c = new CoordinateTransformationMatrix(
70                      m, FrameType.BODY_FRAME,
71                      FrameType.LOCAL_NAVIGATION_FRAME);
72  
73              final NEDFrame nedFrame1 = new NEDFrame(latitude, longitude, height, vn, ve, vd, c);
74  
75              final NEDtoECEFFrameConverter converter1 = new NEDtoECEFFrameConverter();
76  
77              final ECEFFrame ecefFrame = converter1.convertAndReturnNew(nedFrame1);
78  
79              // convert back to NED
80              final ECEFtoNEDFrameConverter converter2 = new ECEFtoNEDFrameConverter();
81  
82              final NEDFrame nedFrame2 = converter2.convertAndReturnNew(ecefFrame);
83  
84              // check
85              if (Math.abs(nedFrame1.getLatitude() - nedFrame2.getLatitude()) > ABSOLUTE_ERROR) {
86                  continue;
87              }
88  
89              assertEquals(nedFrame1.getLatitude(), nedFrame2.getLatitude(), ABSOLUTE_ERROR);
90              assertEquals(nedFrame1.getLongitude(), nedFrame2.getLongitude(), ABSOLUTE_ERROR);
91  
92              if (Math.abs(nedFrame1.getVn() - nedFrame2.getVn()) > ABSOLUTE_ERROR) {
93                  continue;
94              }
95  
96              assertEquals(nedFrame1.getVn(), nedFrame2.getVn(), ABSOLUTE_ERROR);
97              assertEquals(nedFrame1.getVe(), nedFrame2.getVe(), ABSOLUTE_ERROR);
98  
99              if (Math.abs(nedFrame1.getVd() - nedFrame2.getVd()) > ABSOLUTE_ERROR) {
100                 continue;
101             }
102 
103             assertEquals(nedFrame1.getVd(), nedFrame2.getVd(), ABSOLUTE_ERROR);
104 
105             assertEquals(nedFrame1.getCoordinateTransformationMatrix().getSourceType(),
106                     nedFrame2.getCoordinateTransformationMatrix().getSourceType());
107             assertEquals(nedFrame1.getCoordinateTransformationMatrix().getDestinationType(),
108                     nedFrame2.getCoordinateTransformationMatrix().getDestinationType());
109 
110             final Quaternion q1 = new Quaternion();
111             nedFrame1.getCoordinateTransformationMatrix().asRotation(q1);
112 
113             final Quaternion q2 = new Quaternion();
114             nedFrame2.getCoordinateTransformationMatrix().asRotation(q2);
115             assertTrue(q1.equals(q2, ABSOLUTE_ERROR));
116 
117             numValid++;
118             break;
119         }
120 
121         assertTrue(numValid > 0);
122     }
