174     private SatellitePositionAndVelocity computeSatellitePositionAndVelocity(final double time,
175                                                                              final GNSSConfig config, final int j) {
176         final double inclination = Math.toRadians(config.getSatellitesInclinationDegrees());
177 
178         final double omegaIs = Math.sqrt(Constants.EARTH_GRAVITATIONAL_CONSTANT /
179                 Math.pow(config.getOrbitalRadiusOfSatellites(), 3.0));
180 
181         final double constTime = time + config.getConstellationTimingOffset();
182 
183         final double uoso = 2.0 * Math.PI * j / config.getNumberOfSatellites() + omegaIs * constTime;
184 
185         final double[] roso = new double[]{ config.getOrbitalRadiusOfSatellites() * Math.cos(uoso),
186                 config.getOrbitalRadiusOfSatellites() * Math.sin(uoso) };
187 
188         final double omega = Math.PI * ((j + 1) % 6) / 3
189                 + Math.toRadians(config.getConstellationLongitudeOffsetDegrees())
190                 - Constants.EARTH_ROTATION_RATE * constTime;
191 
192         final ECEFPosition position = new ECEFPosition(
193                 roso[0] * Math.cos(omega) - roso[1] * Math.cos(inclination) * Math.sin(omega),
194                 roso[0] * Math.sin(omega) + roso[1] * Math.cos(inclination) * Math.cos(omega),
195                 roso[1] * Math.sin(inclination));
196 
197         final double[] voso = new double[] { - config.getOrbitalRadiusOfSatellites() * omegaIs * Math.sin(uoso),
198                 config.getOrbitalRadiusOfSatellites() * omegaIs * Math.cos(uoso), 0.0 };
199 
200         final ECEFVelocity velocity = new ECEFVelocity(
201                 voso[0] * Math.cos(omega) - voso[1] * Math.cos(inclination) * Math.sin(omega)
202                         + Constants.EARTH_ROTATION_RATE * position.getY(),
203                 voso[0] * Math.sin(omega) + voso[1] * Math.cos(inclination) * Math.cos(omega)
204                         - Constants.EARTH_ROTATION_RATE * position.getX(),
205                 voso[1] * Math.sin(inclination));
206 
207         return new SatellitePositionAndVelocity(position, velocity);
208     }
