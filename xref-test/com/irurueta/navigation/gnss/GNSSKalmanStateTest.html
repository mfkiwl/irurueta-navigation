34      @Test
35      public void testConstructor() throws WrongSizeException {
36          // test empty constructor
37          GNSSKalmanState state = new GNSSKalmanState();
38  
39          // check default values
40          assertNull(state.getEstimation());
41          assertNull(state.getCovariance());
42  
43  
44          // test constructor with estimation and covariance
45          final UniformRandomizer randomizer = new UniformRandomizer(new Random());
46          final double x = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
47          final double y = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
48          final double z = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
49          final double vx = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
50          final double vy = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
51          final double vz = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
52          final double clockOffset = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
53          final double clockDrift = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
54          final GNSSEstimation estimation = new GNSSEstimation(x, y, z, vx, vy, vz,
55                  clockOffset, clockDrift);
56          final Matrix covariance = Matrix.identity(GNSSEstimation.NUM_PARAMETERS,
57                  GNSSEstimation.NUM_PARAMETERS);
58          state = new GNSSKalmanState(estimation, covariance);
59  
60          // check default values
61          assertSame(state.getEstimation(), estimation);
62          assertSame(state.getCovariance(), covariance);
63  
64          // Force IllegalArgumentException
65          state = null;
66          try {
67              state = new GNSSKalmanState(estimation, new Matrix(1, 1));
68              fail("IllegalArgumentException expected but not thrown");
69          } catch (final IllegalArgumentException ignore) { }
70          assertNull(state);
71  
72  
73          // test copy constructor
74          state = new GNSSKalmanState(estimation, covariance);
75          final GNSSKalmanState state2 = new GNSSKalmanState(state);
76  
77          // check default values
78          assertEquals(state2.getEstimation(), estimation);
79          assertEquals(state2.getCovariance(), covariance);
80      }
