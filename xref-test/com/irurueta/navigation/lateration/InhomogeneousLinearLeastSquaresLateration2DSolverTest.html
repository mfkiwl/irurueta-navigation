335     @Test
336     public void testSolveNoError() throws LaterationException, NotReadyException, LockedException {
337         UniformRandomizer randomizer = new UniformRandomizer(new Random());
338 
339         int numValid = 0;
340         for (int t = 0; t < TIMES; t++) {
341             int numCircles = randomizer.nextInt(MIN_CIRCLES, MAX_CIRCLES);
342 
343             InhomogeneousPoint2D position = new InhomogeneousPoint2D(
344                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
345                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
346             InhomogeneousPoint2D center;
347             double radius;
348             Circle[] circles = new Circle[numCircles];
349             for (int i = 0; i < numCircles; i++) {
350                 center = new InhomogeneousPoint2D(
351                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
352                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
353                 radius = center.distanceTo(position);
354                 circles[i] = new Circle(center, radius);
355             }
356 
357             InhomogeneousLinearLeastSquaresLateration2DSolver solver =
358                     new InhomogeneousLinearLeastSquaresLateration2DSolver(
359                     circles, this);
360 
361             reset();
362             assertEquals(solveStart, 0);
363             assertEquals(solveEnd, 0);
364             assertTrue(solver.isReady());
365             assertNull(solver.getEstimatedPosition());
366             assertNull(solver.getEstimatedPositionCoordinates());
367 
368             solver.solve();
369 
370             Point2D estimatedPosition = solver.getEstimatedPosition();
371             if (estimatedPosition.distanceTo(position) > ABSOLUTE_ERROR) {
372                 continue;
373             }
374             assertTrue(position.equals(estimatedPosition, ABSOLUTE_ERROR));
375 
376             numValid++;
377             break;
378         }
379 
380         assertTrue(numValid > 0);
381 
382         // Force NotReadyException
383         InhomogeneousLinearLeastSquaresLateration2DSolver solver =
384                 new InhomogeneousLinearLeastSquaresLateration2DSolver();
385         try {
386             solver.solve();
387             fail("NotReadyException expected but not thrown");
388         } catch (NotReadyException ignore) { }
389 
390         // Force LaterationException
391         Circle[] circles = new Circle[3];
392         InhomogeneousPoint2D center;
393         double radius;
394         for (int i = 0; i < 3; i++) {
395             center = new InhomogeneousPoint2D(0.0, 0.0);
396             radius = LaterationSolver.EPSILON;
397             circles[i] = new Circle(center, radius);
398         }
399         solver.setCircles(circles);
400         try {
401             solver.solve();
402             fail("LaterationException expected but not thrown");
403         } catch (LaterationException ignore) { }
404     }

406     @Test
407     public void testSolveWithError() throws LaterationException, NotReadyException, LockedException {
408         UniformRandomizer randomizer = new UniformRandomizer(new Random());
409 
410         int numValid = 0, numInvalid = 0;
411         for (int t = 0; t < TIMES; t++) {
412             int numPoints = randomizer.nextInt(MIN_CIRCLES, MAX_CIRCLES);
413 
414             InhomogeneousPoint2D position = new InhomogeneousPoint2D(
415                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
416                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
417             InhomogeneousPoint2D point;
418             double distance, error;
419             Point2D[] positions = new Point2D[numPoints];
420             double[] distances = new double[numPoints];
421             for (int i = 0; i < numPoints; i++) {
422                 point = new InhomogeneousPoint2D(
423                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
424                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
425                 distance = point.distanceTo(position);
426                 error = randomizer.nextDouble(MIN_DISTANCE_ERROR, MAX_DISTANCE_ERROR);
427                 positions[i] = point;
428                 distances[i] = distance + error;
429             }
430 
431             InhomogeneousLinearLeastSquaresLateration2DSolver solver =
432                     new InhomogeneousLinearLeastSquaresLateration2DSolver(
433                     positions, distances, this);
434 
435             reset();
436             assertEquals(solveStart, 0);
437             assertEquals(solveEnd, 0);
438             assertTrue(solver.isReady());
439             assertNull(solver.getEstimatedPosition());
440             assertNull(solver.getEstimatedPositionCoordinates());
441 
442             solver.solve();
443 
444             Point2D estimatedPosition = solver.getEstimatedPosition();
445             distance = estimatedPosition.distanceTo(position);
446             if (distance >= LARGE_ABSOLUTE_ERROR) {
447                 numInvalid++;
448             } else {
449                 numValid++;
450                 assertTrue(position.equals(estimatedPosition, LARGE_ABSOLUTE_ERROR));
451             }
452         }
453 
454         assertTrue(numValid > numInvalid);
455     }
