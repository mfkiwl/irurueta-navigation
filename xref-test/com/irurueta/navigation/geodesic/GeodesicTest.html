1044     private static PolygonResult planimeter(final double[][] points) {
1045         POLYGON.clear();
1046         for (double[] point : points) {
1047             POLYGON.addPoint(point[0], point[1]);
1048         }
1049         return POLYGON.compute(false, true);
1050     }

1052     private static PolygonResult polyLength(final double[][] points) {
1053         POLYLINE.clear();
1054         for (double[] point : points) {
1055             POLYLINE.addPoint(point[0], point[1]);
1056         }
1057         return POLYLINE.compute(false, true);
