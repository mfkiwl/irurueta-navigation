16  package com.irurueta.navigation.geodesic.wmm;
17  
18  import com.irurueta.navigation.inertial.NEDMagneticFluxDensity;
19  import com.irurueta.statistics.UniformRandomizer;
20  import com.irurueta.units.Angle;
21  import com.irurueta.units.AngleUnit;
22  import org.junit.Test;
23  
24  import java.util.Random;
25  
26  import static org.junit.Assert.assertEquals;
27  
28  public class EarthMagneticFluxDensityEstimatorTest {
29  
30      private static final double MIN_MAGNITUDE = 30e-6;
31      private static final double MAX_MAGNITUDE = 60e-6;
32  
33      private static final double MIN_DIP_DEGREES = -90.0;
34      private static final double MAX_DIP_DEGREES = 90.0;
35  
36      private static final double MIN_DECLINATION = -180.0;
37      private static final double MAX_DECLINATION = 180.0;
38  
39      private static final double ABSOLUTE_ERROR = 1e-6;
40  
41      @Test
42      public void testEstimate() {
43          final UniformRandomizer randomizer = new UniformRandomizer(
44                  new Random());
45          final double magnitude = randomizer.nextDouble(MIN_MAGNITUDE,
46                  MAX_MAGNITUDE);
47          final double dip = Math.toRadians(
48                  randomizer.nextDouble(MIN_DIP_DEGREES, MAX_DIP_DEGREES));
49          final double declination = Math.toRadians(
50                  randomizer.nextDouble(MIN_DECLINATION, MAX_DECLINATION));
51  
52          final NEDMagneticFluxDensity b1 = new NEDMagneticFluxDensity();
53          EarthMagneticFluxDensityEstimator.estimate(magnitude, declination,
54                  dip, b1);
55          final NEDMagneticFluxDensity b2 = EarthMagneticFluxDensityEstimator
56                  .estimate(magnitude, declination, dip);
57  
58          assertEquals(b1.getBn(),
59                  Math.cos(declination) * Math.cos(dip) * magnitude,
60                  0.0);
61          assertEquals(b1.getBe(),
62                  Math.sin(declination) * Math.cos(dip) * magnitude,
63                  0.0);
64          assertEquals(b1.getBd(), Math.sin(dip) * magnitude,
65                  0.0);
66          assertEquals(b1, b2);
67  
68          final Angle dipAngle = new Angle(dip, AngleUnit.RADIANS);
69          final Angle declinationAngle = new Angle(declination,
70                  AngleUnit.RADIANS);
71  
72          final NEDMagneticFluxDensity b3 = new NEDMagneticFluxDensity();
73          EarthMagneticFluxDensityEstimator.estimate(magnitude, declinationAngle,
74                  dipAngle, b3);
75          final NEDMagneticFluxDensity b4 = EarthMagneticFluxDensityEstimator
76                  .estimate(magnitude, declinationAngle, dipAngle);
77  
78          assertEquals(b1, b3);
79          assertEquals(b1, b4);
80      }
81  
82      @Test
83      public void testDeclination() {
84          final UniformRandomizer randomizer = new UniformRandomizer(
85                  new Random());
86          final double magnitude = randomizer.nextDouble(MIN_MAGNITUDE,
87                  MAX_MAGNITUDE);
88          final double dip = Math.toRadians(
89                  randomizer.nextDouble(MIN_DIP_DEGREES, MAX_DIP_DEGREES));
90          final double declination1 = Math.toRadians(
91                  randomizer.nextDouble(MIN_DECLINATION, MAX_DECLINATION));
92  
93          final NEDMagneticFluxDensity b = EarthMagneticFluxDensityEstimator
94                  .estimate(magnitude, declination1, dip);
95  
96          // get declination
97          final double declination2 = EarthMagneticFluxDensityEstimator
98                  .getDeclination(b);
99          final Angle declination3 = EarthMagneticFluxDensityEstimator
100                 .getDeclinationAsAngle(b);
101         final Angle declination4 = new Angle(0.0, AngleUnit.DEGREES);
102         EarthMagneticFluxDensityEstimator.getDeclinationAsAngle(b,
103                 declination4);
104 
105         assertEquals(declination1, declination2, ABSOLUTE_ERROR);
106         assertEquals(AngleUnit.RADIANS, declination3.getUnit());
107         assertEquals(declination2, declination3.getValue().doubleValue(),
108                 0.0);
109         assertEquals(declination3, declination4);
110     }
111 
112     @Test
113     public void testDip() {
114         final UniformRandomizer randomizer = new UniformRandomizer(
115                 new Random());
116         final double magnitude = randomizer.nextDouble(MIN_MAGNITUDE,
117                 MAX_MAGNITUDE);
118         final double dip1 = Math.toRadians(
119                 randomizer.nextDouble(MIN_DIP_DEGREES, MAX_DIP_DEGREES));
120         final double declination = Math.toRadians(
121                 randomizer.nextDouble(MIN_DECLINATION, MAX_DECLINATION));
122 
123         final NEDMagneticFluxDensity b = EarthMagneticFluxDensityEstimator
124                 .estimate(magnitude, declination, dip1);
125 
126         // get dip
127         final double dip2 = EarthMagneticFluxDensityEstimator.getDip(b);
128         final Angle dip3 = EarthMagneticFluxDensityEstimator
129                 .getDipAsAngle(b);
130         final Angle dip4 = new Angle(0.0, AngleUnit.DEGREES);
131         EarthMagneticFluxDensityEstimator.getDipAsAngle(b, dip4);
132 
133         assertEquals(dip1, dip2, ABSOLUTE_ERROR);
134         assertEquals(AngleUnit.RADIANS, dip3.getUnit());
135         assertEquals(dip2, dip3.getValue().doubleValue(), 0.0);
136         assertEquals(dip3, dip4);
137     }
