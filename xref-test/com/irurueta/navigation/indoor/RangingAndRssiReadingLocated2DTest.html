552     @Test
553     public void testHasSameAccessPoint() {
554         WifiAccessPoint ap1 = new WifiAccessPoint("bssid1", FREQUENCY);
555         WifiAccessPoint ap2 = new WifiAccessPoint("bssid2", FREQUENCY);
556 
557         InhomogeneousPoint2D position = new InhomogeneousPoint2D();
558 
559         RangingAndRssiReadingLocated2D<WifiAccessPoint> reading1 = new RangingAndRssiReadingLocated2D<>(
560                 ap1, 1.5, -50.0, position);
561         RangingAndRssiReadingLocated2D<WifiAccessPoint> reading2 = new RangingAndRssiReadingLocated2D<>(
562                 ap1, 1.5, -50.0, position);
563         RangingAndRssiReadingLocated2D<WifiAccessPoint> reading3 = new RangingAndRssiReadingLocated2D<>(
564                 ap2, 1.5, -50.0, position);
565 
566         //check
567         assertTrue(reading1.hasSameSource(reading1));
568         assertTrue(reading1.hasSameSource(reading2));
569         assertFalse(reading1.hasSameSource(reading3));
570     }
