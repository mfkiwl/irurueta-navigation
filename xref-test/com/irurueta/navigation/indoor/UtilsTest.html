94      @Test
95      public void testPropagatePowerVarianceToDistanceVariance()
96              throws EvaluationException {
97          UniformRandomizer randomizer = new UniformRandomizer(new Random());
98  
99          double minDistanceVariance = Double.MAX_VALUE;
100         double maxDistanceVariance = -Double.MAX_VALUE;
101         double avgDistanceVariance = 0.0;
102         for (int t = 0; t < TIMES; t++) {
103             final double txPowerdBm = randomizer.nextDouble(MIN_RSSI, MAX_RSSI);
104             double txPower = Utils.dBmToPower(txPowerdBm);
105 
106             final double pathLossExponent = randomizer.nextDouble(
107                     MIN_PATH_LOSS_EXPONENT, MAX_PATH_LOSS_EXPONENT);
108             double distance = randomizer.nextDouble(MIN_DISTANCE, MAX_DISTANCE);
109 
110             double rxPowerdBm = Utils.powerTodBm(
111                     receivedPower(txPower, distance,
112                             pathLossExponent));
113 
114             double distanceVariance = Utils.propagatePowerVarianceToDistanceVariance(
115                     txPowerdBm, rxPowerdBm, pathLossExponent, FREQUENCY,
116                     RX_POWER_VARIANCE);
117             assertTrue(distanceVariance > 0.0);
118 
119             double k = SPEED_OF_LIGHT / (4.0 * Math.PI * FREQUENCY);
120             double kdB = 10.0 * Math.log10(k);
121             double derivative = -Math.log(10.0) / (10.0 * pathLossExponent) *
122                     Math.pow(10.0, (pathLossExponent * kdB + txPowerdBm - rxPowerdBm) /
123                     (10.0 * pathLossExponent));
124 
125             DerivativeEstimator derivativeEstimator = new DerivativeEstimator(new SingleDimensionFunctionEvaluatorListener() {
126                 @Override
127                 public double evaluate(double point) {
128                     double k = RssiRadioSourceEstimator.SPEED_OF_LIGHT / (4.0 * Math.PI * FREQUENCY);
129                     double kdB = 10.0 * Math.log10(k);
130                     double logSqrDistance = (pathLossExponent * kdB + txPowerdBm - point) / (5.0 * pathLossExponent);
131                     return Math.pow(10.0,logSqrDistance / 2.0);
132                 }
133             });
134 
135             double derivative2 = derivativeEstimator.derivative(rxPowerdBm);
136             assertEquals(derivative, derivative2, LARGE_ABSOLUTE_ERROR);
137 
138             assertEquals(distanceVariance,
139                     derivative * derivative * RX_POWER_VARIANCE, 0.0);
140 
141             if (distanceVariance < minDistanceVariance) {
142                 minDistanceVariance = distanceVariance;
143             }
144             if (distanceVariance > maxDistanceVariance) {
145                 maxDistanceVariance = distanceVariance;
146             }
147             avgDistanceVariance += distanceVariance / TIMES;
148 
149             //check that if rx power variance is zero, then distance variance is
150             // zero as well
151             distanceVariance = Utils.propagatePowerVarianceToDistanceVariance(
152                     txPowerdBm, rxPowerdBm, pathLossExponent, FREQUENCY,
153                     0.0);
154             assertEquals(distanceVariance, 0.0, 0.0);
155 
156             //test without rx power
157             assertEquals(Utils.propagatePowerVarianceToDistanceVariance(
158                     txPowerdBm, rxPowerdBm, pathLossExponent, FREQUENCY,
159                     null), 0.0, 0.0);
160         }
161 
162         LOGGER.log(Level.INFO, "Min dist variance: {0}", minDistanceVariance);
163         LOGGER.log(Level.INFO, "Max dist variance: {0}", maxDistanceVariance);
164         LOGGER.log(Level.INFO, "Avg dist variance: {0}", avgDistanceVariance);
165     }

167     @Test
168     public void testPropagateVariancesToDistanceVariance()
169             throws IndoorException, EvaluationException {
170         UniformRandomizer randomizer = new UniformRandomizer(new Random());
171 
172         double minDistanceVariance = Double.MAX_VALUE;
173         double maxDistanceVariance = -Double.MAX_VALUE;
174         double avgDistanceVariance = 0.0;
175         for (int t = 0; t < TIMES; t++) {
176             double txPowerdBm = randomizer.nextDouble(MIN_RSSI, MAX_RSSI);
177             double txPower = Utils.dBmToPower(txPowerdBm);
178 
179             double pathLossExponent = randomizer.nextDouble(
180                     MIN_PATH_LOSS_EXPONENT, MAX_PATH_LOSS_EXPONENT);
181             double distance = randomizer.nextDouble(MIN_DISTANCE, MAX_DISTANCE);
182 
183             double rxPowerdBm = Utils.powerTodBm(
184                     receivedPower(txPower, distance,
185                             pathLossExponent));
186 
187             MultivariateNormalDist dist = Utils.propagateVariancesToDistanceVariance(
188                     txPowerdBm, rxPowerdBm, pathLossExponent, FREQUENCY,
189                     TX_POWER_VARIANCE, RX_POWER_VARIANCE, PATHLOSS_EXPONENT_VARIANCE);
190 
191             double k = RssiRadioSourceEstimator.SPEED_OF_LIGHT / (4.0 * Math.PI * FREQUENCY);
192             final double kdB = 10.0 * Math.log10(k);
193 
194             double tenPathLossExponent = 10.0 * pathLossExponent;
195             double tmp = (pathLossExponent * kdB + txPowerdBm - rxPowerdBm) / tenPathLossExponent;
196             double derivativeTxPower = Math.log(10.0) / tenPathLossExponent * Math.pow(10.0,
197                     tmp);
198             double derivativeRxPower = -Math.log(10.0) / tenPathLossExponent * Math.pow(10.0,
199                     tmp);
200 
201             double g = (pathLossExponent * kdB + txPowerdBm - rxPowerdBm) / (10.0 * pathLossExponent);
202             double derivativeG = (kdB * 10.0 * pathLossExponent -
203                     10.0 * (pathLossExponent * kdB + txPowerdBm - rxPowerdBm)) /
204                     Math.pow(10.0 * pathLossExponent, 2.0);
205 
206             double derivativePathLossExponent = Math.log(10.0) * derivativeG * Math.pow(10.0, g);
207 
208             double[] gradient = new double[]{
209                     derivativeTxPower,
210                     derivativeRxPower,
211                     derivativePathLossExponent
212             };
213 
214             GradientEstimator gradientEstimator = new GradientEstimator(new MultiDimensionFunctionEvaluatorListener() {
215                 @Override
216                 public double evaluate(double[] point) {
217                     double txPower = point[0];
218                     double rxPower = point[1];
219                     double pathLossExponent = point[2];
220 
221                     double logSqrDistance = (pathLossExponent * kdB + txPower - rxPower) / (5.0 * pathLossExponent);
222                     return Math.pow(10.0,logSqrDistance / 2.0);
223                 }
224             });
225 
226             double[] gradient2 = gradientEstimator.gradient(
227                     new double[]{
228                             txPowerdBm,
229                             rxPowerdBm,
230                             pathLossExponent});
231 
232             assertArrayEquals(gradient, gradient2, LARGE_ABSOLUTE_ERROR);
233 
234             assertEquals(dist.getMean()[0], distance, ABSOLUTE_ERROR);
235 
236             double distanceVariance = dist.getCovariance().
237                     getElementAt(0, 0);
238             assertTrue(distanceVariance > 0.0);
239 
240             if (distanceVariance < minDistanceVariance) {
241                 minDistanceVariance = distanceVariance;
242             }
243             if (distanceVariance > maxDistanceVariance) {
244                 maxDistanceVariance = distanceVariance;
245             }
246             avgDistanceVariance += distanceVariance / TIMES;
247 
248             //check without variances
249             assertNull(Utils.propagateVariancesToDistanceVariance(
250                     txPowerdBm, rxPowerdBm, pathLossExponent, FREQUENCY,
251                     null, null, null));
252         }
253 
254         LOGGER.log(Level.INFO, "Min dist variance: {0}", minDistanceVariance);
255         LOGGER.log(Level.INFO, "Max dist variance: {0}", maxDistanceVariance);
256         LOGGER.log(Level.INFO, "Avg dist variance: {0}", avgDistanceVariance);
257     }

421     @Test
422     public void testPropagateVariancesToRssiVarianceFirstOrderNonLinear2D()
423             throws IndoorException, AlgebraException {
424         UniformRandomizer randomizer = new UniformRandomizer(new Random());
425 
426         for (int t = 0; t < TIMES; t++) {
427             double fingerprintRssi = randomizer.nextDouble(MIN_RSSI, MAX_RSSI);
428             double pathLossExponent = randomizer.nextDouble(
429                     MIN_PATH_LOSS_EXPONENT, MAX_PATH_LOSS_EXPONENT);
430 
431             double x1 = randomizer.nextDouble(MIN_POS, MAX_POS);
432             double y1 = randomizer.nextDouble(MIN_POS, MAX_POS);
433             Point2D fingerprintPosition = new InhomogeneousPoint2D(x1, y1);
434 
435             double xa = randomizer.nextDouble(MIN_POS, MAX_POS);
436             double ya = randomizer.nextDouble(MIN_POS, MAX_POS);
437             Point2D radioSourcePosition = new InhomogeneousPoint2D(xa, ya);
438 
439             double xi = randomizer.nextDouble(MIN_POS, MAX_POS);
440             double yi = randomizer.nextDouble(MIN_POS, MAX_POS);
441             Point2D estimatedPosition = new InhomogeneousPoint2D(xi, yi);
442 
443             //test without variance values
444             MultivariateNormalDist dist = Utils.propagateVariancesToRssiVarianceFirstOrderNonLinear2D(
445                     fingerprintRssi, pathLossExponent, fingerprintPosition,
446                     radioSourcePosition, estimatedPosition, null,
447                     null, null,
448                     null, null);
449 
450             double diffX1a = x1 - xa;
451             double diffY1a = y1 - ya;
452 
453             double diffXi1 = xi - x1;
454             double diffYi1 = yi - y1;
455 
456             double diffX1a2 = diffX1a * diffX1a;
457             double diffY1a2 = diffY1a * diffY1a;
458 
459             double d1a2 = diffX1a2 + diffY1a2;
460 
461             double rssi = fingerprintRssi - 10.0 * pathLossExponent *
462                     (diffX1a * diffXi1 + diffY1a * diffYi1)/(Math.log(10.0) * d1a2);
463 
464             assertEquals(dist.getMean()[0], rssi, ABSOLUTE_ERROR);
465 
466             double rssiVariance = dist.getCovariance().
467                     getElementAt(0, 0);
468             assertEquals(rssiVariance, 0.0, ABSOLUTE_ERROR);
469 
470 
471             //test with variance values
472             dist = Utils.propagateVariancesToRssiVarianceFirstOrderNonLinear2D(
473                     fingerprintRssi, pathLossExponent, fingerprintPosition,
474                     radioSourcePosition, estimatedPosition, 0.0,
475                     0.0, new Matrix(2,2),
476                     new Matrix(2, 2), new Matrix(2, 2));
477 
478             assertEquals(dist.getMean()[0], rssi, ABSOLUTE_ERROR);
479             rssiVariance = dist.getCovariance().
480                     getElementAt(0, 0);
481             assertEquals(rssiVariance, 0.0, ABSOLUTE_ERROR);
482 
483 
484             assertNull(Utils.propagateVariancesToRssiVarianceFirstOrderNonLinear2D(
485                     fingerprintRssi, pathLossExponent, null, radioSourcePosition,
486                     estimatedPosition, null,
487                     null, null,
488                     null, null));
489             assertNull(Utils.propagateVariancesToRssiVarianceFirstOrderNonLinear2D(
490                     fingerprintRssi, pathLossExponent, fingerprintPosition, null,

492                     null, null,
493                     null));

496                     null, null,
497                     null, null,
498                     null, null));

1013     private double receivedPower(double equivalentTransmittedPower,
1014                                  double distance, double pathLossExponent) {
1015         //Pr = Pt*Gt*Gr*lambda^2/(4*pi*d)^2,    where Pr is the received power
1016         // lambda = c/f, where lambda is wavelength,
1017         // Pte = Pt*Gt*Gr, is the equivalent transmitted power, Gt is the transmitted Gain and Gr is the received Gain
1018         //Pr = Pte*c^2/((4*pi*f)^2 * d^2)
1019         double k = Math.pow(SPEED_OF_LIGHT / (4.0 * Math.PI * FREQUENCY), pathLossExponent);
1020         return equivalentTransmittedPower * k /
1021                 Math.pow(distance, pathLossExponent);
1022     }
