330     @Test
331     public void testSolveNoError() throws TrilaterationException, NotReadyException, LockedException {
332         UniformRandomizer randomizer = new UniformRandomizer(new Random());
333 
334         int numValid = 0;
335         for (int t = 0; t < TIMES; t++) {
336             int numCircles = randomizer.nextInt(MIN_CIRCLES, MAX_CIRCLES);
337 
338             InhomogeneousPoint2D position = new InhomogeneousPoint2D(
339                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
340                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
341             InhomogeneousPoint2D center;
342             double radius;
343             Circle[] circles = new Circle[numCircles];
344             for (int i = 0; i < numCircles; i++) {
345                 center = new InhomogeneousPoint2D(
346                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
347                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
348                 radius = center.distanceTo(position);
349                 circles[i] = new Circle(center, radius);
350             }
351 
352             LinearLeastSquaresTrilateration2DSolver solver = new LinearLeastSquaresTrilateration2DSolver(
353                     circles, this);
354 
355             reset();
356             assertEquals(solveStart, 0);
357             assertEquals(solveEnd, 0);
358             assertTrue(solver.isReady());
359             assertNull(solver.getEstimatedPosition());
360             assertNull(solver.getEstimatedPositionCoordinates());
361 
362             solver.solve();
363 
364             Point2D estimatedPosition = solver.getEstimatedPosition();
365             if (estimatedPosition.distanceTo(position) > ABSOLUTE_ERROR) {
366                 continue;
367             }
368             assertTrue(position.equals(estimatedPosition, ABSOLUTE_ERROR));
369 
370             numValid++;
371             break;
372         }
373 
374         assertTrue(numValid > 0);
375 
376         //Force NotReadyException
377         LinearLeastSquaresTrilateration2DSolver solver = new LinearLeastSquaresTrilateration2DSolver();
378         try {
379             solver.solve();
380             fail("NotReadyException expected but not thrown");
381         } catch (NotReadyException ignore) { }
382 
383         //Force TrilaterationException
384         Circle[] circles = new Circle[3];
385         InhomogeneousPoint2D center;
386         double radius;
387         for (int i = 0; i < 3; i++) {
388             center = new InhomogeneousPoint2D(0.0, 0.0);
389             radius = TrilaterationSolver.EPSILON;
390             circles[i] = new Circle(center, radius);
391         }
392         solver.setCircles(circles);
393         try {
394             solver.solve();
395             fail("TrilaterationException expected but not thrown");
396         } catch (TrilaterationException ignore) { }
397     }

399     @Test
400     public void testSolveWithError() throws TrilaterationException, NotReadyException, LockedException {
401         UniformRandomizer randomizer = new UniformRandomizer(new Random());
402 
403         int numValid = 0, numInvalid = 0;
404         for (int t = 0; t < TIMES; t++) {
405             int numPoints = randomizer.nextInt(MIN_CIRCLES, MAX_CIRCLES);
406 
407             InhomogeneousPoint2D position = new InhomogeneousPoint2D(
408                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
409                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
410             InhomogeneousPoint2D point;
411             double distance, error;
412             Point2D[] positions = new Point2D[numPoints];
413             double[] distances = new double[numPoints];
414             for (int i = 0; i < numPoints; i++) {
415                 point = new InhomogeneousPoint2D(
416                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
417                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
418                 distance = point.distanceTo(position);
419                 error = randomizer.nextDouble(MIN_DISTANCE_ERROR, MAX_DISTANCE_ERROR);
420                 positions[i] = point;
421                 distances[i] = distance + error;
422             }
423 
424             LinearLeastSquaresTrilateration2DSolver solver = new LinearLeastSquaresTrilateration2DSolver(
425                     positions, distances, this);
426 
427             reset();
428             assertEquals(solveStart, 0);
429             assertEquals(solveEnd, 0);
430             assertTrue(solver.isReady());
431             assertNull(solver.getEstimatedPosition());
432             assertNull(solver.getEstimatedPositionCoordinates());
433 
434             solver.solve();
435 
436             Point2D estimatedPosition = solver.getEstimatedPosition();
437             distance = estimatedPosition.distanceTo(position);
438             if (distance >= LARGE_ABSOLUTE_ERROR) {
439                 numInvalid++;
440             } else {
441                 numValid++;
442                 assertTrue(position.equals(estimatedPosition, LARGE_ABSOLUTE_ERROR));
443             }
444         }
445 
446         assertTrue(numValid > numInvalid);
447     }
